109 	/**
110 	 * Get the corresponding LocalChannelReference and set the remoteId if different
111 	 * 
112 	 * @param remoteId
113 	 * @param localId
114 	 * @return the LocalChannelReference
115 	 * @throws OpenR66ProtocolSystemException
116 	 */
117 	public LocalChannelReference getClient(Integer remoteId, Integer localId)
118 			throws OpenR66ProtocolSystemException {
119 		LocalChannelReference localChannelReference = getFromId(localId);
120 		if (localChannelReference != null) {
121 			if (localChannelReference.getRemoteId().compareTo(remoteId) != 0) {
122 				localChannelReference.setRemoteId(remoteId);
123 			}
124 			return localChannelReference;
125 		}
126 		throw new OpenR66ProtocolSystemException(
127 				"Cannot find LocalChannelReference");
128 	}

130 	/**
131 	 * Create a new Client
132 	 * 
133 	 * @param networkChannelReference
134 	 * @param remoteId might be set to ChannelUtils.NOCHANNEL (real creation)
135 	 * @param futureRequest might be null (from NetworkChannel Startup)
136 	 * @return the LocalChannelReference
137 	 * @throws OpenR66ProtocolSystemException
138 	 * @throws OpenR66ProtocolRemoteShutdownException 
139 	 * @throws OpenR66ProtocolNoConnectionException 
140 	 */
141 	public LocalChannelReference createNewClient(NetworkChannelReference networkChannelReference,
142 			Integer remoteId, R66Future futureRequest)
143 			throws OpenR66ProtocolSystemException, OpenR66ProtocolRemoteShutdownException, OpenR66ProtocolNoConnectionException {
144 		networkChannelReference.getLock().lock();
145 		try {
146 			ChannelFuture channelFuture = null;
147 			logger.debug("Status LocalChannelServer: {} {}", serverChannel
148 					.getClass().getName(), serverChannel.getConfig()
149 					.getConnectTimeoutMillis() + " " + serverChannel.isBound());
150 			for (int i = 0; i < Configuration.RETRYNB; i++) {
151 				if (R66ShutdownHook.isShutdownStarting()) {
152 					// Do not try since already locally in shutdown
153 					throw new OpenR66ProtocolNoConnectionException(
154 							"Cannot connect to local handler: " + socketLocalServerAddress +
155 									" " + serverChannel.isBound() + " " + serverChannel +
156 									" since the local server is in shutdown.");
157 				}
158 				channelFuture = clientBootstrap.connect(socketLocalServerAddress);
159 				try {
160 					channelFuture.await();
161 				} catch (InterruptedException e1) {
162 					logger.error("LocalChannelServer Interrupted: " +
163 							serverChannel.getClass().getName() + " " +
164 							serverChannel.getConfig().getConnectTimeoutMillis() +
165 							" " + serverChannel.isBound());
166 					throw new OpenR66ProtocolSystemException(
167 							"Interruption - Cannot connect to local handler: " +
168 									socketLocalServerAddress + " " +
169 									serverChannel.isBound() + " " + serverChannel,
170 							e1);
171 				}
172 				if (channelFuture.isSuccess()) {
173 					final Channel channel = channelFuture.getChannel();
174 					localChannelGroup.add(channel);
175 					logger.debug("Will start localChannelReference and eventually generate a new Db Connection if not-thread-safe");
176 					final LocalChannelReference localChannelReference = new LocalChannelReference(
177 							channel, networkChannelReference, remoteId, futureRequest);
178 					localChannelHashMap.put(channel.getId(), localChannelReference);
179 					logger.debug("Db connection done and Create LocalChannel entry: " + i + " {}",
180 							localChannelReference);
181 					logger.info("Add one localChannel to a Network Channel: "+channel.getId());
182 					// Now send first a Startup message
183 					StartupPacket startup = new StartupPacket(
184 							localChannelReference.getLocalId());
185 					try {
186 						Channels.write(channel, startup).await();
187 					} catch (InterruptedException e) {
188 						logger.error("Can't connect to local server due to interruption" + i);
189 						throw new OpenR66ProtocolSystemException(
190 								"Cannot connect to local handler", e);
191 					}
192 					return localChannelReference;
193 				} else {
194 					logger.error("Can't connect to local server " + i);
195 				}
196 				try {
197 					Thread.sleep(Configuration.RETRYINMS);
198 				} catch (InterruptedException e) {
199 					throw new OpenR66ProtocolSystemException(
200 							"Cannot connect to local handler", e);
201 				}
202 			}
203 			logger.error("LocalChannelServer: " +
204 					serverChannel.getClass().getName() + " " +
205 					serverChannel.getConfig().getConnectTimeoutMillis() + " " +
206 					serverChannel.isBound());
207 			throw new OpenR66ProtocolSystemException(
208 					"Cannot connect to local handler: " + socketLocalServerAddress +
209 							" " + serverChannel.isBound() + " " + serverChannel,
210 					channelFuture.getCause());
211 		} finally {
212 			networkChannelReference.getLock().unlock();
213 		}
214 	}

216 	/**
217 	 * 
218 	 * @param id
219 	 * @return the LocalChannelReference
220 	 */
221 	public LocalChannelReference getFromId(Integer id) {
222 		int maxtry = (int) (Configuration.configuration.TIMEOUTCON / Configuration.RETRYINMS) / 2;
223 		for (int i = 0; i < maxtry; i++) {
224 			LocalChannelReference lcr = localChannelHashMap.get(id);
225 			if (lcr == null) {
226 				try {
227 					Thread.sleep(Configuration.RETRYINMS);
228 					Thread.yield();
229 				} catch (InterruptedException e) {
230 				}
231 			} else {
232 				return lcr;
233 			}
234 		}
235 		return localChannelHashMap.get(id);
236 	}

296 	/**
297 	 * Informs all remote client that the server is shutting down
298 	 */
299 	public void shutdownLocalChannels() {
300 		logger.warn("Will inform LocalChannels of Shutdown: " + localChannelHashMap.size());
301 		Collection<LocalChannelReference> collection = localChannelHashMap.values();
302 		Iterator<LocalChannelReference> iterator = collection.iterator();
303 		ValidPacket packet = new ValidPacket("Shutdown forced", null,
304 				LocalPacketFactory.SHUTDOWNPACKET);
305 		ChannelBuffer buffer = null;
306 		while (iterator.hasNext()) {
307 			LocalChannelReference localChannelReference = iterator.next();
308 			logger.info("Inform Shutdown {}", localChannelReference);
309 			packet.setSmiddle(null);
310 			// If a transfer is running, save the current rank and inform remote
311 			// host
312 			if (localChannelReference.getSession() != null) {
313 				R66Session session = localChannelReference.getSession();
314 				DbTaskRunner runner = session.getRunner();
315 				if (runner != null && runner.isInTransfer()) {
316 					if (!runner.isSender()) {
317 						int newrank = runner.getRank();
318 						packet.setSmiddle(Integer.toString(newrank));
319 					}
320 					// Save File status
321 					try {
322 						runner.saveStatus();
323 					} catch (OpenR66RunnerErrorException e) {
324 					}
325 				}
326 				if (runner != null && ! runner.isFinished()) {
327 					R66Result result = new R66Result(
328 							new OpenR66ProtocolShutdownException(), session,
329 							true, ErrorCode.Shutdown, runner);
330 					result.other = packet;
331 					try {
332 						buffer = packet.getLocalPacket(localChannelReference);
333 					} catch (OpenR66ProtocolPacketException e1) {
334 					}
335 					localChannelReference.sessionNewState(R66FiniteDualStates.SHUTDOWN);
336 					NetworkPacket message = new NetworkPacket(
337 							localChannelReference.getLocalId(),
338 							localChannelReference.getRemoteId(),
339 							packet.getType(), buffer);
340 					try {
341 						Channels.write(localChannelReference.getNetworkChannel(),
342 								message).await();
343 					} catch (InterruptedException e1) {
344 					}
345 					try {
346 						session.setFinalizeTransfer(false, result);
347 					} catch (OpenR66RunnerErrorException e) {
348 					} catch (OpenR66ProtocolSystemException e) {
349 					}
350 				}
351 				Channels.close(localChannelReference.getLocalChannel());
352 				continue;
353 			}
354 			try {
355 				buffer = packet.getLocalPacket(localChannelReference);
356 			} catch (OpenR66ProtocolPacketException e1) {
357 			}
358 			NetworkPacket message = new NetworkPacket(
359 					localChannelReference.getLocalId(),
360 					localChannelReference.getRemoteId(), packet.getType(),
361 					buffer);
362 			Channels.write(localChannelReference.getNetworkChannel(), message);
363 		}
364 	}
