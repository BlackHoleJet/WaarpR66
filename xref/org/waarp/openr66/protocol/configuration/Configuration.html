565 	/**
566 	 * Configure the pipeline for client (to be called ony once)
567 	 */
568 	public void pipelineInit() {
569 		if (configured) {
570 			return;
571 		}
572 		localTransaction = new LocalTransaction();
573 		InternalLoggerFactory.setDefaultFactory(InternalLoggerFactory
574 				.getDefaultFactory());
575 		objectSizeEstimator = new NetworkPacketSizeEstimator();
576 		httpPipelineInit();
577 		logger.warn("Server Thread: " + SERVER_THREAD + " Client Thread: " + CLIENT_THREAD
578 				+ " Runner Thread: " + RUNNER_THREAD);
579 		serverPipelineExecutor = new OrderedMemoryAwareThreadPoolExecutor(
580 				CLIENT_THREAD, maxGlobalMemory / 10, maxGlobalMemory, 1000,
581 				TimeUnit.MILLISECONDS, objectSizeEstimator,
582 				new WaarpThreadFactory("ServerExecutor"));
583 		localPipelineExecutor = new OrderedMemoryAwareThreadPoolExecutor(
584 				CLIENT_THREAD * 100, maxGlobalMemory / 10, maxGlobalMemory,
585 				1000, TimeUnit.MILLISECONDS, new LocalPacketSizeEstimator(),
586 				new WaarpThreadFactory("LocalExecutor"));
587 		localClientPipelineExecutor = new OrderedMemoryAwareThreadPoolExecutor(
588 				CLIENT_THREAD * 100, maxGlobalMemory / 10, maxGlobalMemory,
589 				1000, TimeUnit.MILLISECONDS, new LocalPacketSizeEstimator(),
590 				new WaarpThreadFactory("LocalClientExecutor"));
591 		if (useLocalExec) {
592 			LocalExecClient.initialize();
593 		}
594 		configured = true;
595 	}

609 	public void serverStartup() throws WaarpDatabaseNoConnectionException,
610 			WaarpDatabaseSqlException {
611 		isServer = true;
612 		shutdownConfiguration.timeout = TIMEOUTCON;
613 		R66ShutdownHook.addShutdownHook();
614 		if ((!useNOSSL) && (!useSSL)) {
615 			logger.error("OpenR66 has neither NOSSL nor SSL support included! Stop here!");

617 		}
618 		pipelineInit();
619 		r66Startup();
620 		startHttpSupport();
621 		startMonitoring();
622 	}

624 	public void r66Startup() throws WaarpDatabaseNoConnectionException, WaarpDatabaseSqlException {
625 		logger.debug("Start R66: " + SERVER_PORT + ":" + useNOSSL + " " + SERVER_SSLPORT + ":"
626 				+ useSSL + ":" + HOST_SSLID);
627 		// add into configuration
628 		this.constraintLimitHandler.setServer(true);
629 		// Global Server
630 		serverChannelGroup = new DefaultChannelGroup("OpenR66");
631 
632 		serverChannelFactory = new NioServerSocketChannelFactory(
633 				execServerBoss, execServerWorker, SERVER_THREAD);
634 		if (useNOSSL) {
635 			serverBootstrap = new ServerBootstrap(serverChannelFactory);
636 			networkServerPipelineFactory = new NetworkServerPipelineFactory(true);
637 			serverBootstrap.setPipelineFactory(networkServerPipelineFactory);
638 			serverBootstrap.setOption("child.tcpNoDelay", true);
639 			serverBootstrap.setOption("child.keepAlive", true);
640 			serverBootstrap.setOption("child.reuseAddress", true);
641 			serverBootstrap.setOption("child.connectTimeoutMillis", TIMEOUTCON);
642 			serverBootstrap.setOption("tcpNoDelay", true);
643 			serverBootstrap.setOption("reuseAddress", true);
644 			serverBootstrap.setOption("connectTimeoutMillis", TIMEOUTCON);
645 
646 			serverChannelGroup.add(serverBootstrap.bind(new InetSocketAddress(
647 					SERVER_PORT)));
648 		} else {
649 			networkServerPipelineFactory = null;
650 			logger.warn("NOSSL mode is deactivated");
651 		}
652 
653 		if (useSSL && HOST_SSLID != null) {
654 			serverSslBootstrap = new ServerBootstrap(serverChannelFactory);
655 			networkSslServerPipelineFactory = new NetworkSslServerPipelineFactory(false,
656 					execServerWorker);
657 			serverSslBootstrap.setPipelineFactory(networkSslServerPipelineFactory);
658 			serverSslBootstrap.setOption("child.tcpNoDelay", true);
659 			serverSslBootstrap.setOption("child.keepAlive", true);
660 			serverSslBootstrap.setOption("child.reuseAddress", true);
661 			serverSslBootstrap.setOption("child.connectTimeoutMillis", TIMEOUTCON);
662 			serverSslBootstrap.setOption("tcpNoDelay", true);
663 			serverSslBootstrap.setOption("reuseAddress", true);
664 			serverSslBootstrap.setOption("connectTimeoutMillis", TIMEOUTCON);
665 
666 			serverChannelGroup.add(serverSslBootstrap.bind(new InetSocketAddress(
667 					SERVER_SSLPORT)));
668 		} else {
669 			networkSslServerPipelineFactory = null;
670 			logger.warn("SSL mode is desactivated");
671 		}
672 
673 		// Factory for TrafficShapingHandler
674 		globalTrafficShapingHandler = new GlobalTrafficHandler(
675 				objectSizeEstimator, timerTrafficCounter,
676 				serverGlobalWriteLimit, serverGlobalReadLimit, delayLimit);
677 		this.constraintLimitHandler.setHandler(globalTrafficShapingHandler);
678 
679 		// Now start the InternalRunner
680 		internalRunner = new InternalRunner();
681 		
682 		if (thriftport > 0) {
683 			thriftService = new R66ThriftServerService(new WaarpFuture(true), thriftport);
684 			execOtherWorker.execute(thriftService);
685 			thriftService.awaitInitialization();
686 		} else {
687 			thriftService = null;
688 		}
689 	}

691 	public void startHttpSupport() {
692 		// Now start the HTTP support
693 		httpChannelGroup = new DefaultChannelGroup("HttpOpenR66");
694 		// Configure the server.
695 		httpChannelFactory = new NioServerSocketChannelFactory(
696 				execServerBoss,
697 				execServerWorker,
698 				SERVER_THREAD);
699 		httpBootstrap = new ServerBootstrap(
700 				httpChannelFactory);
701 		// Set up the event pipeline factory.

867 	 * Reset the global monitor for bandwidth limitation and change future channel monitors
868 	 * 
869 	 * @param writeGlobalLimit
870 	 * @param readGlobalLimit
871 	 * @param writeSessionLimit
872 	 * @param readSessionLimit
873 	 * @param delayLimit
874 	 */
875 	public void changeNetworkLimit(long writeGlobalLimit, long readGlobalLimit,
876 			long writeSessionLimit, long readSessionLimit, long delayLimit) {
877 		long newWriteLimit = writeGlobalLimit > 1024 ? writeGlobalLimit
878 				: serverGlobalWriteLimit;
879 		if (writeGlobalLimit <= 0) {
880 			newWriteLimit = 0;
881 		}
882 		long newReadLimit = readGlobalLimit > 1024 ? readGlobalLimit
883 				: serverGlobalReadLimit;
884 		if (readGlobalLimit <= 0) {
885 			newReadLimit = 0;
886 		}
887 		serverGlobalReadLimit = newReadLimit;
888 		serverGlobalWriteLimit = newWriteLimit;
889 		this.delayLimit = delayLimit;
890 		if (globalTrafficShapingHandler != null) {
891 			globalTrafficShapingHandler.configure(serverGlobalWriteLimit, serverGlobalReadLimit,
892 					delayLimit);
893 			logger.warn("Bandwidth limits change: {}", globalTrafficShapingHandler);
894 		}
895 		newWriteLimit = writeSessionLimit > 1024 ? writeSessionLimit
896 				: serverChannelWriteLimit;
897 		if (writeSessionLimit <= 0) {
898 			newWriteLimit = 0;
899 		}
900 		newReadLimit = readSessionLimit > 1024 ? readSessionLimit
901 				: serverChannelReadLimit;
902 		if (readSessionLimit <= 0) {
903 			newReadLimit = 0;
904 		}
905 		serverChannelReadLimit = newReadLimit;
906 		serverChannelWriteLimit = newWriteLimit;
907 		anyBandwidthLimitation = (serverGlobalReadLimit > 0 || serverGlobalWriteLimit > 0 ||
908 				serverChannelReadLimit > 0 || serverChannelWriteLimit > 0);
909 	}
