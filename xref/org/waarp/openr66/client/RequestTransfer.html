95  	/**
96  	 * Parse the parameter and set current values
97  	 * 
98  	 * @param args
99  	 * @return True if all parameters were found and correct
100 	 */
101 	protected static boolean getParams(String[] args) {
102 		_INFO_ARGS = Messages.getString("RequestTransfer.0"); //$NON-NLS-1$
103 		if (args.length < 5) {
104 			logger
105 					.error(_INFO_ARGS);
106 			return false;
107 		}
108 		if (!FileBasedConfiguration
109 				.setClientConfigurationFromXml(Configuration.configuration, args[0])) {
110 			logger
111 					.error(Messages.getString("Configuration.NeedCorrectConfig")); //$NON-NLS-1$
112 			return false;
113 		}
114 		for (int i = 1; i < args.length; i++) {
115 			if (args[i].equalsIgnoreCase("-id")) {
116 				i++;
117 				try {
118 					sspecialId = Long.parseLong(args[i]);
119 				} catch (NumberFormatException e) {
120 					logger.error(Messages.getString("RequestTransfer.1")+args[i],e); //$NON-NLS-1$
121 					return false;
122 				}
123 			} else if (args[i].equalsIgnoreCase("-to")) {
124 				i++;
125 				srequested = args[i];
126 				try {
127 					srequester = Configuration.configuration.getHostId(DbConstant.admin.session,
128 							srequested);
129 				} catch (WaarpDatabaseException e) {
130 					logger.error(Messages.getString("RequestTransfer.5") + srequester, e); //$NON-NLS-1$
131 					return false;
132 				}
133 			} else if (args[i].equalsIgnoreCase("-from")) {
134 				i++;
135 				srequester = args[i];
136 				try {
137 					srequested = Configuration.configuration.getHostId(DbConstant.admin.session,
138 							srequester);
139 				} catch (WaarpDatabaseException e) {
140 					logger.error(Messages.getString("RequestTransfer.5") + srequested, e); //$NON-NLS-1$
141 					return false;
142 				}
143 			} else if (args[i].equalsIgnoreCase("-cancel")) {
144 				scancel = true;
145 			} else if (args[i].equalsIgnoreCase("-stop")) {
146 				sstop = true;
147 			} else if (args[i].equalsIgnoreCase("-restart")) {
148 				srestart = true;
149 			} else if (args[i].equalsIgnoreCase("-start")) {
150 				i++;
151 				srestarttime = args[i];
152 			} else if (args[i].equalsIgnoreCase("-delay")) {
153 				i++;
154 				if (args[i].charAt(0) == '+') {
155 					Date date = new Date(System.currentTimeMillis() +
156 							Long.parseLong(args[i].substring(1)));
157 					SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
158 					srestarttime = dateFormat.format(date);
159 				} else {
160 					Date date = new Date(Long.parseLong(args[i]));
161 					SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
162 					srestarttime = dateFormat.format(date);
163 				}
164 			}
165 		}
166 		if ((scancel && srestart) || (scancel && sstop) || (srestart && sstop)) {
167 			logger.error(Messages.getString("RequestTransfer.15")+_INFO_ARGS); //$NON-NLS-1$
168 			return false;
169 		}
170 		if (sspecialId == DbConstant.ILLEGALVALUE || srequested == null) {
171 			logger.error(Messages.getString("RequestTransfer.16")+_INFO_ARGS); //$NON-NLS-1$
172 			return false;
173 		}
174 
175 		return true;
176 	}

219 	public void run() {
220 		if (logger == null) {
221 			logger = WaarpInternalLoggerFactory.getLogger(RequestTransfer.class);
222 		}
223 		DbTaskRunner runner = null;
224 		try {
225 			runner = new DbTaskRunner(DbConstant.admin.session, null, null,
226 					specialId, requester, requested);
227 			logger.info("Found previous Runner: "+runner.toString());
228 		} catch (WaarpDatabaseException e) {
229 			// Maybe we can ask to the remote
230 			R66Future futureInfo = new R66Future(true);
231 			RequestInformation requestInformation = new RequestInformation(futureInfo, srequested, null, null, (byte) -1, specialId, true, networkTransaction);
232 			requestInformation.run();
233 			futureInfo.awaitUninterruptibly();
234 			if (futureInfo.isSuccess()) {
235 				R66Result r66result = futureInfo.getResult();
236 				ValidPacket info = (ValidPacket) r66result.other;
237 				String xml = info.getSheader();
238 				Document document;
239 				try {
240 					document = DocumentHelper.parseText(xml);
241 				} catch (DocumentException e1) {
242 					logger.error(Messages.getString("RequestTransfer.18")); //$NON-NLS-1$
243 					future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e1), null, true,
244 							ErrorCode.Internal, null));
245 					future.setFailure(e);
246 					return;
247 				}
248 				try {
249 					runner = DbTaskRunner.fromXml(DbConstant.admin.session, document.getRootElement(), true);
250 					logger.info("Get Runner from remote: "+runner.toString());
251 					if (runner.getSpecialId() == DbConstant.ILLEGALVALUE || ! runner.isSender()) {
252 						logger.error(Messages.getString("RequestTransfer.18")); //$NON-NLS-1$
253 						future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
254 								ErrorCode.Internal, null));
255 						future.setFailure(e);
256 						return;
257 					}
258 					if (runner.isAllDone()) {
259 						logger.error(Messages.getString("RequestTransfer.21")); //$NON-NLS-1$
260 						future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
261 								ErrorCode.Internal, null));
262 						future.setFailure(e);
263 						return;
264 					}
265 				} catch (OpenR66ProtocolBusinessException e1) {
266 					logger.error(Messages.getString("RequestTransfer.18")); //$NON-NLS-1$
267 					future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e1), null, true,
268 							ErrorCode.Internal, null));
269 					future.setFailure(e);
270 					return;
271 				}
272 			} else {
273 				logger.error(Messages.getString("RequestTransfer.18")); //$NON-NLS-1$
274 				future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
275 						ErrorCode.Internal, null));
276 				future.setFailure(e);
277 				return;
278 			}
279 		}
280 		if (cancel || stop || restart) {
281 			if (cancel) {
282 				// Cancel the task and delete any file if in retrieve
283 				if (runner.isAllDone()) {
284 					// nothing to do since already finished
285 					setDone(runner);
286 					logger.info("Transfer already finished: " + runner.toString());
287 					future.setResult(new R66Result(null, true, ErrorCode.TransferOk, runner));
288 					future.getResult().runner = runner;
289 					future.setSuccess();
290 					return;
291 				} else {
292 					// Send a request of cancel
293 					ErrorCode code = sendValid(runner, LocalPacketFactory.CANCELPACKET);
294 					switch (code) {
295 						case CompleteOk:
296 							logger.info("Transfer cancel requested and done: {}",
297 									runner);
298 							break;
299 						case TransferOk:
300 							logger.info("Transfer cancel requested but already finished: {}",
301 									runner);
302 							break;
303 						default:
304 							logger.info("Transfer cancel requested but internal error: {}",
305 									runner);
306 							break;
307 					}
308 				}
309 			} else if (stop) {
310 				// Just stop the task
311 				// Send a request
312 				ErrorCode code = sendValid(runner, LocalPacketFactory.STOPPACKET);
313 				switch (code) {
314 					case CompleteOk:
315 						logger.info("Transfer stop requested and done: {}", runner);
316 						break;
317 					case TransferOk:
318 						logger.info("Transfer stop requested but already finished: {}",
319 								runner);
320 						break;
321 					default:
322 						logger.info("Transfer stop requested but internal error: {}",
323 								runner);
324 						break;
325 				}
326 			} else if (restart) {
327 				// Restart if already stopped and not finished
328 				ErrorCode code = sendValid(runner, LocalPacketFactory.VALIDPACKET);
329 				switch (code) {
330 					case QueryStillRunning:
331 						logger.info(
332 								"Transfer restart requested but already active and running: {}",
333 								runner);
334 						break;
335 					case Running:
336 						logger.info("Transfer restart requested but already running: {}",
337 								runner);
338 						break;
339 					case PreProcessingOk:
340 						logger.info("Transfer restart requested and restarted: {}",
341 								runner);
342 						break;
343 					case CompleteOk:
344 						logger.info("Transfer restart requested but already finished: {}",
345 								runner);
346 						break;
347 					case RemoteError:
348 						logger.info("Transfer restart requested but remote error: {}",
349 								runner);
350 						break;
351 					case PassThroughMode:
352 						logger.info("Transfer not restarted since it is in PassThrough mode: {}",
353 								runner);
354 						break;
355 					default:
356 						logger.info("Transfer restart requested but internal error: {}",
357 								runner);
358 						break;
359 				}
360 			}
361 		} else {
362 			// Only request
363 			logger.info("Transfer information:     " + runner.toShortString());
364 			future.setResult(new R66Result(null, true, runner.getErrorInfo(), runner));
365 			future.setSuccess();
366 		}
367 	}

370 	 * Set the runner to DONE
371 	 * 
372 	 * @param runner
373 	 */
374 	private void setDone(DbTaskRunner runner) {
375 		if (runner.getUpdatedInfo() != UpdatedInfo.DONE) {
376 			runner.changeUpdatedInfo(UpdatedInfo.DONE);
377 			runner.forceSaveStatus();
378 		}

381 	private ErrorCode sendValid(DbTaskRunner runner, byte code) {
382 		DbHostAuth host;
383 		host = R66Auth.getServerAuth(DbConstant.admin.session,
384 				this.requester);
385 		if (host == null) {
386 			logger.error(Messages.getString("RequestTransfer.39") + this.requester); //$NON-NLS-1$
387 			OpenR66Exception e =
388 					new OpenR66RunnerErrorException("Requester host cannot be found");
389 			future.setResult(new R66Result(
390 					e,
391 					null, true,
392 					ErrorCode.TransferError, null));
393 			future.setFailure(e);
394 			return ErrorCode.Internal;
395 		}
396 		// check if requester is "client" so no connect from him but direct action
397 		logger.debug("Requester Host isClient: "+host.isClient());
398 		if (host.isClient()) {
399 			if (code == LocalPacketFactory.VALIDPACKET) {
400 				logger.warn(Messages.getString("RequestTransfer.42")+ //$NON-NLS-1$
401 						runner.toShortString());
402 				R66Future transfer = new R66Future(true);
403 				DirectTransfer transaction = new DirectTransfer(transfer,
404 						runner.getRequested(), runner.getOriginalFilename(), 
405 						runner.getRuleId(), runner.getFileInformation(), false, 
406 						runner.getBlocksize(), runner.getSpecialId(), networkTransaction);
407 				transaction.run();
408 				transfer.awaitUninterruptibly();
409 				logger.info("Request done with " + (transfer.isSuccess() ? "success" : "error"));
410 				if (transfer.isSuccess()) {
411 					future.setResult(new R66Result(null, true, ErrorCode.PreProcessingOk, runner));
412 					future.getResult().runner = runner;
413 					future.setSuccess();
414 					return ErrorCode.PreProcessingOk;
415 				} else {
416 					R66Result result = transfer.getResult();
417 					ErrorCode error = ErrorCode.Internal;
418 					if (result != null) {
419 						error = result.code;
420 					}
421 					OpenR66Exception e =
422 							new OpenR66RunnerErrorException("Transfer in direct mode failed: "+error.mesg);
423 					future.setFailure(e);
424 					return error;
425 				}
426 			} else {
427 				// get remote host instead
428 				host = R66Auth.getServerAuth(DbConstant.admin.session,
429 						this.requested);
430 				if (host == null) {
431 					logger.error(Messages.getString("Message.HostNotFound") + this.requested); //$NON-NLS-1$
432 					OpenR66Exception e =
433 							new OpenR66RunnerErrorException("Requested host cannot be found");
434 					future.setResult(new R66Result(
435 							e,
436 							null, true,
437 							ErrorCode.TransferError, null));
438 					future.setFailure(e);
439 					return ErrorCode.Internal;
440 				}
441 			}
442 		}
443 
444 		logger.info("Try RequestTransfer to "+host.toString());
445 		SocketAddress socketAddress = host.getSocketAddress();
446 		boolean isSSL = host.isSsl();
447 
448 		LocalChannelReference localChannelReference = networkTransaction
449 				.createConnectionWithRetry(socketAddress, isSSL, future);
450 		socketAddress = null;
451 		if (localChannelReference == null) {
452 			logger.debug("Cannot connect to " + host.toString());
453 			host = null;
454 			future.setResult(new R66Result(null, true,
455 					ErrorCode.ConnectionImpossible, null));
456 			future.cancel();
457 			return ErrorCode.Internal;
458 		}
459 		boolean useJson = PartnerConfiguration.useJson(host.getHostid());
460 		logger.debug("UseJson: "+useJson);
461 		AbstractLocalPacket packet = null;
462 		if (useJson) {
463 			ValidJsonPacket node = new ValidJsonPacket();
464 			node.setComment("Request on Transfer");
465 			node.setRequested(requested);
466 			node.setRequester(requester);
467 			node.setSpecialid(specialId);
468 			if (restarttime != null && code == LocalPacketFactory.VALIDPACKET) {
469 				// restart time set
470 				logger.debug("Restart with time: "+restarttime);
471 				// time to reschedule in yyyyMMddHHmmss format
472 				SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
473 				try {
474 					Date date = dateFormat.parse(restarttime);
475 					node.setRestarttime(date);
476 				} catch (ParseException e) {
477 				}
478 				packet = new JsonCommandPacket(node, code);
479 			} else {
480 				packet = new JsonCommandPacket(node, code);
481 			}
482 		} else {
483 			if (restarttime != null && code == LocalPacketFactory.VALIDPACKET) {
484 				// restart time set
485 				logger.debug("Restart with time: "+restarttime);
486 				packet = new ValidPacket("Request on Transfer",
487 						this.requested + " " + this.requester + " " + this.specialId+" "+restarttime,
488 						code);
489 			} else {
490 				packet = new ValidPacket("Request on Transfer",
491 					this.requested + " " + this.requester + " " + this.specialId,
492 					code);
493 			}
494 		}
495 		localChannelReference.sessionNewState(R66FiniteDualStates.VALIDOTHER);
496 		try {
497 			ChannelUtils.writeAbstractLocalPacket(localChannelReference, packet, false);
498 		} catch (OpenR66ProtocolPacketException e) {
499 			logger.error(Messages.getString("RequestTransfer.63") + host.toString()); //$NON-NLS-1$
500 			Channels.close(localChannelReference.getLocalChannel());
501 			localChannelReference = null;
502 			host = null;
503 			packet = null;
504 			logger.debug("Bad Protocol", e);
505 			future.setResult(new R66Result(e, null, true,
506 					ErrorCode.TransferError, null));
507 			future.setFailure(e);
508 			return ErrorCode.Internal;
509 		}
510 		packet = null;
511 		host = null;
512 		future.awaitUninterruptibly();
513 
514 		Channels.close(localChannelReference.getLocalChannel());
515 		localChannelReference = null;
516 
517 		logger.info("Request done with " + (future.isSuccess() ? "success" : "error"));
518 		R66Result result = future.getResult();
519 		if (result != null) {
520 			return result.code;
521 		}
522 		return ErrorCode.Internal;
523 	}

526 	 * @param args
527 	 */
528 	public static void main(String[] args) {
529 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
530 		if (logger == null) {
531 			logger = WaarpInternalLoggerFactory.getLogger(RequestTransfer.class);
532 		}
533 		if (!getParams(args)) {
534 			logger.error(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
535 			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
536 				DbConstant.admin.close();
537 			}
538 			ChannelUtils.stopLogger();
539 			System.exit(1);
540 		}
541 		int value = 99;
542 		try {
543 			Configuration.configuration.pipelineInit();
544 			NetworkTransaction networkTransaction = new NetworkTransaction();
545 			R66Future result = new R66Future(true);
546 			RequestTransfer requestTransfer =
547 					new RequestTransfer(result, sspecialId, srequested, srequester,
548 							scancel, sstop, srestart, srestarttime,
549 							networkTransaction);
550 			requestTransfer.run();
551 			result.awaitUninterruptibly();
552 			R66Result finalValue = result.getResult();
553 			if (scancel || sstop || srestart) {
554 				if (scancel) {
555 					if (result.isSuccess()) {
556 						value = 0;
557 						logger.warn(Messages.getString("RequestTransfer.21") + //$NON-NLS-1$
558 								finalValue.runner.toShortString());
559 					} else {
560 						switch (finalValue.code) {
561 							case CompleteOk:
562 								value = 0;
563 								logger.warn(Messages.getString("RequestTransfer.70") + //$NON-NLS-1$
564 										finalValue.runner.toShortString());
565 								break;
566 							case TransferOk:
567 								value = 3;
568 								logger.warn(Messages.getString("RequestTransfer.71") //$NON-NLS-1$
569 										+
570 										finalValue.runner.toShortString());
571 								break;
572 							default:
573 								value = 4;
574 								logger.error(Messages.getString("RequestTransfer.72") + //$NON-NLS-1$
575 										finalValue.runner.toShortString());
576 								break;
577 						}
578 					}
579 				} else if (sstop) {
580 					switch (finalValue.code) {
