18  package org.waarp.openr66.context.task;
19  
20  import java.io.File;
21  import java.io.FileOutputStream;
22  import java.io.IOException;
23  
24  import org.waarp.common.logging.WaarpInternalLogger;
25  import org.waarp.common.logging.WaarpInternalLoggerFactory;
26  import org.waarp.openr66.context.R66Session;
27  import org.waarp.openr66.context.filesystem.R66Dir;
28  import org.waarp.openr66.context.task.exception.OpenR66RunnerException;
29  
30  /**
31   * This task add 1 byte to empty file if the current file is empty (0 length).<br><br>
32   * 
33   * The task will be in error only if the file is of length 0 but cannot be unzeroed.<br>
34   * The content of PATH, if not empty, will be the content when unzeroed. If empty, the 'blank' character will be used.<br>
35   * 
36   * delay >= 1 will make a log using info level for 1, warn level for 2.
37   * @author Frederic Bregier
38   * 
39   */
40  public class UnzeroedFileTask extends AbstractTask {
41  	/**
42  	 * Internal Logger
43  	 */
44  	private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
45  			.getLogger(UnzeroedFileTask.class);
46  
47  	/**
48  	 * @param argRule
49  	 * @param delay
50  	 * @param argTransfer
51  	 * @param session
52  	 */
53  	public UnzeroedFileTask(String argRule, int delay, String argTransfer,
54  			R66Session session) {
55  		super(TaskType.UNZEROED, delay, argRule, argTransfer, session);
56  	}
57  
58  	@Override
59  	public void run() {
60  		File currentFile = session.getFile().getTrueFile();
61  		String toWrite = argRule.isEmpty() ? " " : argRule;
62  		String curpath = R66Dir.normalizePath(currentFile.getAbsolutePath());
63  		if (currentFile.exists() && currentFile.length() == 0) {
64  			FileOutputStream out = null;
65  			try {
66  				out = new FileOutputStream(currentFile);
67  				out.write(toWrite.getBytes());
68  				if (delay > 0) {
69  					if (delay > 1) {
70  						logger.warn("Unzeroed File: " + curpath + " from " +
71  							session.toString());
72  					} else {
73  						logger.info("Unzeroed File: " + curpath + " from " +
74  								session.toString());
75  					}
76  				}
77  				futureCompletion.setSuccess();
78  			} catch (IOException e) {
79  				logger.error("Cannot unzeroed File: " + curpath + " from " +
80  						session.toString());
81  				futureCompletion.setFailure(new OpenR66RunnerException("File not Unzeroed"));
82  			} finally {
83  				if (out != null) {
84  					try {
85  						out.close();
86  					} catch (IOException e) {
87  					}
88  				}
89  			}
90  			return;
91  		}
92  		if (delay > 0) {
93  			if (delay > 1) {
94  				logger.warn("Unzeroed File not applicable to " + curpath + " from " +
95  					session.toString());
96  			} else {
97  				logger.info("Unzeroed File not applicable to " + curpath + " from " +
98  						session.toString());
99  			}
100 		}
101 		futureCompletion.setSuccess();
102 	}
103 
104 }
