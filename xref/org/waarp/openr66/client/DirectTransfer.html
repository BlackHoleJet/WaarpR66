44  /**
45   * Direct Transfer from a client with or without database connection

47   * @author Frederic Bregier

50  public class DirectTransfer extends AbstractTransfer {
51  	protected final NetworkTransaction networkTransaction;
52  
53  	public DirectTransfer(R66Future future, String remoteHost,
54  			String filename, String rulename, String fileinfo, boolean isMD5, int blocksize,
55  			long id,
56  			NetworkTransaction networkTransaction) {
57  		// no starttime since it is direct (blocking request, no delay)
58  		super(DirectTransfer.class,
59  				future, filename, rulename, fileinfo, isMD5, remoteHost, blocksize, id, null);
60  		this.networkTransaction = networkTransaction;
61  	}

63  	/**
64  	 * Prior to call this method, the pipeline and NetworkTransaction must have been initialized. It
65  	 * is the responsibility of the caller to finish all network resources.
66  	 */
67  	public void run() {
68  		if (logger == null) {
69  			logger = WaarpInternalLoggerFactory.getLogger(DirectTransfer.class);
70  		}
71  		DbTaskRunner taskRunner = this.initRequest();
72  		if (taskRunner == null) {
73  			// already an error from there
74  			return;
75  		}
76  		ClientRunner runner = new ClientRunner(networkTransaction, taskRunner, future);
77  		OpenR66ProtocolNotYetConnectionException exc = null;
78  		for (int i = 0; i < Configuration.RETRYNB; i++) {
79  			try {
80  				runner.runTransfer();
81  				exc = null;
82  				break;
83  			} catch (OpenR66RunnerErrorException e) {
84  				logger.debug("Cannot Transfer", e);
85  				future.setResult(new R66Result(e, null, true,
86  						ErrorCode.Internal, taskRunner));
87  				future.setFailure(e);
88  				return;
89  			} catch (OpenR66ProtocolNoConnectionException e) {
90  				logger.debug("Cannot Connect", e);
91  				future.setResult(new R66Result(e, null, true,
92  						ErrorCode.ConnectionImpossible, taskRunner));
93  				// since no connection : just forget it
94  				if (nolog || taskRunner.shallIgnoreSave()) {
95  					try {
96  						taskRunner.delete();
97  					} catch (WaarpDatabaseException e1) {
98  					}
99  				}
100 				future.setFailure(e);
101 				return;
102 			} catch (OpenR66ProtocolPacketException e) {
103 				logger.debug("Bad Protocol", e);
104 				future.setResult(new R66Result(e, null, true,
105 						ErrorCode.TransferError, taskRunner));
106 				future.setFailure(e);
107 				return;
108 			} catch (OpenR66ProtocolNotYetConnectionException e) {
109 				logger.debug("Not Yet Connected", e);
110 				exc = e;
111 				continue;
112 			}
113 		}
114 		if (exc != null) {
115 			taskRunner.setLocalChannelReference(new LocalChannelReference());
116 			logger.debug("Cannot Connect", exc);
117 			future.setResult(new R66Result(exc, null, true,
118 					ErrorCode.ConnectionImpossible, taskRunner));
119 			// since no connection : just forget it
120 			if (nolog || taskRunner.shallIgnoreSave()) {
121 				try {
122 					taskRunner.delete();
123 				} catch (WaarpDatabaseException e1) {
124 				}
125 			}
126 			future.setFailure(exc);
127 			return;
128 		}
129 	}

131 	public static void main(String[] args) {
132 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
133 		if (logger == null) {
134 			logger = WaarpInternalLoggerFactory.getLogger(DirectTransfer.class);
135 		}
136 		if (!getParams(args, false)) {
137 			logger.error(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
138 			if (! OutputFormat.isQuiet()) {
139 				System.out.println(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
140 			}
141 			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
142 				DbConstant.admin.close();
143 			}
144 			ChannelUtils.stopLogger();
145 			System.exit(2);
146 		}
147 		long time1 = System.currentTimeMillis();
148 		R66Future future = new R66Future(true);
149 
150 		Configuration.configuration.pipelineInit();
151 		NetworkTransaction networkTransaction = new NetworkTransaction();
152 		try {
153 			DirectTransfer transaction = new DirectTransfer(future,
154 					rhost, localFilename, rule, fileInfo, ismd5, block, idt,
155 					networkTransaction);
156 			transaction.normalInfoAsWarn = snormalInfoAsWarn;
157 			logger.debug("rhost: "+rhost+":"+transaction.remoteHost);
158 			transaction.run();
159 			future.awaitUninterruptibly();
160 			long time2 = System.currentTimeMillis();
161 			logger.debug("finish transfer: " + future.isSuccess());
162 			long delay = time2 - time1;
163 			R66Result result = future.getResult();
164 			OutputFormat outputFormat = new OutputFormat(DirectTransfer.class.getSimpleName(), args);
165 			if (future.isSuccess()) {
166 				if (result.runner.getErrorInfo() == ErrorCode.Warning) {
167 					outputFormat.setValue(FIELDS.status.name(), 1);
168 					outputFormat.setValue(FIELDS.statusTxt.name(), Messages.getString("Transfer.Status")+Messages.getString("RequestInformation.Warned")); //$NON-NLS-1$
169 				} else {
170 					outputFormat.setValue(FIELDS.status.name(), 0);
171 					outputFormat.setValue(FIELDS.statusTxt.name(), Messages.getString("Transfer.Status")+Messages.getString("RequestInformation.Success")); //$NON-NLS-1$
172 				}
173 				outputFormat.setValue(FIELDS.remote.name(), rhost);
174 				Map<String, String> map = DbTaskRunner.getMapFromRunner(result.runner);
175 				outputFormat.setValueString(map);
176 				outputFormat.setValue("filefinal", (result.file != null ? result.file.toString() : "no file"));
177 				outputFormat.setValue("delay", delay);
178 				if (transaction.normalInfoAsWarn) {
179 					logger.warn(outputFormat.loggerOut());
180 				} else {
181 					logger.info(outputFormat.loggerOut());
182 				}
183 				if (! OutputFormat.isQuiet()) {
184 					outputFormat.sysout();
185 				}
186 				if (nolog || result.runner.shallIgnoreSave()) {
187 					// In case of success, delete the runner
188 					try {
189 						result.runner.delete();
190 					} catch (WaarpDatabaseException e) {
191 						logger.warn("Cannot apply nolog to     " + result.runner.toShortString(),
192 								e);
193 					}
194 				}
195 			} else {
196 				if (result == null || result.runner == null) {
197 					outputFormat.setValue(FIELDS.status.name(), 2);
198 					outputFormat.setValue(FIELDS.statusTxt.name(), Messages.getString("Transfer.FailedNoId")); //$NON-NLS-1$
199 					outputFormat.setValue(FIELDS.remote.name(), rhost);
200 					logger.error(outputFormat.loggerOut(), future.getCause());
201 					outputFormat.setValue(FIELDS.error.name(), future.getCause().getMessage());
202 					if (! OutputFormat.isQuiet()) {
203 						outputFormat.sysout();
204 					}
205 					networkTransaction.closeAll();
206 					System.exit(ErrorCode.Unknown.ordinal());
207 				}
208 				if (result.runner.getErrorInfo() == ErrorCode.Warning) {
209 					outputFormat.setValue(FIELDS.status.name(), 1);
210 					outputFormat.setValue(FIELDS.statusTxt.name(), Messages.getString("Transfer.Status")+Messages.getString("RequestInformation.Warned")); //$NON-NLS-1$
211 				} else {
212 					outputFormat.setValue(FIELDS.status.name(), 2);
213 					outputFormat.setValue(FIELDS.statusTxt.name(), Messages.getString("Transfer.Status")+Messages.getString("RequestInformation.Failure")); //$NON-NLS-1$
214 				}
215 				outputFormat.setValue(FIELDS.remote.name(), rhost);
216 				Map<String, String> map = DbTaskRunner.getMapFromRunner(result.runner);
217 				outputFormat.setValueString(map);
218 				if (result.runner.getErrorInfo() == ErrorCode.Warning) {
219 					logger.warn(outputFormat.loggerOut(), future.getCause());
220 				} else {
221 					logger.error(outputFormat.loggerOut(), future.getCause());
222 				}
223 				outputFormat.setValue(FIELDS.error.name(), future.getCause().getMessage());
224 				if (! OutputFormat.isQuiet()) {
225 					outputFormat.sysout();
226 				}
227 				networkTransaction.closeAll();
228 				System.exit(result.code.ordinal());
229 			}
230 		} catch (Throwable e) {
231 			logger.error("Exception", e);
232 		} finally {
233 			logger.debug("finish transfer: " + future.isDone() + ":" + future.isSuccess());
234 			networkTransaction.closeAll();
235 			// In case something wrong append
236 			if (future.isDone() && future.isSuccess()) {
237 				System.exit(0);
238 			} else {
239 				System.exit(66);
240 			}
241 		}
242 	}
