419 	public static void launchThreads() {
420         // init thread model
421         ExecutorService pool = Executors.newFixedThreadPool(NB);
422         HttpTestRestR66Client[] clients = new HttpTestRestR66Client[NB];
423         for (int i = 0; i < NB; i ++) {
424             clients[i] = new HttpTestRestR66Client();
425         }
426         for (int i = 0; i < NB; i ++) {
427             pool.execute(clients[i]);
428         }
429         pool.shutdown();
430         try {
431             while (! pool.awaitTermination(100000, TimeUnit.SECONDS)) ;
432         } catch (InterruptedException e) {
433         }
434     }
