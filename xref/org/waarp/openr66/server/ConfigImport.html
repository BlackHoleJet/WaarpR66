136 	 * Prior to call this method, the pipeline and NetworkTransaction must have been initialized. It
137 	 * is the responsibility of the caller to finish all network resources.

139 	public void run() {
140 		if (logger == null) {
141 			logger = WaarpInternalLoggerFactory.getLogger(ConfigImport.class);
142 		}
143 		SocketAddress socketAddress = dbhost.getSocketAddress();
144 		boolean isSSL = dbhost.isSsl();
145 
146 		LocalChannelReference localChannelReference = networkTransaction
147 				.createConnectionWithRetry(socketAddress, isSSL, future);
148 		socketAddress = null;
149 		if (localChannelReference == null) {
150 			dbhost = null;
151 			logger.error("Cannot Connect");
152 			future.setResult(new R66Result(
153 					new OpenR66ProtocolNoConnectionException("Cannot connect to server"),
154 					null, true, ErrorCode.Internal, null));
155 			future.setFailure(future.getResult().exception);
156 			return;
157 		}
158 		localChannelReference.sessionNewState(R66FiniteDualStates.VALIDOTHER);
159 		boolean useJson = PartnerConfiguration.useJson(dbhost.getHostid());
160 		logger.debug("UseJson: "+useJson);
161 		AbstractLocalPacket valid = null;
162 		if (useJson) {
163 			ObjectNode node = JsonHandler.createObjectNode();
164 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.host, host);
165 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.rule, rule);
166 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.business, business);
167 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.alias, alias);
168 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.roles, role);
169 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.purgehost, hostPurge);
170 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.purgerule, rulePurge);
171 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.purgebusiness, businessPurge);
172 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.purgealias, aliasPurge);
173 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.purgeroles, rolePurge);
174 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.hostid, hostid);
175 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.ruleid, ruleid);
176 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.businessid, businessid);
177 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.aliasid, aliasid);
178 			JsonHandler.setValue(node, JsonCommandPacket.CONFIMPORTPACKET.rolesid, roleid);
179 			valid = new JsonCommandPacket(node, LocalPacketFactory.CONFIMPORTPACKET);
180 		} else {
181 			valid = new ValidPacket((hostPurge ? "1 " : "0 ") + host,
182 				(rulePurge ? "1 " : "0 ") + rule,
183 				LocalPacketFactory.CONFIMPORTPACKET);
184 		}
185 		try {
186 			ChannelUtils.writeAbstractLocalPacket(localChannelReference, valid, false);
187 		} catch (OpenR66ProtocolPacketException e) {
188 			logger.error("Bad Protocol", e);
189 			Channels.close(localChannelReference.getLocalChannel());
190 			localChannelReference = null;
191 			dbhost = null;
192 			valid = null;
193 			future.setResult(new R66Result(e, null, true,
194 					ErrorCode.TransferError, null));
195 			future.setFailure(e);
196 			return;
197 		}
198 		dbhost = null;
199 		future.awaitUninterruptibly();
200 		logger.debug("Request done with " + (future.isSuccess() ? "success" : "error"));
201 		Channels.close(localChannelReference.getLocalChannel());
202 		localChannelReference = null;
203 	}

365 	public static void main(String[] args) {
366 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
367 		if (logger == null) {
368 			logger = WaarpInternalLoggerFactory.getLogger(ConfigImport.class);
369 		}
370 		if (!getParams(args)) {
371 			logger.error("Wrong initialization");
372 			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
373 				DbConstant.admin.close();
374 			}
375 			System.exit(1);
376 		}
377 		long time1 = System.currentTimeMillis();
378 		R66Future future = new R66Future(true);
379 
380 		Configuration.configuration.pipelineInit();
381 		NetworkTransaction networkTransaction = new NetworkTransaction();
382 		try {
383 			ConfigImport transaction = new ConfigImport(future, shostpurge, srulepurge, 
384 					sbusinesspurge, saliaspurge, srolepurge, shost, srule, 
385 					sbusiness, salias, srole, networkTransaction);
386 			transaction.setSpecialIds(lhost, lrule, lbusiness, lalias, lrole);
387 			if (stohost != null) {
388 				try {
389 					transaction.setHost(new DbHostAuth(DbConstant.admin.session, stohost));
390 				} catch (WaarpDatabaseException e) {
391 					logger.error("ConfigImport in     FAILURE since Host is not found: "+stohost, e);
392 					networkTransaction.closeAll();
393 					System.exit(10);
394 				}
395 			} else {
396 				stohost = Configuration.configuration.HOST_SSLID;
397 			}
398 			transaction.run();
399 			future.awaitUninterruptibly();
400 			long time2 = System.currentTimeMillis();
401 			long delay = time2 - time1;
402 			R66Result result = future.getResult();
403 			if (future.isSuccess()) {
404 				boolean useJson = PartnerConfiguration.useJson(stohost);
405 				logger.debug("UseJson: "+useJson);
406 				String message = null;
407 				if (useJson) {
408 					message = (result.other != null ? ((JsonCommandPacket) result.other).getRequest() :
409 							"no file");
410 				} else {
411 					message = (result.other != null ? ((ValidPacket) result.other).getSheader() :
412 							"no file");
413 				}
414 				if (result.code == ErrorCode.Warning) {
415 					logger.warn("WARNED on import:     " +
416 							message
417 							+ "     delay: " + delay);
418 				} else {
419 					logger.warn("SUCCESS on import:     " +
420 							message
421 							+ "     delay: " + delay);
422 				}
423 			} else {
424 				if (result.code == ErrorCode.Warning) {
425 					logger.warn("ConfigImport is     WARNED", future.getCause());
426 					networkTransaction.closeAll();
427 					System.exit(result.code.ordinal());
428 				} else {
429 					logger.error("ConfigImport in     FAILURE", future.getCause());
430 					networkTransaction.closeAll();
431 					System.exit(result.code.ordinal());
432 				}
433 			}
434 		} finally {
435 			networkTransaction.closeAll();
436 		}
437 	}
