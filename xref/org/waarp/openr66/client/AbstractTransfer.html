100 	/**
101 	 * Initiate the Request and return a potential DbTaskRunner
102 	 * 
103 	 * @return null if an error occurs or a DbTaskRunner
104 	 */
105 	protected DbTaskRunner initRequest() {
106 		DbRule rule;
107 		try {
108 			rule = new DbRule(DbConstant.admin.session, rulename);
109 		} catch (WaarpDatabaseException e) {
110 			logger.error("Cannot get Rule: " + rulename, e);
111 			future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
112 					ErrorCode.Internal, null));
113 			future.setFailure(e);
114 			return null;
115 		}
116 		int mode = rule.mode;
117 		if (isMD5) {
118 			mode = RequestPacket.getModeMD5(mode);
119 		}
120 		DbTaskRunner taskRunner = null;
121 		if (id != DbConstant.ILLEGALVALUE) {
122 			try {
123 				taskRunner = new DbTaskRunner(DbConstant.admin.session, id,
124 						remoteHost);
125 			} catch (WaarpDatabaseException e) {
126 				logger.error("Cannot get task", e);
127 				future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
128 						ErrorCode.QueryRemotelyUnknown, null));
129 				future.setFailure(e);
130 				return null;
131 			}
132 			if (fileinfo != null && !fileinfo.equals(NOINFO)) {
133 				taskRunner.setFileInformation(fileinfo);
134 			}
135 			if (startTime != null) {
136 				taskRunner.setStart(startTime);
137 			}
138 		} else {
139 			RequestPacket request = new RequestPacket(rulename,
140 					mode, filename, blocksize, 0,
141 					id, fileinfo);
142 			// Not isRecv since it is the requester, so send => isRetrieve is true
143 			boolean isRetrieve = !RequestPacket.isRecvMode(request.getMode());
144 			try {
145 				taskRunner =
146 						new DbTaskRunner(DbConstant.admin.session, rule, isRetrieve, request,
147 								remoteHost, startTime);
148 			} catch (WaarpDatabaseException e) {
149 				logger.error("Cannot get task", e);
150 				future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
151 						ErrorCode.Internal, null));
152 				future.setFailure(e);
153 				return null;
154 			}
155 		}
156 		return taskRunner;
157 	}

172 	/**
173 	 * Parse the parameter and set current values
174 	 * 
175 	 * @param args
176 	 * @param submitOnly
177 	 *            True if the client is only a submitter (through database)
178 	 * @return True if all parameters were found and correct
179 	 */
180 	protected static boolean getParams(String[] args, boolean submitOnly) {
181 		if (args.length < 2) {
182 			logger
183 					.error("Needs at least 3 or 4 arguments:\n"
184 							+
185 							"  the XML client configuration file,\n"
186 							+
187 							"  '-to' the remoteHost Id,\n"
188 							+
189 							"  '-file' the file to transfer,\n"
190 							+
191 							"  '-rule' the rule\n"
192 							+
193 							"Or\n"
194 							+
195 							"  '-to' the remoteHost Id,\n"
196 							+
197 							"  '-id' \"Id of a previous transfer\",\n"
198 							+
199 							"Other options:\n"
200 							+
201 							"  '-info' \"information to send\",\n"
202 							+
203 							"  '-md5' to force MD5 by packet control,\n"
204 							+
205 							"  '-block' size of packet > 1K (prefered is 64K),\n"
206 							+
207 							"  '-nolog' to not log locally this action\n"
208 							+
209 							"  '-start' \"time start\" as yyyyMMddHHmmss (override previous -delay options)\n"
210 							+
211 							"  '-delay' \"+delay in ms\" as delay in ms from current time(override previous -start options)\n"
212 							+
213 							"  '-delay' \"delay in ms\" as time in ms (override previous -start options)");
214 			return false;
215 		}
216 		if (submitOnly) {
217 			if (!FileBasedConfiguration
218 					.setSubmitClientConfigurationFromXml(Configuration.configuration, args[0])) {
219 				logger
220 						.error("Needs a correct configuration file as first argument");
221 				return false;
222 			}
223 		} else if (!FileBasedConfiguration
224 				.setClientConfigurationFromXml(Configuration.configuration, args[0])) {
225 			logger
226 					.error("Needs a correct configuration file as first argument");
227 			return false;
228 		}
229 		// Now set default values from configuration
230 		block = Configuration.configuration.BLOCKSIZE;
231 		int i = 1;
232 		try {
233 			for (i = 1; i < args.length; i++) {
234 				if (args[i].equalsIgnoreCase("-to")) {
235 					i++;
236 					rhost = args[i];
237 				} else if (args[i].equalsIgnoreCase("-file")) {
238 					i++;
239 					localFilename = args[i];
240 				} else if (args[i].equalsIgnoreCase("-rule")) {
241 					i++;
242 					rule = args[i];
243 				} else if (args[i].equalsIgnoreCase("-info")) {
244 					i++;
245 					fileInfo = args[i];
246 				} else if (args[i].equalsIgnoreCase("-md5")) {
247 					ismd5 = true;
248 				} else if (args[i].equalsIgnoreCase("-block")) {
249 					i++;
250 					block = Integer.parseInt(args[i]);
251 					if (block < 100) {
252 						logger.error("Block size is too small: " + block);
253 						return false;
254 					}
255 				} else if (args[i].equalsIgnoreCase("-nolog")) {
256 					nolog = true;
257 					i++;
258 				} else if (args[i].equalsIgnoreCase("-id")) {
259 					i++;
260 					idt = Long.parseLong(args[i]);
261 				} else if (args[i].equalsIgnoreCase("-start")) {
262 					i++;
263 					Date date;
264 					try {
265 						date = dateFormat.parse(args[i]);
266 						ttimestart = new Timestamp(date.getTime());
267 					} catch (ParseException e) {
268 					}
269 				} else if (args[i].equalsIgnoreCase("-delay")) {
270 					i++;
271 					if (args[i].charAt(0) == '+') {
272 						ttimestart = new Timestamp(System.currentTimeMillis() +
273 								Long.parseLong(args[i].substring(1)));
274 					} else {
275 						ttimestart = new Timestamp(Long.parseLong(args[i]));
276 					}
277 				}
278 			}
279 		} catch (NumberFormatException e) {
280 			logger.error("Number Format exception at Rank "+i);
281 			return false;
282 		}
283 		if (fileInfo == null) {
284 			fileInfo = NOINFO;
285 		}
286 		if (rhost != null && rule != null && localFilename != null) {
287 			return true;
288 		} else if (idt != DbConstant.ILLEGALVALUE && rhost != null) {
289 			try {
290 				DbTaskRunner runner = new DbTaskRunner(DbConstant.admin.session, idt,
291 						rhost);
292 				rule = runner.getRuleId();
293 				localFilename = runner.getOriginalFilename();
294 				return true;
295 			} catch (WaarpDatabaseException e) {
296 				logger.error(
297 						"All params are not correctly set! Need at least (-to -rule and -file)" +
298 								" or (-to and -id) params", e);
299 				return false;
300 			}
301 
302 		}
303 		logger.error("All params are not set! Need at least (-to -rule and -file)" +
304 				" or (-to and -id) params");
305 		return false;
306 	}
