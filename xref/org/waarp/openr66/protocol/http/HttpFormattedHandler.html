182 	private String readFileHeader(String filename) {
183 		String value;
184 		try {
185 			value = WaarpStringUtils.readFileException(filename);
186 		} catch (InvalidArgumentException e) {
187 			logger.error("Error while trying to open: " + filename, e);
188 			return "";
189 		} catch (FileTransferException e) {
190 			logger.error("Error while trying to read: " + filename, e);
191 			return "";
192 		}
193 		StringBuilder builder = new StringBuilder(value);
194 
195 		WaarpStringUtils.replace(builder, REPLACEMENT.XXXDATEXXX.toString(),
196 				(new Date()).toString());
197 		WaarpStringUtils.replace(builder, REPLACEMENT.XXXLOCACTIVEXXX.toString(),
198 				Integer.toString(
199 						Configuration.configuration.getLocalTransaction().
200 								getNumberLocalChannel()));
201 		WaarpStringUtils.replace(builder, REPLACEMENT.XXXNETACTIVEXXX.toString(),
202 				Integer.toString(
203 						DbAdmin.getNbConnection()));
204 		WaarpStringUtils.replace(builder, REPLACEMENT.XXXHOSTIDXXX.toString(),
205 				Configuration.configuration.HOST_ID);
206 		TrafficCounter trafficCounter =
207 				Configuration.configuration.getGlobalTrafficShapingHandler().getTrafficCounter();
208 		WaarpStringUtils.replace(builder, REPLACEMENT.XXXBANDWIDTHXXX.toString(),
209 				"IN:" + (trafficCounter.getLastReadThroughput() / 131072) +
210 						"Mbits&nbsp;&nbsp;OUT:" +
211 						(trafficCounter.getLastWriteThroughput() / 131072) + "Mbits");
212 		WaarpStringUtils.replace(builder, REPLACEMENT.XXXLANGXXX.toString(), lang);
213 		return builder.toString();
214 	}

216 	private String getTrimValue(String varname) {
217 		String value = null;
218 		try {
219 			value = params.get(varname).get(0).trim();
220 		} catch (NullPointerException e) {
221 			return null;
222 		}
223 		if (value.isEmpty()) {
224 			value = null;
225 		}
226 		return value;
227 	}

229 	@Override
230 	public void messageReceived(ChannelHandlerContext ctx, MessageEvent e)
231 			throws Exception {
232 		isCurrentRequestXml = false;
233 		status = HttpResponseStatus.OK;
234 		HttpRequest request = this.request = (HttpRequest) e.getMessage();
235 		QueryStringDecoder queryStringDecoder = new QueryStringDecoder(request
236 				.getUri());
237 		uriRequest = queryStringDecoder.getPath();
238 		logger.debug("Msg: " + uriRequest);
239 		if (uriRequest.contains("gre/") || uriRequest.contains("img/") ||
240 				uriRequest.contains("res/") || uriRequest.contains("favicon.ico")) {
241 			HttpWriteCacheEnable.writeFile(request,
242 					e.getChannel(), Configuration.configuration.httpBasePath + uriRequest,
243 					"XYZR66NOSESSION");
244 			return;
245 		}
246 		/*try {
247 			if (DbConstant.admin.isConnected) {
248 				this.dbSession = new DbSession(DbConstant.admin, false);
249 				DbAdmin.nbHttpSession++;
250 				this.isPrivateDbSession = true;
251 			}
252 		} catch (WaarpDatabaseNoConnectionException e1) {
253 			// Cannot connect so use default connection
254 			logger.warn("Use default database connection");
255 			this.dbSession = DbConstant.admin.session;
256 		}*/
257 		try {
258 			char cval = 'z';
259 			long nb = LIMITROW;
260 			// check the URI
261 			if (uriRequest.equalsIgnoreCase("/active")) {
262 				cval = '0';
263 			} else if (uriRequest.equalsIgnoreCase("/error")) {
264 				cval = '1';
265 			} else if (uriRequest.equalsIgnoreCase("/done")) {
266 				cval = '2';
267 			} else if (uriRequest.equalsIgnoreCase("/all")) {
268 				cval = '3';
269 			} else if (uriRequest.equalsIgnoreCase("/status")) {
270 				cval = '4';
271 			} else if (uriRequest.equalsIgnoreCase("/statusxml")) {
272 				cval = '5';
273 				nb = 0; // since it could be the default or setup by request
274 				isCurrentRequestXml = true;
275 			} else if (uriRequest.toLowerCase().startsWith("/spooled")) {
276 				cval = '6';
277 			}
278 			// Get the params according to get or post
279 			if (request.getMethod() == HttpMethod.GET) {
280 				params = queryStringDecoder.getParameters();
281 			} else if (request.getMethod() == HttpMethod.POST) {
282 				ChannelBuffer content = request.getContent();
283 				if (content.readable()) {
284 					String param = content.toString(WaarpStringUtils.UTF8);
285 					queryStringDecoder = new QueryStringDecoder("/?" + param);
286 				} else {
287 					responseContent.append(REQUEST.index.readFileUnique(this));
288 					writeResponse(e);
289 					return;
290 				}
291 				params = queryStringDecoder.getParameters();
292 			}
293 			boolean getMenu = (cval == 'z');
294 			boolean extraBoolean = false;
295 			if (!params.isEmpty()) {
296 				// if not uri, from get or post
297 				if (getMenu) {
298 					String info = getTrimValue(sINFO);
299 					if (info != null) {
300 						getMenu = false;
301 						cval = info.charAt(0);
302 					} else {
303 						getMenu = true;
304 					}
305 				}
306 				// search the nb param
307 				String snb = getTrimValue(sNB);
308 				if (snb != null) {
309 					try {
310 						nb = Long.parseLong(snb);
311 					} catch (Exception e1) {
312 					}
313 				}
314 				// search the detail param
315 				String sdetail = getTrimValue(sDETAIL);
316 				if (sdetail != null) {
317 					try {
318 						if (Integer.parseInt(sdetail) > 0) {
319 							extraBoolean = true;
320 						}
321 					} catch (Exception e1) {
322 					}
323 				}
324 				String langarg = getTrimValue("setLng");
325 				if (langarg != null && ! langarg.isEmpty()) {
326 					lang = langarg;
327 				}
328 			}
329 			if (getMenu) {
330 				responseContent.append(REQUEST.index.readFileUnique(this));
331 			} else {
332 				// Use value 0=Active 1=Error 2=Done 3=All
333 				switch (cval) {
334 					case '0':
335 						active(ctx, (int) nb);
336 						break;
337 					case '1':
338 						error(ctx, (int) nb);
339 						break;
340 					case '2':
341 						done(ctx, (int) nb);
342 						break;
343 					case '3':
344 						all(ctx, (int) nb);
345 						break;
346 					case '4':
347 						status(ctx, (int) nb);
348 						break;
349 					case '5':
350 						statusxml(ctx, nb, extraBoolean);
351 						break;
352 					case '6':
353 						String name = null;
354 						if (params.containsKey("name")) {
355 							name = getTrimValue("name");
356 						}
357 						int istatus = 0;
358 						if (params.containsKey("status")) {
359 							String status = getTrimValue("status");
360 							try {
361 								istatus = Integer.parseInt(status);
362 							} catch (NumberFormatException e1) {
363 								istatus = 0;
364 							}
365 						}
366 						if (uriRequest.toLowerCase().startsWith("/spooleddetail")) {
367 							extraBoolean = true;
368 						}
369 						spooled(ctx, extraBoolean, name, istatus);
370 						break;
371 					default:
372 						responseContent.append(REQUEST.index.readFileUnique(this));
373 				}
374 			}
375 			writeResponse(e);
376 		} finally {
377 			if (this.isPrivateDbSession && dbSession != null) {
378 				dbSession.forceDisconnect();
379 				DbAdmin.nbHttpSession--;
380 				dbSession = null;
381 			}
382 		}
383 	}

385 	/**
386 	 * Add all runners from preparedStatement for type
387 	 * 
388 	 * @param preparedStatement
389 	 * @param type
390 	 * @param nb
391 	 * @throws WaarpDatabaseNoConnectionException
392 	 * @throws WaarpDatabaseSqlException
393 	 */
394 	private void addRunners(DbPreparedStatement preparedStatement, String type,
395 			int nb) throws WaarpDatabaseNoConnectionException,
396 			WaarpDatabaseSqlException {
397 		try {
398 			preparedStatement.executeQuery();
399 			responseContent
400 					.append("<style>td{font-size: 8pt;}</style><table border=\"2\">");
401 			responseContent.append("<tr><td>");
402 			responseContent.append(type);
403 			responseContent.append("</td>");
404 			responseContent.append(DbTaskRunner.headerHtml());
405 			responseContent.append("</tr>\r\n");
406 			int i = 0;
407 			while (preparedStatement.getNext()) {
408 				DbTaskRunner taskRunner = DbTaskRunner
409 						.getFromStatement(preparedStatement);
410 				responseContent.append("<tr><td>");
411 				responseContent.append(taskRunner.isSender() ? "S" : "R");
412 				responseContent.append("</td>");
413 				LocalChannelReference lcr =
414 						Configuration.configuration.getLocalTransaction().
415 								getFromRequest(taskRunner.getKey());
416 				responseContent.append(taskRunner.toHtml(authentHttp,
417 						lcr != null ? Messages.getString("HttpSslHandler.Active") : Messages.getString("HttpSslHandler.NotActive")));
418 				responseContent.append("</tr>\r\n");
419 				if (nb > 0) {
420 					i++;
421 					if (i >= nb) {
422 						break;
423 					}
424 				}
425 			}
426 			responseContent.append("</table><br>\r\n");
427 		} finally {
428 			if (preparedStatement != null) {
429 				preparedStatement.realClose();
430 			}
431 		}
432 	}

434 	/**
435 	 * print all active transfers
436 	 * 
437 	 * @param ctx
438 	 * @param nb
439 	 */
440 	private void active(ChannelHandlerContext ctx, int nb) {
441 		responseContent.append(REQUEST.active.readHeader(this));
442 		DbPreparedStatement preparedStatement = null;
443 		try {
444 			preparedStatement = DbTaskRunner.getStatusPrepareStatement(dbSession,
445 					ErrorCode.Running, nb);
446 			addRunners(preparedStatement, ErrorCode.Running.mesg, nb);
447 			preparedStatement = DbTaskRunner.getSelectFromInfoPrepareStatement(
448 					dbSession, UpdatedInfo.INTERRUPTED, true, nb);
449 			DbTaskRunner.finishSelectOrCountPrepareStatement(preparedStatement);
450 			addRunners(preparedStatement, UpdatedInfo.INTERRUPTED.name(), nb);
451 			preparedStatement = DbTaskRunner.getSelectFromInfoPrepareStatement(
452 					dbSession, UpdatedInfo.TOSUBMIT, true, nb);
453 			DbTaskRunner.finishSelectOrCountPrepareStatement(preparedStatement);
454 			addRunners(preparedStatement, UpdatedInfo.TOSUBMIT.name(), nb);
455 			preparedStatement = DbTaskRunner.getStatusPrepareStatement(dbSession,
456 					ErrorCode.InitOk, nb);
457 			addRunners(preparedStatement, ErrorCode.InitOk.mesg, nb);
458 			preparedStatement = DbTaskRunner.getStatusPrepareStatement(dbSession,
459 					ErrorCode.PreProcessingOk, nb);
460 			addRunners(preparedStatement, ErrorCode.PreProcessingOk.mesg, nb);
461 			preparedStatement = DbTaskRunner.getStatusPrepareStatement(dbSession,
462 					ErrorCode.TransferOk, nb);
463 			addRunners(preparedStatement, ErrorCode.TransferOk.mesg, nb);
464 			preparedStatement = DbTaskRunner.getStatusPrepareStatement(dbSession,
465 					ErrorCode.PostProcessingOk, nb);
466 			addRunners(preparedStatement, ErrorCode.PostProcessingOk.mesg, nb);
467 			preparedStatement = null;
468 		} catch (WaarpDatabaseException e) {
469 			if (preparedStatement != null) {
470 				preparedStatement.realClose();
471 			}
472 			logger.warn("OpenR66 Web Error {}", e.getMessage());
473 			sendError(ctx, HttpResponseStatus.SERVICE_UNAVAILABLE);
474 			return;
475 		}
476 		responseContent.append(REQUEST.active.readEnd());
477 	}

560 	/**
561 	 * print only status
562 	 * 
563 	 * @param ctx
564 	 * @param nb
565 	 */
566 	private void status(ChannelHandlerContext ctx, int nb) {
567 		responseContent.append(REQUEST.status.readHeader(this));
568 		DbPreparedStatement preparedStatement = null;
569 		try {
570 			preparedStatement = DbTaskRunner.getSelectFromInfoPrepareStatement(
571 					dbSession, UpdatedInfo.INERROR, true, 1);
572 			DbTaskRunner.finishSelectOrCountPrepareStatement(preparedStatement);
573 			try {
574 				preparedStatement.executeQuery();
575 				if (preparedStatement.getNext()) {
576 					responseContent.append("<p>Some Transfers are in ERROR</p><br>");
577 					status = HttpResponseStatus.INTERNAL_SERVER_ERROR;
578 				}
579 			} finally {
580 				if (preparedStatement != null) {
581 					preparedStatement.realClose();
582 				}
583 			}
584 			preparedStatement = DbTaskRunner.getSelectFromInfoPrepareStatement(
585 					dbSession, UpdatedInfo.INTERRUPTED, true, 1);
586 			DbTaskRunner.finishSelectOrCountPrepareStatement(preparedStatement);
587 			try {
588 				preparedStatement.executeQuery();
589 				if (preparedStatement.getNext()) {
590 					responseContent.append("<p>Some Transfers are INTERRUPTED</p><br>");
591 					status = HttpResponseStatus.INTERNAL_SERVER_ERROR;
592 				}
593 			} finally {
594 				if (preparedStatement != null) {
595 					preparedStatement.realClose();
596 				}
597 			}
598 			preparedStatement = DbTaskRunner.getStepPrepareStatement(dbSession,
599 					TASKSTEP.ERRORTASK, 1);
600 			try {
