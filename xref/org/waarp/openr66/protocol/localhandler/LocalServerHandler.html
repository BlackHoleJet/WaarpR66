117 	/*
118 	 * (non-Javadoc)
119 	 * @see org.jboss.netty.channel.SimpleChannelHandler#messageReceived(org.jboss
120 	 * .netty.channel.ChannelHandlerContext, org.jboss.netty.channel.MessageEvent)
121 	 */
122 	@Override
123 	public void messageReceived(ChannelHandlerContext ctx, MessageEvent e)
124 			throws OpenR66Exception {
125 		// action as requested and answer if necessary
126 		final AbstractLocalPacket packet = (AbstractLocalPacket) e.getMessage();
127 		if (packet.getType() == LocalPacketFactory.STARTUPPACKET) {
128 			serverHandler.startup(e.getChannel(), (StartupPacket) packet);
129 		} else {
130 			if (serverHandler.getLocalChannelReference() == null) {
131 				logger.error("No LocalChannelReference at " +
132 						packet.getClass().getName());
133 				serverHandler.getSession().newState(ERROR);
134 				final ErrorPacket errorPacket = new ErrorPacket(
135 						"No LocalChannelReference at " +
136 								packet.getClass().getName(),
137 						ErrorCode.ConnectionImpossible.getCode(),
138 						ErrorPacket.FORWARDCLOSECODE);
139 				try {
140 					Channels.write(e.getChannel(), errorPacket).await();
141 				} catch (InterruptedException e1) {
142 				}
143 				ChannelUtils.close(e.getChannel());
144 				if (Configuration.configuration.r66Mib != null) {
145 					Configuration.configuration.r66Mib.notifyWarning(
146 							"No LocalChannelReference", packet.getClass().getSimpleName());
147 				}
148 				return;
149 			}
150 			switch (packet.getType()) {
151 				case LocalPacketFactory.AUTHENTPACKET: {
152 					serverHandler.authent(e.getChannel(), (AuthentPacket) packet);
153 					break;
154 				}
155 				// Already done case LocalPacketFactory.STARTUPPACKET:
156 				case LocalPacketFactory.DATAPACKET: {
157 					if (((DataPacket) packet).getPacketRank() % 100 == 1 || serverHandler.getSession().getState() != R66FiniteDualStates.DATAR) {
158 						serverHandler.getSession().newState(DATAR);
159 						logger.debug("DATA RANK: " + ((DataPacket) packet).getPacketRank() + " : " +
160 								serverHandler.getSession().getRunner().getRank());
161 					}
162 					serverHandler.data(e.getChannel(), (DataPacket) packet);
163 					break;
164 				}
165 				case LocalPacketFactory.VALIDPACKET: {
166 					// SHUTDOWNPACKET does not need authentication
167 					if (((ValidPacket) packet).getTypeValid() != LocalPacketFactory.SHUTDOWNPACKET &&
168 							(!serverHandler.getSession().isAuthenticated())) {
169 						logger.warn("Valid packet received while not authenticated: {} {}", packet, serverHandler.getSession());
170 						serverHandler.getSession().newState(ERROR);
171 						throw new OpenR66ProtocolNotAuthenticatedException(
172 								"Not authenticated while Valid received");
173 					}
174 					if (((ValidPacket) packet).getTypeValid() == LocalPacketFactory.REQUESTPACKET) {
175 						String [] fields = ((ValidPacket) packet).getSmiddle().split(PartnerConfiguration.BAR_SEPARATOR_FIELD);
176 						String newfilename = fields[0];
177 						// potential file size changed
178 						long newSize = -1;
179 						if (fields.length > 1) {
180 							try {
181 								newSize = Long.parseLong(fields[fields.length-1]);
182 							} catch (NumberFormatException e2) {
183 								newfilename += PartnerConfiguration.BAR_SEPARATOR_FIELD + fields[fields.length-1];
184 								newSize = -1;
185 							}
186 						}
187 						serverHandler.requestChangeNameSize(e.getChannel(), newfilename, newSize);
188 					} else {
189 						serverHandler.valid(e.getChannel(), (ValidPacket) packet);
190 					}
191 					break;
192 				}
193 				case LocalPacketFactory.ERRORPACKET: {
194 					serverHandler.getSession().newState(ERROR);
195 					serverHandler.errorMesg(e.getChannel(), (ErrorPacket) packet);
196 					break;
197 				}
198 				case LocalPacketFactory.CONNECTERRORPACKET: {
199 					serverHandler.connectionError(e.getChannel(),
200 							(ConnectionErrorPacket) packet);
201 					break;
202 				}
203 				case LocalPacketFactory.REQUESTPACKET: {
204 					serverHandler.request(e.getChannel(), (RequestPacket) packet);
205 					break;
206 				}
207 				case LocalPacketFactory.SHUTDOWNPACKET: {
208 					serverHandler.getSession().newState(SHUTDOWN);
209 					serverHandler.shutdown(e.getChannel(), (ShutdownPacket) packet);
210 					break;
211 				}
212 				case LocalPacketFactory.STOPPACKET:
213 				case LocalPacketFactory.CANCELPACKET:
214 				case LocalPacketFactory.CONFIMPORTPACKET:
215 				case LocalPacketFactory.CONFEXPORTPACKET:
216 				case LocalPacketFactory.BANDWIDTHPACKET: {
217 					logger.error("Unimplemented Mesg: " +
218 							packet.getClass().getName());
219 					serverHandler.getSession().newState(ERROR);
220 					serverHandler.getLocalChannelReference().invalidateRequest(new R66Result(
221 							new OpenR66ProtocolSystemException(
222 									"Not implemented"), serverHandler.getSession(), true,
223 							ErrorCode.Unimplemented, null));
224 					final ErrorPacket errorPacket = new ErrorPacket(
225 							"Unimplemented Mesg: " +
226 									packet.getClass().getName(),
227 							ErrorCode.Unimplemented.getCode(),
228 							ErrorPacket.FORWARDCLOSECODE);
229 					ChannelUtils.writeAbstractLocalPacket(serverHandler.getLocalChannelReference(), errorPacket, true);
230 					ChannelUtils.close(e.getChannel());
231 					break;
232 				}
233 				case LocalPacketFactory.TESTPACKET: {
234 					serverHandler.getSession().newState(TEST);
235 					serverHandler.test(e.getChannel(), (TestPacket) packet);
236 					break;
237 				}
238 				case LocalPacketFactory.ENDTRANSFERPACKET: {
239 					serverHandler.endTransfer(e.getChannel(), (EndTransferPacket) packet);
240 					break;
241 				}
242 				case LocalPacketFactory.INFORMATIONPACKET: {
243 					serverHandler.getSession().newState(INFORMATION);
244 					serverHandler.information(e.getChannel(), (InformationPacket) packet);
245 					break;
246 				}
247 				case LocalPacketFactory.ENDREQUESTPACKET: {
248 					serverHandler.endRequest(e.getChannel(), (EndRequestPacket) packet);
249 					break;
250 				}
251 				case LocalPacketFactory.BUSINESSREQUESTPACKET: {
252 					serverHandler.businessRequest(e.getChannel(), (BusinessRequestPacket) packet);
253 					break;
254 				}
255 				case LocalPacketFactory.BLOCKREQUESTPACKET: {
256 					serverHandler.blockRequest(e.getChannel(), (BlockRequestPacket) packet);
257 					break;
258 				}
259 				case LocalPacketFactory.JSONREQUESTPACKET: {
260 					if (!serverHandler.getSession().isAuthenticated()) {
261 						logger.warn("JsonCommand packet received while not authenticated: {} {}", packet, serverHandler.getSession());
262 						serverHandler.getSession().newState(ERROR);
263 						throw new OpenR66ProtocolNotAuthenticatedException(
264 								"Not authenticated while Valid received");
265 					}
266 					JsonPacket json = ((JsonCommandPacket) packet).getJsonRequest();
267 					if (json == null) {
268 						ErrorCode code = ErrorCode.CommandNotFound;
269 						R66Result resulttest = new R66Result(serverHandler.getSession(), true,
270 								code, serverHandler.getSession().getRunner());
271 						json = new JsonPacket();
272 						json.setComment("Invalid command");
273 						json.setRequestUserPacket(((JsonCommandPacket) packet).getTypeValid());
274 						JsonCommandPacket valid = new JsonCommandPacket(json, resulttest.code.getCode(),
275 								LocalPacketFactory.REQUESTUSERPACKET);
276 						resulttest.other = packet;
277 						serverHandler.getLocalChannelReference().validateRequest(resulttest);
278 						try {
279 							ChannelUtils.writeAbstractLocalPacket(serverHandler.getLocalChannelReference(),
280 									valid, true);
281 						} catch (OpenR66ProtocolPacketException e2) {
282 						}
283 						serverHandler.getSession().setStatus(99);
284 						Channels.close(e.getChannel());
285 						return;
286 					}
287 					json.setRequestUserPacket(((JsonCommandPacket) packet).getTypeValid());
288 					if (((JsonCommandPacket) packet).getTypeValid() == LocalPacketFactory.REQUESTPACKET) {
289 						RequestJsonPacket node = (RequestJsonPacket) json;
290 						String newfilename = node.getFilename();
291 						if (newfilename == null) {
292 							// error so ignore
293 							return;
294 						}
295 						long newSize = node.getFilesize();
296 						logger.debug("NewSize "+ newSize + " NewName "+newfilename);
297 						// potential file size changed
298 						serverHandler.requestChangeNameSize(e.getChannel(), newfilename, newSize);
299 					} else {
300 						serverHandler.jsonCommand(e.getChannel(), (JsonCommandPacket) packet);
301 					}
302 					break;
303 				}
304 				default: {
305 					logger
306 							.error("Unknown Mesg: " +
307 									packet.getClass().getName());
308 					serverHandler.getSession().newState(ERROR);
309 					serverHandler.getLocalChannelReference().invalidateRequest(new R66Result(
310 							new OpenR66ProtocolSystemException(
311 									"Unknown Message"), serverHandler.getSession(), true,
312 							ErrorCode.Unimplemented, null));
313 					final ErrorPacket errorPacket = new ErrorPacket(
314 							"Unkown Mesg: " + packet.getClass().getName(),
315 							ErrorCode.Unimplemented.getCode(), ErrorPacket.FORWARDCLOSECODE);
316 					ChannelUtils.writeAbstractLocalPacket(serverHandler.getLocalChannelReference(), errorPacket, true);
317 					ChannelUtils.close(e.getChannel());
318 				}
319 			}
320 		}
321 	}

323 	/*
324 	 * (non-Javadoc)
325 	 * @see org.jboss.netty.channel.SimpleChannelHandler#exceptionCaught(org.jboss
326 	 * .netty.channel.ChannelHandlerContext, org.jboss.netty.channel.ExceptionEvent)
327 	 */
328 	@Override
329 	public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) {
330 		// inform clients
331 		logger.debug("Exception and isFinished: "+
332 				(serverHandler.getLocalChannelReference() != null && serverHandler.getLocalChannelReference().getFutureRequest().isDone()), e.getCause());
333 		if (serverHandler.getLocalChannelReference() != null && serverHandler.getLocalChannelReference().getFutureRequest().isDone()) {
334 			Channels.close(e.getChannel());
335 			return;
336 		}
337 		OpenR66Exception exception = OpenR66ExceptionTrappedFactory
338 				.getExceptionFromTrappedException(e.getChannel(), e);
339 		ErrorCode code = null;
340 		if (exception != null) {
341 			serverHandler.getSession().newState(ERROR);
342 			boolean isAnswered = false;
343 			if (exception instanceof OpenR66ProtocolShutdownException) {
344 				R66ShutdownHook.shutdownWillStart();
345 				logger.warn(Messages.getString("LocalServerHandler.0") + //$NON-NLS-1$
346 						serverHandler.getSession().getAuth().getUser());
347 				if (serverHandler.getLocalChannelReference() != null) {
348 					R66Result finalValue = new R66Result(exception, serverHandler.getSession(), true,
349 							ErrorCode.Shutdown, serverHandler.getSession().getRunner());
350 					try {
351 						serverHandler.tryFinalizeRequest(finalValue);
352 					} catch (OpenR66RunnerErrorException e2) {
353 					} catch (OpenR66ProtocolSystemException e2) {
354 					}
355 					if (!serverHandler.getLocalChannelReference().getFutureRequest().isDone()) {
356 						try {
357 							serverHandler.getSession().setFinalizeTransfer(false, finalValue);
358 						} catch (OpenR66RunnerErrorException e1) {
359 							serverHandler.getLocalChannelReference().invalidateRequest(finalValue);
360 						} catch (OpenR66ProtocolSystemException e1) {
361 							serverHandler.getLocalChannelReference().invalidateRequest(finalValue);
362 						}
363 					}
364 				}
365 				// dont'close, thread will do
366 				ChannelUtils.startShutdown();
367 				// set global shutdown info and before close, send a valid
368 				// shutdown to all
369 				serverHandler.getSession().setStatus(54);
370 				return;
371 			} else {
372 				if (serverHandler.getLocalChannelReference() != null
373 						&& serverHandler.getLocalChannelReference().getFutureRequest() != null) {
374 					if (serverHandler.getLocalChannelReference().getFutureRequest().isDone()) {
375 						R66Result result = serverHandler.getLocalChannelReference().getFutureRequest()
376 								.getResult();
377 						if (result != null) {
378 							isAnswered = result.isAnswered;
379 						}
380 					}
381 				}
382 				if (exception instanceof OpenR66ProtocolNoConnectionException) {
383 					code = ErrorCode.ConnectionImpossible;
384 					DbTaskRunner runner = serverHandler.getSession().getRunner();
385 					if (runner != null) {
386 						runner.stopOrCancelRunner(code);
387 					}
388 				} else if (exception instanceof OpenR66ProtocolBusinessCancelException) {
389 					code = ErrorCode.CanceledTransfer;
390 					DbTaskRunner runner = serverHandler.getSession().getRunner();
391 					if (runner != null) {
392 						runner.stopOrCancelRunner(code);
393 					}
394 				} else if (exception instanceof OpenR66ProtocolBusinessStopException) {
395 					code = ErrorCode.StoppedTransfer;
396 					DbTaskRunner runner = serverHandler.getSession().getRunner();
397 					if (runner != null) {
398 						runner.stopOrCancelRunner(code);
399 					}
400 				} else if (exception instanceof OpenR66ProtocolBusinessQueryAlreadyFinishedException) {
401 					code = ErrorCode.QueryAlreadyFinished;
402 					try {
403 						serverHandler.tryFinalizeRequest(new R66Result(serverHandler.getSession(), true, code, serverHandler.getSession().getRunner()));
404 						ChannelCloseTimer.closeFutureChannel(e.getChannel());
405 						return;
406 					} catch (OpenR66RunnerErrorException e1) {
407 					} catch (OpenR66ProtocolSystemException e1) {
408 					}
409 				} else if (exception instanceof OpenR66ProtocolBusinessQueryStillRunningException) {
410 					code = ErrorCode.QueryStillRunning;
411 					// nothing is to be done
412 					logger.error("Will close channel since ", exception);
413 					Channels.close(e.getChannel());
414 					serverHandler.getSession().setStatus(56);
415 					return;
416 				} else if (exception instanceof OpenR66ProtocolBusinessRemoteFileNotFoundException) {
417 					code = ErrorCode.FileNotFound;
418 				} else if (exception instanceof OpenR66RunnerException) {
419 					code = ErrorCode.ExternalOp;
420 				} else if (exception instanceof OpenR66RunnerErrorException) {
421 					code = ErrorCode.ExternalOp;
422 				} else if (exception instanceof OpenR66ProtocolNotAuthenticatedException) {
423 					code = ErrorCode.BadAuthent;
424 				} else if (exception instanceof OpenR66ProtocolNetworkException) {
425 					code = ErrorCode.Disconnection;
426 					DbTaskRunner runner = serverHandler.getSession().getRunner();
427 					if (runner != null) {
428 						R66Result finalValue = new R66Result(
429 								new OpenR66ProtocolSystemException(
430 										Messages.getString("LocalServerHandler.2")), //$NON-NLS-1$
431 								serverHandler.getSession(), true, code, serverHandler.getSession().getRunner());
432 						try {
433 							serverHandler.tryFinalizeRequest(finalValue);
434 						} catch (OpenR66Exception e2) {
435 						}
436 					}
437 				} else if (exception instanceof OpenR66ProtocolRemoteShutdownException) {
438 					code = ErrorCode.RemoteShutdown;
439 					DbTaskRunner runner = serverHandler.getSession().getRunner();
440 					if (runner != null) {
441 						runner.stopOrCancelRunner(code);
442 					}
443 				} else {
444 					DbTaskRunner runner = serverHandler.getSession().getRunner();
445 					if (runner != null) {
446 						switch (runner.getErrorInfo()) {
447 							case InitOk:
448 							case PostProcessingOk:
449 							case PreProcessingOk:
450 							case Running:
451 							case TransferOk:
452 								code = ErrorCode.Internal;
453 								break;
454 							default:
455 								code = runner.getErrorInfo();
456 						}
457 					} else {
458 						code = ErrorCode.Internal;
459 					}
460 				}
461 				if ((!isAnswered) &&
462 						(!(exception instanceof OpenR66ProtocolBusinessNoWriteBackException)) &&
463 						(!(exception instanceof OpenR66ProtocolNoConnectionException))) {
464 					if (code == null || code == ErrorCode.Internal) {
465 						code = ErrorCode.RemoteError;
466 					}
467 					final ErrorPacket errorPacket = new ErrorPacket(exception
468 							.getMessage(),
469 							code.getCode(), ErrorPacket.FORWARDCLOSECODE);
470 					try {
471 						if (serverHandler.getLocalChannelReference() != null) {
472 							ChannelUtils.writeAbstractLocalPacket(serverHandler.getLocalChannelReference(),
473 								errorPacket, true);
474 						}
475 					} catch (OpenR66ProtocolPacketException e1) {
476 						// should not be
477 					}
478 				}
479 				R66Result finalValue =
480 						new R66Result(
481 								exception, serverHandler.getSession(), true, code, serverHandler.getSession().getRunner());
482 				try {
483 					serverHandler.getSession().setFinalizeTransfer(false, finalValue);
484 					if (serverHandler.getLocalChannelReference() != null) {
485 						serverHandler.getLocalChannelReference().invalidateRequest(finalValue);
486 					}
487 				} catch (OpenR66RunnerErrorException e1) {
488 					if (serverHandler.getLocalChannelReference() != null)
489 						serverHandler.getLocalChannelReference().invalidateRequest(finalValue);
490 				} catch (OpenR66ProtocolSystemException e1) {
491 					if (serverHandler.getLocalChannelReference() != null)
492 						serverHandler.getLocalChannelReference().invalidateRequest(finalValue);
493 				}
494 			}
495 			if (exception instanceof OpenR66ProtocolBusinessNoWriteBackException) {
496 				logger.error("Will close channel {}", exception.getMessage());
497 				Channels.close(e.getChannel());
498 				serverHandler.getSession().setStatus(56);
499 				return;
500 			} else if (exception instanceof OpenR66ProtocolNoConnectionException) {
501 				logger.error("Will close channel {}", exception.getMessage());
502 				Channels.close(e.getChannel());
503 				serverHandler.getSession().setStatus(57);
504 				return;
505 			}
506 			serverHandler.getSession().setStatus(58);
507 			ChannelCloseTimer.closeFutureChannel(e.getChannel());
508 		} else {
