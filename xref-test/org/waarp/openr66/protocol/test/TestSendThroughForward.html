48  /**
49   * <b>WARNING: This class is not functional neither integrated</b><br>
50   * 
51   * Test class for Send Through to another R66 Server as forward<br>
52   * Only a subpart of SenThroughClient is to be made since steps 1-2 and steps 7-8 are only for
53   * client, not for server.
54   * 
55   * 3) Prepare the request of transfer:<br>
56   * <tt>     R66Future futureReq = new R66Future(true);</tt><br>
57   * <tt>     TestSendThroughForward transaction = new TestSendThroughForward(futureReq,...);</tt><br>
58   * <tt>     if (! transaction.initiateRequest()) { error }</tt><br>
59   * <br>
60   * 4) Once initiateRequest() gives true, you are ready to send the data in through mode using the
61   * TestRecvThroughForwardHandler:<br>
62   * <br>
63   * 5) Once you have finished, so this is the last block, you have to do the following:<br>
64   * If the last block is not empty:<br>
65   * <tt>     DataBlock block = transaction.transformToDataBlock(data);</tt><br>
66   * <tt>     block.setEOF(true);</tt><br>
67   * <tt>     futureWrite = transaction.writeWhenPossible(block);</tt><br>
68   * <tt>     futureWrite.awaitUninterruptibly();</tt><br>
69   * <br>
70   * If the last block is empty, it is already handled by TestRecvThroughForwardHandler<br>
71   * <br>
72   * 6) If everything is in success:<br>
73   * <tt>     transaction.finalizeRequest();</tt><br>
74   * <br>
75   * And now wait for the transfer to finish:<br>
76   * <tt>     futureReq.awaitUninterruptibly();</tt><br>
77   * <tt>     R66Result result = futureReq.getResult();</tt><br>
78   * <br>
79   * 
80   * @author Frederic Bregier
81   * 
82   */
83  public class TestSendThroughForward extends SendThroughClient {

87  	protected DbTaskRunner sourceRunner;

89  	public static class TestRecvThroughForwardHandler extends RecvThroughHandler {
90  
91  		protected TestSendThroughForward client;
92  
93  		/*
94  		 * (non-Javadoc)
95  		 * @see
96  		 * org.waarp.openr66.client.RecvThroughHandler#writeChannelBuffer(org.jboss.netty.buffer
97  		 * .ChannelBuffer)
98  		 */
99  		@Override
100 		public void writeChannelBuffer(ChannelBuffer buffer)
101 				throws OpenR66ProtocolBusinessException {
102 			DataBlock block = new DataBlock();
103 			if (buffer.readableBytes() <= 0) {
104 				// last block
105 				block.setEOF(true);
106 			} else {
107 				block.setBlock(buffer);
108 			}
109 			try {
110 				client.writeWhenPossible(block).await();
111 			} catch (OpenR66RunnerErrorException e) {
112 				client.transferInError(e);
113 			} catch (OpenR66ProtocolPacketException e) {
114 				client.transferInError(e);
115 			} catch (OpenR66ProtocolSystemException e) {
116 				client.transferInError(e);
117 			} catch (InterruptedException e) {
118 				client.transferInError(new OpenR66ProtocolSystemException(e));
119 			}
120 			if (block.isEOF()) {
121 				client.finalizeRequest();
122 				client.foundEOF = true;
123 			}
124 		}

126 	}

155 	/*
156 	 * (non-Javadoc)
157 	 * @see org.waarp.openr66.client.SendThroughClient#initiateRequest()
158 	 */
159 	@Override
160 	public boolean initiateRequest() {
161 		if (logger == null) {
162 			logger = WaarpInternalLoggerFactory.getLogger(TestSendThroughForward.class);
163 		}
164 		DbRule rule;
165 		try {
166 			rule = new DbRule(DbConstant.admin.session, rulename);
167 		} catch (WaarpDatabaseException e) {
168 			logger.error("Cannot get Rule: " + rulename, e);
169 			future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
170 					ErrorCode.Internal, null));
171 			future.setFailure(e);
172 			return false;
173 		}
174 		int mode = rule.mode;
175 		if (isMD5) {
176 			mode = RequestPacket.getModeMD5(mode);
177 		}
178 		String sep = Configuration.getSeparator(remoteHost);
179 		RequestPacket request = new RequestPacket(rulename,
180 				mode, filename, blocksize, sourceRunner.getRank(),
181 				id, fileinfo, -1, sep);
182 		// Not isRecv since it is the requester, so send => isSender is true
183 		boolean isSender = true;
184 		try {
185 			try {
186 				// no delay
187 				taskRunner =
188 						new DbTaskRunner(DbConstant.admin.session, rule, isSender, request,
189 								remoteHost, null);
190 			} catch (WaarpDatabaseException e) {
191 				logger.error("Cannot get task", e);
192 				future.setResult(new R66Result(new OpenR66DatabaseGlobalException(e), null, true,
193 						ErrorCode.Internal, null));
194 				future.setFailure(e);
195 				return false;
196 			}
197 			ClientRunner runner = new ClientRunner(networkTransaction, taskRunner, future);
198 			runner.setRecvThroughHandler(handler);
199 			runner.setSendThroughMode();
200 			OpenR66ProtocolNotYetConnectionException exc = null;
201 			for (int i = 0; i < Configuration.RETRYNB; i++) {
202 				try {
203 					localChannelReference = runner.initRequest();
204 					exc = null;
205 					break;
206 				} catch (OpenR66RunnerErrorException e) {
207 					logger.error("Cannot Transfer", e);
208 					future.setResult(new R66Result(e, null, true,
209 							ErrorCode.Internal, taskRunner));
210 					future.setFailure(e);
211 					return false;
212 				} catch (OpenR66ProtocolNoConnectionException e) {
213 					logger.error("Cannot Connect", e);
214 					future.setResult(new R66Result(e, null, true,
215 							ErrorCode.ConnectionImpossible, taskRunner));
216 					future.setFailure(e);
217 					return false;
218 				} catch (OpenR66ProtocolPacketException e) {
219 					logger.error("Bad Protocol", e);
220 					future.setResult(new R66Result(e, null, true,
221 							ErrorCode.TransferError, taskRunner));
222 					future.setFailure(e);
223 					return false;
224 				} catch (OpenR66ProtocolNotYetConnectionException e) {
225 					logger.debug("Not Yet Connected", e);
226 					exc = e;
227 					continue;
228 				}
229 			}
230 			if (exc != null) {
231 				taskRunner.setLocalChannelReference(new LocalChannelReference());
232 				logger.error("Cannot Connect", exc);
233 				future.setResult(new R66Result(exc, null, true,
234 						ErrorCode.ConnectionImpossible, taskRunner));
235 				future.setFailure(exc);
236 				return false;
237 			}
238 			try {
239 				localChannelReference.waitReadyForSendThrough();
240 			} catch (OpenR66Exception e) {
241 				logger.error("Cannot Transfer", e);
242 				future.setResult(new R66Result(e, null, true,
243 						ErrorCode.Internal, taskRunner));
244 				future.setFailure(e);
245 				return false;
246 			}
247 			if (taskRunner.getRank() < sourceRunner.getRank()) {
248 				sourceRunner.setRankAtStartup(taskRunner.getRank());
249 			}
250 			// now start the send from external data
251 			return true;
252 		} finally {
253 			if (taskRunner != null) {
254 				// not delete but sourceRunner and taskRunner should be stopped
255 				// and taskRunner not allowed to be restarted alone
256 				if (future.isFailed()) {
257 					taskRunner.changeUpdatedInfo(UpdatedInfo.INERROR);
258 					taskRunner.forceSaveStatus();
259 				}
260 			}
261 		}
262 	}
