40  /**
41   * Client to submit a transfer for multiple files to multiple hosts at once.<br>
42   * Files will have to be separated by ','.<br>
43   * Hosts will have to be separated by ','.<br>
44   * <br>
45   * For instance: -to host1,host2,host3 -file file1,file2 <br>
46   * Will generate: <br>
47   * -to host1 -file file1<br>
48   * -to host1 -file file2<br>
49   * -to host2 -file file1<br>
50   * -to host2 -file file2<br>
51   * -to host3 -file file1<br>
52   * -to host3 -file file2<br>
53   * <br>
54   * <br>Extra option is -client which allows the filename resolution on remote (recv files) when using wildcards.<br>
55   * 
56   * @author Frederic Bregier
57   * 
58   */
59  public class MultipleSubmitTransfer extends SubmitTransfer {
60  	public int errorMultiple = 0;
61  	public int doneMultiple = 0;
62  	protected boolean submit = false;
63  	protected NetworkTransaction networkTransaction = null;
64  	public List<OutputFormat> results = new ArrayList<OutputFormat>();
65  
66  	public MultipleSubmitTransfer(R66Future future, String remoteHost,
67  			String filename, String rulename, String fileinfo, boolean isMD5, int blocksize,
68  			long id,
69  			Timestamp starttime, NetworkTransaction networkTransaction) {
70  		super(future, remoteHost, filename, rulename, fileinfo, isMD5, blocksize, id, starttime);
71  		this.networkTransaction = networkTransaction;
72  	}
73  
74  	
75  	@Override
76  	public void run() {
77  		String [] localfilenames = filename.split(",");
78  		String [] rhosts = remoteHost.split(",");
79  		R66Result resultError = null;
80  
81  		// first check if filenames contains wildcards
82  		DbRule dbrule = null;
83  		try {
84  			dbrule = new DbRule(DbConstant.admin.session, rulename);
85  		} catch (WaarpDatabaseException e) {
86  			logger.error(Messages.getString("SubmitTransfer.2")+rule); //$NON-NLS-1$
87  			ChannelUtils.stopLogger();
88  			System.exit(2);
89  		}
90  		List<String> files = null;
91  		if (dbrule.isSendMode()) {
92  			files = MultipleDirectTransfer.getLocalFiles(dbrule, localfilenames);
93  		} else if (submit) {
94  			files = new ArrayList<String>();
95  			for (String string : localfilenames) {
96  				files.add(string);
97  			}
98  		}
99  		for (String host : rhosts) {
100 			host = host.trim();
101 			if (host != null && ! host.isEmpty()) {
102 				if (! submit && dbrule.isRecvMode()) {
103 					files = MultipleDirectTransfer.getRemoteFiles(dbrule, localfilenames, host, networkTransaction);
104 				}
105 				for (String filename : files) {
106 					filename = filename.trim();
107 					if (filename != null && ! filename.isEmpty()) {
108 						R66Future future = new R66Future(true);
109 						SubmitTransfer transaction = new SubmitTransfer(future,
110 								host, filename, rule, fileInfo, ismd5, block, idt,
111 								ttimestart);
112 						transaction.normalInfoAsWarn = normalInfoAsWarn;
113 						transaction.run();
114 						future.awaitUninterruptibly();
115 						DbTaskRunner runner = future.getResult().runner;
116 						OutputFormat outputFormat = new OutputFormat(MultipleSubmitTransfer.class.getSimpleName(), null);
117 						if (future.isSuccess()) {
118 							outputFormat.setValue(FIELDS.status.name(), 0);
119 							outputFormat.setValue(FIELDS.statusTxt.name(), Messages.getString("SubmitTransfer.3")+Messages.getString("RequestInformation.Success")); //$NON-NLS-1$
120 							outputFormat.setValue(FIELDS.remote.name(), host);
121 							outputFormat.setValueString(runner.getJson());
122 							results.add(outputFormat);
123 							if (transaction.normalInfoAsWarn) {
124 								logger.warn(outputFormat.loggerOut());
125 							} else {
126 								logger.info(outputFormat.loggerOut());
127 							}
128 							doneMultiple++;
129 						} else {
130 							outputFormat.setValue(FIELDS.status.name(), 2);
131 							if (runner == null) {
132 								outputFormat.setValue(FIELDS.statusTxt.name(), Messages.getString("SubmitTransfer.3")+Messages.getString("Transfer.FailedNoId")); //$NON-NLS-1$
133 								outputFormat.setValue(FIELDS.remote.name(), host);
134 							} else {
135 								outputFormat.setValue(FIELDS.statusTxt.name(), Messages.getString("SubmitTransfer.3")+Messages.getString("RequestInformation.Failure")); //$NON-NLS-1$
136 								outputFormat.setValue(FIELDS.remote.name(), host);
137 								outputFormat.setValueString(runner.getJson());
138 							}
139 							logger.error(outputFormat.loggerOut(), future.getCause());
140 							if (future.getCause() != null) {
141 								outputFormat.setValue(FIELDS.error.name(), future.getCause().getMessage());
142 							}
143 							results.add(outputFormat);
144 							errorMultiple++;
145 							resultError = future.getResult();
146 						}
147 					}
148 				}
149 			}
150 		}
151 		if (errorMultiple > 0) {
152 			if (resultError != null) {
153 				this.future.setResult(resultError);
154 			}
155 			this.future.cancel();
156 		} else {
157 			this.future.setSuccess();
158 		}
159 	}

162 	/**
163 	 * 
164 	 * @param args
165 	 *            configuration file, the remoteHost Id, the file to transfer, the rule, file
166 	 *            transfer information as arguments and optionally isMD5=1 for true or 0 for
167 	 *            false(default) and the blocksize if different than default
168 	 */
169 	public static void main(String[] args) {
170 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
171 		if (logger == null) {
172 			logger = WaarpInternalLoggerFactory.getLogger(MultipleSubmitTransfer.class);
173 		}
174 		boolean submit = true;
175 		for (String string : args) {
176 			if (string.equalsIgnoreCase("-client")) {
177 				submit = false;
178 			}
179 		}
180 		if (!getParams(args, submit)) {
181 			logger.error(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
182 			if (! OutputFormat.isQuiet()) {
183 				System.out.println(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
184 			}
185 			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
186 				DbConstant.admin.close();
187 			}
188 			ChannelUtils.stopLogger();
189 			System.exit(1);
190 		}
191 		NetworkTransaction networkTransaction = null;
192 		if (! submit) {
193 			Configuration.configuration.pipelineInit();
194 			networkTransaction = new NetworkTransaction();
195 		}
196 		try {
197 			R66Future future = new R66Future(true);
198 			MultipleSubmitTransfer transaction = new MultipleSubmitTransfer(future,
199 					rhost, localFilename, rule, fileInfo, ismd5, block, idt,
200 					ttimestart, networkTransaction);
201 			transaction.normalInfoAsWarn = snormalInfoAsWarn;
202 			transaction.run();
203 			future.awaitUninterruptibly();
204 			OutputFormat outputFormat = new OutputFormat("Unique "+MultipleSubmitTransfer.class.getSimpleName(), args);
205 			if (future.isSuccess()) {
206 				outputFormat.setValue(FIELDS.status.name(), 0);
207 				outputFormat.setValue(FIELDS.statusTxt.name(), "Multiple "+Messages.getString("SubmitTransfer.3")+Messages.getString("RequestInformation.Success")); //$NON-NLS-1$
208 				outputFormat.setValue(FIELDS.remote.name(), rhost);
209 				outputFormat.setValue("ok", transaction.doneMultiple);
210 				if (transaction.normalInfoAsWarn) {
211 					logger.warn(outputFormat.loggerOut());
212 				} else {
213 					logger.info(outputFormat.loggerOut());
214 				}
215 				if (! OutputFormat.isQuiet()) {
216 					outputFormat.sysout();
217 					for (OutputFormat result : transaction.results) {
218 						System.out.println();
219 						result.sysout();
220 					}
221 				}
222 			} else {
223 				outputFormat.setValue(FIELDS.status.name(), 2);
224 				outputFormat.setValue(FIELDS.statusTxt.name(), "Multiple "+Messages.getString("SubmitTransfer.14")+Messages.getString("RequestInformation.Failure")); //$NON-NLS-1$
225 				outputFormat.setValue(FIELDS.remote.name(), rhost);
226 				outputFormat.setValue("ok", transaction.doneMultiple);
227 				outputFormat.setValue("ko", transaction.errorMultiple);
228 				logger.error(outputFormat.loggerOut());
229 				if (! OutputFormat.isQuiet()) {
230 					outputFormat.sysout();
231 					for (OutputFormat result : transaction.results) {
232 						System.out.println();
233 						result.sysout();
234 					}
235 				}
236 				if (networkTransaction != null)  {
237 					networkTransaction.closeAll();
238 					networkTransaction = null;
239 				}
240 				DbConstant.admin.close();
241 				ChannelUtils.stopLogger();
242 				System.exit(transaction.errorMultiple);
243 			}
244 			DbConstant.admin.close();
245 		} finally {
246 			if (networkTransaction != null)  {
247 				networkTransaction.closeAll();
248 			}
249 		}
250 	}
