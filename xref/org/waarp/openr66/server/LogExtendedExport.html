142 		this.ruleDownload = ruleDownload;
143 		if (ruleDownload != null && tryimport && 
144 				! this.host.getHostid().equals(Configuration.configuration.HOST_ID) &&
145 				! this.host.getHostid().equals(Configuration.configuration.HOST_SSLID)) {

153 	 * Prior to call this method, the pipeline and NetworkTransaction must have been initialized. It
154 	 * is the responsibility of the caller to finish all network resources.
155 	 */
156 	public void run() {
157 		if (logger == null) {
158 			logger = WaarpInternalLoggerFactory.getLogger(LogExtendedExport.class);
159 		}
160 		if (! (statusdone || statuserror || statuspending || statustransfer)) {
161 			logger.error("No action required");
162 			future.setResult(new R66Result(
163 					new OpenR66ProtocolNoDataException("No action required"),
164 					null, true, ErrorCode.IncorrectCommand, null));
165 			future.setFailure(future.getResult().exception);
166 			return;
167 		}
168 
169 		byte type = (purgeLog) ? LocalPacketFactory.LOGPURGEPACKET : LocalPacketFactory.LOGPACKET;
170 		LogJsonPacket node = new LogJsonPacket();
171 		node.setClean(clean);
172 		node.setPurge(purgeLog);
173 		node.setStart(start);
174 		node.setStop(stop);
175 		node.setStartid(startid);
176 		node.setStopid(stopid);
177 		node.setRule(rule);
178 		node.setRequest(request);
179 		node.setStatuspending(statuspending);
180 		node.setStatustransfer(statustransfer);
181 		node.setStatuserror(statuserror);
182 		node.setStatusdone(statusdone);
183 		
184 		JsonCommandPacket valid = new JsonCommandPacket(node, type);
185 		logger.debug("ExtendedLogCommand: " +valid.getRequest());
186 		SocketAddress socketAddress = host.getSocketAddress();
187 		boolean isSSL = host.isSsl();
188 
189 		R66Future newFuture = new R66Future(true);
190 		LocalChannelReference localChannelReference = networkTransaction
191 				.createConnectionWithRetry(socketAddress, isSSL, newFuture);
192 		socketAddress = null;
193 		if (localChannelReference == null) {
194 			host = null;
195 			logger.error("Cannot Connect");
196 			future.setResult(new R66Result(
197 					new OpenR66ProtocolNoConnectionException("Cannot connect to server"),
198 					null, true, ErrorCode.Internal, null));
199 			future.setFailure(future.getResult().exception);
200 			return;
201 		}
202 		localChannelReference.sessionNewState(R66FiniteDualStates.VALIDOTHER);
203 		try {
204 			ChannelUtils.writeAbstractLocalPacket(localChannelReference, valid, false);
205 		} catch (OpenR66ProtocolPacketException e) {
206 			logger.error("Bad Protocol", e);
207 			Channels.close(localChannelReference.getLocalChannel());
208 			localChannelReference = null;
209 			host = null;
210 			valid = null;
211 			future.setResult(new R66Result(e, null, true,
212 					ErrorCode.TransferError, null));
213 			future.setFailure(e);
214 			return;
215 		}
216 		host = null;
217 		newFuture.awaitUninterruptibly();
218 		logger.info("Request done with " + (newFuture.isSuccess() ? "success" : "error"));
219 		if (newFuture.isSuccess() && ruleDownload != null && ! ruleDownload.isEmpty()) {
220 			try {
221 				importLog(newFuture);
222 			} catch (OpenR66ProtocolBusinessException e) {
223 				Channels.close(localChannelReference.getLocalChannel());
224 				localChannelReference = null;
225 				return;
226 			}
227 		}
228 		future.filesize = newFuture.filesize;
229 		future.runner = newFuture.runner;
230 		future.setResult(newFuture.getResult());
231 		if (newFuture.isSuccess()) {
232 			future.setSuccess();
233 		} else {
234 			if (newFuture.getCause() != null) {
235 				future.setFailure(newFuture.getCause());
236 			}
237 			future.cancel();
238 		}
239 		Channels.close(localChannelReference.getLocalChannel());
240 		localChannelReference = null;
241 	}

243 	public void importLog(R66Future future) throws OpenR66ProtocolBusinessException {
244 		if (future.isSuccess()) {
245 			JsonCommandPacket packet = (JsonCommandPacket) future.getResult().other;
246 			if (packet != null) {
247 				LogResponseJsonPacket res = (LogResponseJsonPacket) packet.getJsonRequest();
248 				String fileExported = res.getFilename();
249 				// download logs
250 				if (fileExported != null && ! fileExported.isEmpty()) {
251 					String ruleToExport = ruleDownload;
252 					R66Future futuretemp = new R66Future(true);
253 					DirectTransfer transfer = new DirectTransfer(futuretemp, host.getHostid(),
254 							fileExported,
255 							ruleToExport, "Get Exported Logs from "
256 									+ Configuration.configuration.HOST_ID,
257 							false, Configuration.configuration.BLOCKSIZE,
258 							DbConstant.ILLEGALVALUE,
259 							networkTransaction);
260 					transfer.run();
261 					File logsFile = null;
262 					if (!futuretemp.isSuccess()) {
263 						if (futuretemp.getCause() != null) {
264 							throw new OpenR66ProtocolBusinessException(futuretemp.getCause());
265 						}
266 						throw new OpenR66ProtocolBusinessException("Download of exported logs in error");
267 					} else {
268 						logsFile = futuretemp.getResult().file.getTrueFile();
269 					}
270 					if (tryimport && DbConstant.admin.isConnected) {
271 						try {
272 							DbTaskRunner.loadXml(logsFile);
273 						} catch (OpenR66ProtocolBusinessException e) {
274 							logger.warn("Cannot load the logs from "+logsFile.getAbsolutePath()
275 									+" since: "+ e.getMessage());
276 							throw new OpenR66ProtocolBusinessException("Cannot load the logs from "+logsFile.getAbsolutePath(),
277 									e);
278 						}
279 					}
280 				} else {
281 					throw new OpenR66ProtocolBusinessException("Export log with no file result");
282 				}
283 			} else {
284 				throw new OpenR66ProtocolBusinessException("Export log with no file result");
285 			}
286 		} else {
287 			if (future.getCause() != null) {
288 				throw new OpenR66ProtocolBusinessException(future.getCause());
289 			}
290 			throw new OpenR66ProtocolBusinessException("Export log in error");
291 		}
292 	}

304 	protected static boolean getParams(String[] args) {
305 		if (args.length < 1) {
306 			logger.error(_INFO_ARGS);
307 			return false;
308 		}
309 		if (!FileBasedConfiguration
310 				.setClientConfigurationFromXml(Configuration.configuration, args[0])) {
311 			logger.error(_INFO_ARGS);
312 			return false;
313 		}
314 		String ssstart = null;
315 		String ssstop = null;
316 		for (int i = 1; i < args.length; i++) {
317 			if (args[i].equalsIgnoreCase("-purge")) {
318 				spurgeLog = true;
319 			} else if (args[i].equalsIgnoreCase("-clean")) {
320 				sclean = true;
321 			} else if (args[i].equalsIgnoreCase("-start")) {
322 				i++;
323 				ssstart = args[i];
324 			} else if (args[i].equalsIgnoreCase("-stop")) {
325 				i++;
326 				ssstop = args[i];
327 			} else if (args[i].equalsIgnoreCase("-startid")) {
328 				i++;
329 				sstartid = args[i];
330 			} else if (args[i].equalsIgnoreCase("-stopid")) {
331 				i++;
332 				sstopid = args[i];
333 			} else if (args[i].equalsIgnoreCase("-rule")) {
334 				i++;
335 				srule = args[i];
336 			} else if (args[i].equalsIgnoreCase("-request")) {
337 				i++;
338 				srequest = args[i];
339 			} else if (args[i].equalsIgnoreCase("-pending")) {
340 				sstatuspending = true;
341 			} else if (args[i].equalsIgnoreCase("-transfer")) {
342 				sstatustransfer = true;
343 			} else if (args[i].equalsIgnoreCase("-done")) {
344 				sstatusdone = true;
345 			} else if (args[i].equalsIgnoreCase("-error")) {
346 				sstatuserror = true;
347 			} else if (args[i].equalsIgnoreCase("-import")) {
348 				stryimport = true;
349 			} else if (args[i].equalsIgnoreCase("-host")) {
350 				i++;
351 				stohost = args[i];
352 			} else if (args[i].equalsIgnoreCase("-ruleDownload")) {
353 				i++;
354 				sruleDownload = args[i];
355 			}
356 		}
357 		if (ssstart != null) {
358 			Timestamp tstart = WaarpStringUtils.fixDate(ssstart);
359 			if (tstart != null) {
360 				sstart = tstart;
361 			}
362 		}
363 		if (ssstop != null) {
364 			Timestamp tstop = WaarpStringUtils.fixDate(ssstop, sstart);
365 			if (tstop != null) {
366 				sstop = tstop;
367 			}
368 		}
369 		if (ssstart == null && ssstop == null) {
370 			sstop = WaarpStringUtils.getTodayMidnight();
371 		}
372 		if (stohost == null) {
373 			stryimport = false;
374 		}
375 		return true;
376 	}

378 	public static void main(String[] args) {
379 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
380 		if (logger == null) {
381 			logger = WaarpInternalLoggerFactory.getLogger(LogExtendedExport.class);
382 		}
383 		if (!getParams(args)) {
384 			logger.error("Wrong initialization");
385 			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
386 				DbConstant.admin.close();
387 			}
388 			System.exit(1);
389 		}
390 		long time1 = System.currentTimeMillis();
391 		DbHostAuth dbhost = null;
392 		if (stohost != null) {
393 			try {
394 				dbhost = new DbHostAuth(DbConstant.admin.session, stohost);
395 			} catch (WaarpDatabaseException e) {
396 				logger.error("Wrong initialization");
397 				if (DbConstant.admin != null && DbConstant.admin.isConnected) {
398 					DbConstant.admin.close();
399 				}
400 				System.exit(2);
401 			}
402 		} else {
403 			dbhost = Configuration.configuration.HOST_SSLAUTH;
404 			stohost = Configuration.configuration.HOST_SSLID;
405 		}
406 		R66Future future = new R66Future(true);
407 
408 		Configuration.configuration.pipelineInit();
409 		NetworkTransaction networkTransaction = new NetworkTransaction();
410 		try {
411 			LogExtendedExport transaction = new LogExtendedExport(future,
412 					sclean, spurgeLog, sstart, sstop,
413 					sstartid, sstopid, srule, srequest, 
414 					sstatuspending, sstatustransfer, sstatusdone, sstatuserror, 
415 					networkTransaction, dbhost);
416 			transaction.setDownloadTryImport(sruleDownload, stryimport);
417 			transaction.run();
418 			future.awaitUninterruptibly();
419 			long time2 = System.currentTimeMillis();
420 			long delay = time2 - time1;
421 			R66Result result = future.getResult();
422 			if (future.isSuccess()) {
423 				if (result.code == ErrorCode.Warning) {
424 					logger.warn("WARNED on file:     " +
425 							(result.other != null ? ((JsonCommandPacket) result.other).getRequest() :
426 									"no file")
427 							+ "     delay: " + delay);
428 				} else {
429 					logger.warn("SUCCESS on Final file:     " +
430 							(result.other != null ? ((JsonCommandPacket) result.other).getRequest() :
431 									"no file")
432 							+ "     delay: " + delay);
433 				}
434 			} else {
435 				if (result.code == ErrorCode.Warning) {
436 					logger.warn("LogExtendedExport is     WARNED", future.getCause());
437 					networkTransaction.closeAll();
438 					System.exit(result.code.ordinal());
439 				} else {
440 					logger.error("LogExtendedExport in     FAILURE", future.getCause());
441 					networkTransaction.closeAll();
442 					System.exit(result.code.ordinal());
443 				}
444 			}
445 		} finally {
446 			networkTransaction.closeAll();
447 		}
448 	}
