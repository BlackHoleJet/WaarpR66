114 	 * @see java.lang.Runnable#run()

116 	@Override
117 	public void run() {
118 		if (Configuration.configuration.isShutdown) {
119 			taskRunner.changeUpdatedInfo(UpdatedInfo.TOSUBMIT);
120 			taskRunner.forceSaveStatus();
121 			return;
122 		}
123 		try {
124 			if (activeRunners != null) {
125 				activeRunners.add(this);
126 			}
127 			// fix for SelfRequest
128 			if (taskRunner.isSelfRequest()) {
129 				taskRunner.setSenderByRequestToValidate(false);
130 			}
131 			R66Future transfer;
132 			try {
133 				transfer = this.runTransfer();
134 			} catch (OpenR66RunnerErrorException e) {
135 				logger.error("Runner Error: {} {}", e.getMessage(),
136 						taskRunner.toShortString());
137 				return;
138 			} catch (OpenR66ProtocolNoConnectionException e) {
139 				logger.error("No connection Error {}", e.getMessage());
140 				if (localChannelReference != null) {
141 					localChannelReference.setErrorMessage(
142 							ErrorCode.ConnectionImpossible.mesg,
143 							ErrorCode.ConnectionImpossible);
144 				}
145 				taskRunner.setErrorTask(localChannelReference);
146 				try {
147 					taskRunner.forceSaveStatus();
148 					taskRunner.run();
149 				} catch (OpenR66RunnerErrorException e1) {
150 					this.changeUpdatedInfo(UpdatedInfo.INERROR,
151 							ErrorCode.ConnectionImpossible, true);
152 				}
153 				return;
154 			} catch (OpenR66ProtocolPacketException e) {
155 				logger.error("Protocol Error", e);
156 				return;
157 			} catch (OpenR66ProtocolNotYetConnectionException e) {
158 				logger.warn("No connection warning {}", e.getMessage());
159 				return;
160 			}
161 			R66Result result = transfer.getResult();
162 			if (result != null) {
163 				if (result.code == ErrorCode.QueryAlreadyFinished) {
164 					logger.warn(Messages.getString("Transfer.Status")+
165 							(transfer.isSuccess() ? Messages.getString("RequestInformation.Success") : Messages.getString("RequestInformation.Failure")) +
166 							"     " + ErrorCode.QueryAlreadyFinished.mesg +
167 							":" +
168 							(result != null ? result.toString() : "no result"));
169 				} else {
170 					if (transfer.isSuccess()) {
171 						logger.info(Messages.getString("Transfer.Status")+Messages.getString("RequestInformation.Success")+"     " +
172 								(result != null ? result.toString()
173 										: "no result"));
174 					} else {
175 						logger.error(Messages.getString("Transfer.Status")+Messages.getString("RequestInformation.Success")+"     " +
176 								(result != null ? result.toString()
177 										: "no result"));
178 					}
179 				}
180 			} else {
181 				if (transfer.isSuccess()) {
182 					logger.warn(Messages.getString("Transfer.Status")+Messages.getString("RequestInformation.Success")+"     no result");
183 				} else {
184 					logger.error(Messages.getString("Transfer.Status")+Messages.getString("RequestInformation.Failure")+"     no result");
185 				}
186 			}
187 			transfer = null;
188 			Thread.currentThread().setName(
189 					"Finished_" + Thread.currentThread().getName());
190 		} finally {
191 			if (activeRunners != null) {
192 				activeRunners.remove(this);
193 			}
194 		}
195 	}

198 	 * 
199 	 * @param runner
200 	 * @param limit
201 	 * @return True if the task was run less than limit, else False
202 	 */
203 	public boolean incrementTaskRunnerTry(DbTaskRunner runner, int limit) {
204 		String key = runner.getKey();
205 		Integer tries = taskRunnerRetryHashMap.get(key);
206 		logger.debug("try to find integer: " + tries);
207 		if (tries == null) {
208 			tries = Integer.valueOf(1);
209 		} else {
210 			tries = tries + 1;
211 		}
212 		if (limit <= tries) {
213 			taskRunnerRetryHashMap.remove(key);
214 			return false;
215 		} else {
216 			taskRunnerRetryHashMap.put(key, tries);
217 			return true;
218 		}
219 	}

221 	/**
222 	 * True transfer run (can be called directly to enable exception outside any executors)
223 	 * 
224 	 * @return The R66Future of the transfer operation
225 	 * @throws OpenR66RunnerErrorException
226 	 * @throws OpenR66ProtocolNoConnectionException
227 	 * @throws OpenR66ProtocolPacketException
228 	 * @throws OpenR66ProtocolNotYetConnectionException
229 	 */
230 	public R66Future runTransfer() throws OpenR66RunnerErrorException,
231 			OpenR66ProtocolNoConnectionException,
232 			OpenR66ProtocolPacketException,
233 			OpenR66ProtocolNotYetConnectionException {
234 		logger.debug("Start attempt Transfer");
235 		localChannelReference = initRequest();
236 		try {
237 			localChannelReference.getFutureValidRequest().await();
238 		} catch (InterruptedException e) {
239 		}
240 		if (localChannelReference.getFutureValidRequest().isSuccess()) {
241 			return finishTransfer(localChannelReference);
242 		} else if (localChannelReference.getFutureValidRequest().getResult() != null && 
243 				localChannelReference.getFutureValidRequest().getResult().code == ErrorCode.ServerOverloaded) {
244 			return tryAgainTransferOnOverloaded(true, localChannelReference);
245 		} else
246 			return finishTransfer(localChannelReference);
247 	}

249 	/**
250 	 * In case an overloaded signal is returned by the requested
251 	 * 
252 	 * @param retry
253 	 *            if True, it will retry in case of overloaded remote server, else it just stops
254 	 * @param localChannelReference
255 	 * @return The R66Future of the transfer operation
256 	 * @throws OpenR66RunnerErrorException
257 	 * @throws OpenR66ProtocolNoConnectionException
258 	 * @throws OpenR66ProtocolPacketException
259 	 * @throws OpenR66ProtocolNotYetConnectionException
260 	 */
261 	public R66Future tryAgainTransferOnOverloaded(boolean retry,
262 			LocalChannelReference localChannelReference)
263 			throws OpenR66RunnerErrorException,
264 			OpenR66ProtocolNoConnectionException,
265 			OpenR66ProtocolPacketException,
266 			OpenR66ProtocolNotYetConnectionException {
267 		if (this.localChannelReference == null) {
268 			this.localChannelReference = localChannelReference;
269 		}
270 		boolean incRetry = incrementTaskRunnerTry(taskRunner,
271 				Configuration.RETRYNB);
272 		logger.debug("tryAgainTransferOnOverloaded: " + retry + ":" + incRetry);
273 		switch (taskRunner.getUpdatedInfo()) {
274 			case DONE:
275 			case INERROR:
276 			case INTERRUPTED:
277 				break;
278 			default:
279 				this.changeUpdatedInfo(UpdatedInfo.INERROR,
280 						ErrorCode.ServerOverloaded, true);
281 		}
282 		// redo if possible
283 		if (retry && incRetry) {
284 			try {
285 				Thread.sleep(Configuration.configuration.constraintLimitHandler
286 						.getSleepTime());
287 			} catch (InterruptedException e) {
288 			}
289 			return runTransfer();
290 		} else {
291 			if (localChannelReference == null) {
292 				taskRunner
293 						.setLocalChannelReference(new LocalChannelReference());
294 			}
295 			taskRunner.getLocalChannelReference().setErrorMessage(
296 					ErrorCode.ConnectionImpossible.mesg,
297 					ErrorCode.ConnectionImpossible);
298 			this.taskRunner.setErrorTask(localChannelReference);
299 			this.taskRunner.run();
300 			throw new OpenR66ProtocolNoConnectionException(
301 					"End of retry on ServerOverloaded");
302 		}
303 	}

305 	/**
306 	 * Finish the transfer (called at the end of runTransfer)
307 	 * 
308 	 * @param localChannelReference
309 	 * @return The R66Future of the transfer operation
310 	 * @throws OpenR66ProtocolNotYetConnectionException
311 	 * @throws OpenR66ProtocolPacketException
312 	 * @throws OpenR66ProtocolNoConnectionException
313 	 * @throws OpenR66RunnerErrorException
314 	 */
315 	public R66Future finishTransfer(LocalChannelReference localChannelReference)
316 			throws OpenR66RunnerErrorException {
317 		if (this.localChannelReference == null) {
318 			this.localChannelReference = localChannelReference;
319 		}
320 		R66Future transfer = localChannelReference.getFutureRequest();
321 		try {
322 			transfer.await();
323 		} catch (InterruptedException e1) {
324 		}
325 		taskRunnerRetryHashMap.remove(taskRunner.getKey());
326 		logger.info("Request done with {}", (transfer.isSuccess() ? "success"
327 				: "error"));
328 		Channels.close(localChannelReference.getLocalChannel());
329 		// now reload TaskRunner if it still exists (light client can forget it)
330 		boolean isSender = taskRunner.isSender();
331 		if (transfer.isSuccess()) {
332 			try {
333 				taskRunner.select();
334 			} catch (WaarpDatabaseException e) {
335 				logger.debug("Not a problem but cannot find at the end the task");
336 				taskRunner.setFrom(transfer.runner);
337 			}
338 			taskRunner.setSender(isSender);
339 			this.changeUpdatedInfo(UpdatedInfo.DONE, ErrorCode.CompleteOk, false);
340 		} else {
341 			try {
342 				taskRunner.select();
343 			} catch (WaarpDatabaseException e) {
344 				logger.debug("Not a problem but cannot find at the end the task");
345 				taskRunner.setFrom(transfer.runner);
346 			}
347 			taskRunner.setSender(isSender);
348 			// Case when we were interrupted
349 			if (transfer.getResult() == null) {
350 				switch (taskRunner.getUpdatedInfo()) {
351 					case DONE:
352 						R66Result ok = new R66Result(null, true,
353 								ErrorCode.CompleteOk, taskRunner);
354 						transfer.setResult(ok);
355 						transfer.setSuccess();
356 						this.changeUpdatedInfo(UpdatedInfo.DONE,
357 								ErrorCode.CompleteOk, false);
358 						break;
359 					case INERROR:
360 					case INTERRUPTED:
361 					default:
362 						R66Result error = new R66Result(null, true,
363 								ErrorCode.Internal, taskRunner);
364 						transfer.setResult(error);
365 						transfer.cancel();
366 						this.changeUpdatedInfo(UpdatedInfo.INERROR,
367 								ErrorCode.Internal, false);
368 				}
369 				return transfer;
370 			}
371 			if (transfer.getResult().code == ErrorCode.QueryAlreadyFinished) {
372 				// check if post task to execute
373 				logger.warn("WARN QueryAlreadyFinished:     " +
374 						transfer.toString() + "     " +
375 						taskRunner.toShortString());
376 				try {
377 					TransferUtils.finalizeTaskWithNoSession(taskRunner,
378 							localChannelReference);
379 				} catch (OpenR66RunnerErrorException e) {
380 					this.taskRunner.changeUpdatedInfo(UpdatedInfo.INERROR);
381 					this.taskRunner.forceSaveStatus();
382 				}
383 			} else {
384 				switch (taskRunner.getUpdatedInfo()) {
385 					case DONE:
386 					case INERROR:
387 					case INTERRUPTED:
388 					case TOSUBMIT:
389 						break;
390 					default:
391 						this.changeUpdatedInfo(UpdatedInfo.INERROR,
392 								transfer.getResult().code, false);
393 				}
394 			}
395 		}
396 		return transfer;
397 	}

399 	/**
400 	 * Initialize the request
401 	 * 
402 	 * @return the localChannelReference holding the transfer request
403 	 * @throws OpenR66ProtocolNoConnectionException
404 	 * @throws OpenR66RunnerErrorException
405 	 * @throws OpenR66ProtocolPacketException
406 	 * @throws OpenR66ProtocolNotYetConnectionException
407 	 */
408 	public LocalChannelReference initRequest()
409 			throws OpenR66ProtocolNoConnectionException,
410 			OpenR66RunnerErrorException, OpenR66ProtocolPacketException,
411 			OpenR66ProtocolNotYetConnectionException {
412 		this.changeUpdatedInfo(UpdatedInfo.RUNNING, ErrorCode.Running, true);
413 		long id = taskRunner.getSpecialId();
414 		String tid;
415 		if (id == DbConstant.ILLEGALVALUE) {
416 			tid = taskRunner.getRuleId() + "_" + taskRunner.getMode() +
417 					"_NEWTRANSFER";
418 		} else {
419 			tid = taskRunner.getRuleId() + "_" + taskRunner.getMode() + "_" +
420 					id;
421 		}
422 		Thread.currentThread().setName(tid);
423 		logger.debug("Will run {}", this.taskRunner);
424 		boolean restartPost = false;
425 		if (taskRunner.getGloballaststep() == TASKSTEP.POSTTASK.ordinal()) {
426 			// Send a validation to requested
427 			if (!taskRunner.isSelfRequested()) {
428 				// restart
429 				restartPost = true;
430 			}
431 		}
432 		if (taskRunner.isSelfRequested()) {
433 			// Don't have to restart a task for itself (or should use requester)
434 			logger.warn("Requested host cannot initiate itself the request");
435 			this.changeUpdatedInfo(UpdatedInfo.INERROR,
436 					ErrorCode.LoopSelfRequestedHost, true);
437 			throw new OpenR66ProtocolNoConnectionException(
438 					"Requested host cannot initiate itself the request");
439 		}
440 		DbHostAuth host = R66Auth.getServerAuth(DbConstant.admin.session,
441 				taskRunner.getRequested());
442 		if (host == null) {
443 			logger.error("Requested host cannot be found: " +
444 					taskRunner.getRequested());
445 			this.changeUpdatedInfo(UpdatedInfo.INERROR, ErrorCode.NotKnownHost, true);
446 			throw new OpenR66ProtocolNoConnectionException(
447 					"Requested host cannot be found " +
448 							taskRunner.getRequested());
449 		}
450 		if (host.isClient()) {
451 			logger.debug("Cannot initiate a connection with a client: {}", host);
452 			this.changeUpdatedInfo(UpdatedInfo.INERROR,
453 					ErrorCode.ConnectionImpossible, true);
454 			throw new OpenR66ProtocolNoConnectionException(
455 					"Cannot connect to client " + host.toString());
456 		}
457 		SocketAddress socketAddress = host.getSocketAddress();
458 		boolean isSSL = host.isSsl();
459 
460 		LocalChannelReference localChannelReference = networkTransaction
461 				.createConnectionWithRetry(socketAddress, isSSL, futureRequest);
462 		taskRunner.setLocalChannelReference(localChannelReference);
463 		if (localChannelReference == null) {
464 			// propose to redo
465 			// See if reprogramming is ok (not too many tries)
466 			String retry;
467 			if (incrementTaskRunnerTry(taskRunner, Configuration.RETRYNB)) {
468 				logger.debug("Will retry since Cannot connect to {}", host);
469 				retry = " but will retry";
470 				// now wait
471 				try {
472 					Thread.sleep(Configuration.configuration.delayRetry);
473 				} catch (InterruptedException e) {
474 					logger.debug(
475 							"Will not retry since limit of connection attemtps is reached for {}",
476 							host);
477 					retry = " and retries limit is reached so stop here";
478 					this.changeUpdatedInfo(UpdatedInfo.INERROR,
479 							ErrorCode.ConnectionImpossible, true);
480 					taskRunner
481 							.setLocalChannelReference(new LocalChannelReference());
482 					throw new OpenR66ProtocolNoConnectionException(
483 							"Cannot connect to server " + host.toString() + retry);
484 				}
485 				this.changeUpdatedInfo(UpdatedInfo.TOSUBMIT,
486 						ErrorCode.ConnectionImpossible, true);
487 				throw new OpenR66ProtocolNotYetConnectionException(
488 						"Cannot connect to server " + host.toString() + retry);
489 			} else {
490 				logger.debug(
491 						"Will not retry since limit of connection attemtps is reached for {}",
492 						host);
493 				retry = " and retries limit is reached so stop here";
494 				this.changeUpdatedInfo(UpdatedInfo.INERROR,
495 						ErrorCode.ConnectionImpossible, true);
496 				taskRunner
497 						.setLocalChannelReference(new LocalChannelReference());
498 				// set this server as being in shutdown status
499 				NetworkTransaction.proposeShutdownNetworkChannel(socketAddress);
500 				throw new OpenR66ProtocolNoConnectionException(
501 						"Cannot connect to server " + host.toString() + retry);
502 			}
503 		}
504 		socketAddress = null;
505 		if (handler != null) {
506 			localChannelReference.setRecvThroughHandler(handler);
507 		}
508 		localChannelReference.setSendThroughMode(isSendThroughMode);
509 		if (restartPost) {
510 			RequestPacket request = taskRunner.getRequest();
511 			logger.debug("Will send request {} ", request);
512 			localChannelReference.setClientRunner(this);
513 			localChannelReference.sessionNewState(R66FiniteDualStates.REQUESTR);
514 			try {
515 				ChannelUtils.writeAbstractLocalPacket(localChannelReference,
516 						request, true);
517 			} catch (OpenR66ProtocolPacketException e) {
518 				// propose to redo
519 				logger.warn("Cannot transfer request to " + host.toString());
520 				this.changeUpdatedInfo(UpdatedInfo.INTERRUPTED,
521 						ErrorCode.Internal, true);
522 				Channels.close(localChannelReference.getLocalChannel());
523 				localChannelReference = null;
524 				host = null;
525 				request = null;
526 				throw e;
527 			}
528 			logger.debug("Wait for request to {}", host);
529 			request = null;
530 			host = null;
531 			return localChannelReference;
532 		}
533 		// If Requester is NOT Sender, and if TransferTask then decrease now if
534 		// possible the rank
535 		if (!taskRunner.isSender() &&
536 				(taskRunner.getGloballaststep() == TASKSTEP.TRANSFERTASK
537 						.ordinal())) {
538 			logger.debug(
539 					"Requester is not Sender so decrease if possible the rank {}",
540 					taskRunner);
541 			taskRunner.restartRank();
542 			taskRunner.forceSaveStatus();
543 			logger.debug(
544 					"Requester is not Sender so new rank is " +
545 							taskRunner.getRank() + " {}", taskRunner);
546 		}
547 		RequestPacket request = taskRunner.getRequest();
548 		logger.debug("Will send request {} {}", request, localChannelReference);
549 		localChannelReference.setClientRunner(this);
550 		localChannelReference.sessionNewState(R66FiniteDualStates.REQUESTR);
551 		try {
552 			ChannelUtils.writeAbstractLocalPacket(localChannelReference,
553 					request, true);
554 		} catch (OpenR66ProtocolPacketException e) {
555 			// propose to redo
556 			logger.warn("Cannot transfer request to " + host.toString());
557 			this.changeUpdatedInfo(UpdatedInfo.INTERRUPTED, ErrorCode.Internal, true);
558 			Channels.close(localChannelReference.getLocalChannel());
559 			localChannelReference = null;
560 			host = null;
561 			request = null;
562 			throw e;
563 		}
564 		logger.debug("Wait for request to {}", host);
565 		request = null;
566 		host = null;
567 		return localChannelReference;
568 	}
