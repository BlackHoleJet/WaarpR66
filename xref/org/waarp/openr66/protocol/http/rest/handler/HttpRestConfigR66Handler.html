51  /**
52   * Configuration Http REST interface: http://host/config?... + ConfigExportJsonPacket or ConfigImportJsonPacket as GET or PUT
53   * @author "Frederic Bregier"
54   *
55   */
56  public class HttpRestConfigR66Handler extends HttpRestAbstractR66Handler {
57  	
58  	public static final String BASEURI = "config";
59  	/**
60       * Internal Logger
61       */
62      private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
63              .getLogger(HttpRestConfigR66Handler.class);
64     
65  	public HttpRestConfigR66Handler(RestConfiguration config, METHOD ... methods) {
66  		super(BASEURI, config, METHOD.OPTIONS);
67  		setIntersectionMethods(methods, METHOD.GET, METHOD.PUT);
68  	}
69  	
70  	@Override
71  	public void endParsingRequest(HttpRestHandler handler, RestArgument arguments, RestArgument result, Object body)
72  			throws HttpIncorrectRequestException, HttpInvalidAuthenticationException {
73  		logger.debug("debug: {} ### {}",arguments,result);
74  		if (body != null) {
75  			logger.debug("Obj: {}", body);
76  		}
77  		handler.setWillClose(false);
78  		ServerActions serverHandler = ((HttpRestR66Handler) handler).serverHandler;
79  		// now action according to body
80  		JsonPacket json = (JsonPacket) body;
81  		if (json == null) {
82  			result.setDetail("not enough information");
83  			setError(handler, result, HttpResponseStatus.BAD_REQUEST);
84  			return;
85  		}
86  		result.getAnswer().put(AbstractDbData.JSON_MODEL, RESTHANDLERS.Config.name());
87  		try {
88  			if (json instanceof ConfigExportJsonPacket && arguments.getMethod() == METHOD.GET) {//
89  				result.setCommand(ACTIONS_TYPE.ExportConfig.name());
90  				// host, rule, business, alias, roles
91  				ConfigExportJsonPacket node = (ConfigExportJsonPacket) json;
92  				boolean bhost = node.isHost();
93  				boolean brule = node.isRule();
94  				boolean bbusiness = node.isBusiness();
95  				boolean balias = node.isAlias();
96  				boolean broles = node.isRoles();
97  				String sresult[] = serverHandler.configExport(bhost, brule, bbusiness, balias, broles);
98  				// Now answer
99  				ConfigExportResponseJsonPacket resp = new ConfigExportResponseJsonPacket();
100 				resp.fromJson(node);
101 				resp.setFilehost(sresult[0]);
102 				resp.setFilerule(sresult[1]);
103 				resp.setFilebusiness(sresult[2]);
104 				resp.setFilealias(sresult[3]);
105 				resp.setFileroles(sresult[4]);
106 				if (resp.getFilerule() != null || resp.getFilehost() != null || 
107 						resp.getFilebusiness() != null || resp.getFilealias() != null || 
108 								resp.getFileroles() != null) {
109 					setOk(handler, result, resp, HttpResponseStatus.OK);
110 				} else {
111 					result.setDetail("Export configuration in error");
112 					setError(handler, result, resp, HttpResponseStatus.NOT_ACCEPTABLE);
113 				}
114 			} else if (json instanceof ConfigImportJsonPacket && arguments.getMethod() == METHOD.PUT) {//
115 				result.setCommand(ACTIONS_TYPE.ImportConfig.name());
116 				ConfigImportResponseJsonPacket resp = serverHandler.configImport((ConfigImportJsonPacket) json);
117 				if (resp.isImportedhost() || resp.isImportedrule() || 
118 						resp.isImportedbusiness() || resp.isImportedalias() || 
119 						resp.isImportedroles()) {
120 					setOk(handler, result, resp, HttpResponseStatus.OK);
121 				} else {
122 					result.setDetail("Import configuration in error");
123 					setError(handler, result, resp, HttpResponseStatus.NOT_ACCEPTABLE);
124 				}
125 			} else {
126 				logger.info("Validation is ignored: " + json);
127 				result.setDetail("Unknown command");
128 				setError(handler, result, json, HttpResponseStatus.PRECONDITION_FAILED);
129 			}
130 		} catch (OpenR66ProtocolNotAuthenticatedException e) {
131 			throw new HttpInvalidAuthenticationException(e);
132 		} catch (OpenR66ProtocolSystemException e) {
133 			throw new HttpIncorrectRequestException(e);
134 		}
135 	}

137 	protected ArrayNode getDetailedAllow() {
138 		ArrayNode node = JsonHandler.createArrayNode();
139 		
140 		if (this.methods.contains(METHOD.GET)) {
141 			ConfigExportJsonPacket node3 = new ConfigExportJsonPacket();
142 			node3.setRequestUserPacket();
143 			node3.setComment("ConfigExport request (GET)");
144 			ObjectNode node2;
145 			ArrayNode node1 = JsonHandler.createArrayNode();
146 			ConfigExportResponseJsonPacket resp = new ConfigExportResponseJsonPacket();
147 			resp.setComment("ConfigExport response");
148 			resp.setFilealias("filepath");
149 			resp.setFilebusiness("filepath");
150 			resp.setFilehost("filepath");
151 			resp.setFileroles("filepath");
152 			resp.setFilerule("filepath");
153 			resp.setRequestUserPacket();
154 			try {
155 				node1.add(resp.createObjectNode());
156 				node2 = RestArgument.fillDetailedAllow(METHOD.GET, this.path, ACTIONS_TYPE.ExportConfig.name(), node3.createObjectNode(), node1);
157 				node.add(node2);
158 			} catch (OpenR66ProtocolPacketException e1) {
159 			}
160 		}
161 		if (this.methods.contains(METHOD.PUT)) {
162 			ConfigImportJsonPacket node4 = new ConfigImportJsonPacket();
163 			node4.setRequestUserPacket();
164 			node4.setComment("ConfigImport request (PUT) where items are either set through transfer Id, either set directly with a filename");
165 			node4.setAlias("AliasFilename if not through TransferId");
166 			node4.setBusiness("BusinessFilename if not through TransferId");
167 			node4.setHost("HostFilename if not through TransferId");
168 			node4.setRoles("RolesFilename if not through TransferId");
169 			node4.setRule("RuleFilename if not through TransferId");
170 			ConfigImportResponseJsonPacket resp2 = new ConfigImportResponseJsonPacket();
171 			resp2.setComment("ConfigImport response");
172 			resp2.setAlias("filepath");
173 			resp2.setBusiness("filepath");
174 			resp2.setHost("filepath");
175 			resp2.setRoles("filepath");
176 			resp2.setRule("filepath");
177 			resp2.setRequestUserPacket();
178 			ArrayNode node1 = JsonHandler.createArrayNode();
179 			try {
180 				node1.add(resp2.createObjectNode());
181 				ObjectNode node2 = RestArgument.fillDetailedAllow(METHOD.PUT, this.path, ACTIONS_TYPE.ImportConfig.name(), node4.createObjectNode(), node1);
182 				node.add(node2);
183 			} catch (OpenR66ProtocolPacketException e1) {
184 			}
185 		}
186 		
187 		ObjectNode node2 = RestArgument.fillDetailedAllow(METHOD.OPTIONS, this.path, COMMAND_TYPE.OPTIONS.name(), null, null);
188 		node.add(node2);
189 
190 		return node;
191 	}
