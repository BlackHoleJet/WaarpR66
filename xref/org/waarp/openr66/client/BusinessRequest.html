34  /**
35   * class for direct Business Request call
36   * 
37   * @author Frederic Bregier
38   * 
39   */
40  public class BusinessRequest extends AbstractBusinessRequest {
41  	/**
42  	 * Internal Logger
43  	 */
44  	private static WaarpInternalLogger logger;
45  	/**
46  	 * Default class
47  	 */
48  	public static final String DEFAULT_CLASS = "org.waarp.openr66.context.task.ExecBusinessTask";
49  
50  	public BusinessRequest(NetworkTransaction networkTransaction,
51  			R66Future future, String remoteHost, BusinessRequestPacket packet) {
52  		super(BusinessRequest.class, future, remoteHost, networkTransaction, packet);
53  	}
54  
55  	public static void main(String[] args) {
56  		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(
57  				null));
58  		if (logger == null) {
59  			logger = WaarpInternalLoggerFactory.getLogger(BusinessRequest.class);
60  		}
61  		if (args.length < 5) {
62  			logger
63  					.error("Needs at least the configuration file as first argument and Business Java class and -to host as second and third arguments and -arg arguments");
64  			return;
65  		}
66  		classname = DEFAULT_CLASS;
67  		if (!getParams(args) || classarg == null) {
68  			logger.error("Wrong initialization");
69  			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
70  				DbConstant.admin.close();
71  			}
72  			ChannelUtils.stopLogger();
73  			System.exit(2);
74  		}
75  		Configuration.configuration.pipelineInit();
76  		NetworkTransaction networkTransaction = new NetworkTransaction();
77  		R66Future future = new R66Future(true);
78  
79  		logger.info("Start Test of Transaction");
80  		long time1 = System.currentTimeMillis();
81  
82  		BusinessRequestPacket packet =
83  				new BusinessRequestPacket(classname + " " + classarg, 0);
84  		BusinessRequest transaction = new BusinessRequest(
85  				networkTransaction, future, rhost, packet);
86  		transaction.run();
87  		future.awaitUninterruptibly();
88  
89  		long time2 = System.currentTimeMillis();
90  		logger.debug("Finish Business Request: " + future.isSuccess());
91  		long delay = time2 - time1;
92  		if (future.isSuccess()) {
93  			logger.info("Business Request in status: SUCCESS" +
94  					"    <REMOTE>" + rhost + "</REMOTE>" +
95  					"    delay: " + delay);
96  		} else {
97  			logger.info("Business Request in status: FAILURE" +
98  					"    <REMOTE>" + rhost + "</REMOTE>" +
99  					"    <ERROR>" + future.getCause() + "</ERROR>" +
100 					"    delay: " + delay);
101 			networkTransaction.closeAll();
102 			System.exit(ErrorCode.Unknown.ordinal());
103 		}
104 		networkTransaction.closeAll();
105 	}

107 }
