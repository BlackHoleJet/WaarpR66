42  /**
43   * Direct Transfer from a client with or without database connection to transfer for multiple files to multiple hosts at once.<br>
44   * Files will have to be separated by ','.<br>
45   * Hosts will have to be separated by ','.<br>
46   * <br>
47   * For instance: -to host1,host2,host3 -file file1,file2 <br>
48   * Will generate: <br>
49   * -to host1 -file file1<br>
50   * -to host1 -file file2<br>
51   * -to host2 -file file1<br>
52   * -to host2 -file file2<br>
53   * -to host3 -file file1<br>
54   * -to host3 -file file2<br>
55   * 
56   * @author Frederic Bregier
57   * 
58   */
59  public class MultipleDirectTransfer extends DirectTransfer {
60  	public MultipleDirectTransfer(R66Future future, String remoteHost,
61  			String filename, String rulename, String fileinfo, boolean isMD5, int blocksize,
62  			long id,
63  			NetworkTransaction networkTransaction) {
64  		// no starttime since it is direct (blocking request, no delay)
65  		super(future, remoteHost, filename, rulename, fileinfo, isMD5, blocksize, id, networkTransaction);
66  	}

68  	public static List<String> getRemoteFiles(DbRule dbrule, String []localfilenames, String requested, NetworkTransaction networkTransaction) {
69  		List<String> files = new ArrayList<String>();
70  		for (String filename : localfilenames) {
71  			if (!(filename.contains("*") || filename.contains("?") || filename.contains("~"))) {
72  				files.add(filename);
73  			} else {
74  				// remote query
75  				R66Future futureInfo = new R66Future(true);
76  				logger.info("Ask for "+filename+ " to "+requested);
77  				RequestInformation info = new RequestInformation(futureInfo, requested, rule, filename,
78  						(byte) InformationPacket.ASKENUM.ASKLIST.ordinal(), -1, false, networkTransaction);
79  				info.run();
80  				futureInfo.awaitUninterruptibly();
81  				if (futureInfo.isSuccess()) {
82  					ValidPacket valid = (ValidPacket) futureInfo.getResult().other;
83  					if (valid != null) {
84  						String line = valid.getSheader();
85  						String []lines = line.split("\n");
86  						for (String string : lines) {
87  							File tmpFile = new File(string);
88  							files.add(tmpFile.getName());
89  						}
90  					}
91  				} else {
92  					logger.error("Error Asking for "+filename+ " to "+requested + ": "+ futureInfo.getCause().getMessage());
93  				}
94  			}
95  		}
96  		return files;
97  	}

100 	public static List<String> getLocalFiles(DbRule dbrule, String []localfilenames) {
101 		List<String> files = new ArrayList<String>();
102 		R66Session session = new R66Session();
103 		session.getAuth().specialNoSessionAuth(true, Configuration.configuration.HOST_ID);
104 		R66Dir dir = new R66Dir(session);
105 		try {
106 			dir.changeDirectory(dbrule.getSendPath());
107 		} catch (CommandAbstractException e) {
108 		}
109 		if (localfilenames != null) {
110 			for (String filename : localfilenames) {
111 				if (!(filename.contains("*") || filename.contains("?") || filename.contains("~"))) {
112 					files.add(filename);
113 				} else {
114 					// local: must check
115 					logger.info("Local Ask for "+filename+" from "+dir.getFullPath());
116 					List<String> list;
117 					try {
118 						list = dir.list(filename);
119 						if (list != null) {
120 							files.addAll(list);
121 						}
122 					} catch (CommandAbstractException e) {
123 						logger.warn("Issue with file: "+filename + " : "+e.getMessage());
124 					}
125 				}
126 			}
127 		}
128 		return files;
129 	}

131 	public static void main(String[] args) {
132 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
133 		if (logger == null) {
134 			logger = WaarpInternalLoggerFactory.getLogger(MultipleDirectTransfer.class);
135 		}
136 		if (!getParams(args, false)) {
137 			logger.error("Wrong initialization");
138 			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
139 				DbConstant.admin.close();
140 			}
141 			ChannelUtils.stopLogger();
142 			System.exit(2);
143 		}
144 		long time1 = System.currentTimeMillis();
145 
146 		Configuration.configuration.pipelineInit();
147 		NetworkTransaction networkTransaction = new NetworkTransaction();
148 		try {
149 			String [] localfilenames = localFilename.split(",");
150 			String [] rhosts = rhost.split(",");
151 			int error = 0;
152 			boolean inError = false;
153 			// first check if filenames contains wildcards
154 			DbRule dbrule = new DbRule(DbConstant.admin.session, rule);
155 			List<String> files = null;
156 			if (dbrule.isSendMode()) {
157 				files = getLocalFiles(dbrule, localfilenames);
158 			}
159 			for (String host : rhosts) {
160 				host = host.trim();
161 				if (host != null && ! host.isEmpty()) {
162 					if (dbrule.isRecvMode()) {
163 						files = getRemoteFiles(dbrule, localfilenames, host, networkTransaction);
164 					}
165 					for (String filename : files) {
166 						filename = filename.trim();
167 						if (filename != null && ! filename.isEmpty()) {
168 							logger.warn("Launch transfer to "+host+" with file "+filename);
169 							R66Future future = new R66Future(true);
170 							MultipleDirectTransfer transaction = new MultipleDirectTransfer(future,
171 									host, filename, rule, fileInfo, ismd5, block, idt,
172 									networkTransaction);
173 							logger.debug("rhost: "+rhost+":"+transaction.remoteHost);
174 							transaction.run();
175 							future.awaitUninterruptibly();
176 							long time2 = System.currentTimeMillis();
177 							logger.debug("finish transfer: " + future.isSuccess());
178 							long delay = time2 - time1;
179 							R66Result result = future.getResult();
180 							if (future.isSuccess()) {
181 								if (result.runner.getErrorInfo() == ErrorCode.Warning) {
182 									logger.warn("Transfer in status: WARNED     "
183 											+ result.runner.toShortString()
184 											+
185 											"     <REMOTE>"
186 											+ rhost
187 											+ "</REMOTE>"
188 											+
189 											"     <FILEFINAL>"
190 											+
191 											(result.file != null ? result.file.toString() + "</FILEFINAL>"
192 													: "no file")
193 											+ "     delay: " + delay);
194 								} else {
195 									logger.warn("Transfer in status: SUCCESS     "
196 											+ result.runner.toShortString()
197 											+
198 											"     <REMOTE>"
199 											+ rhost
200 											+ "</REMOTE>"
201 											+
202 											"     <FILEFINAL>"
203 											+
204 											(result.file != null ? result.file.toString() + "</FILEFINAL>"
205 													: "no file")
206 											+ "     delay: " + delay);
207 								}
208 								if (nolog) {
209 									// In case of success, delete the runner
210 									try {
211 										result.runner.delete();
212 									} catch (WaarpDatabaseException e) {
213 										logger.warn("Cannot apply nolog to     " + result.runner.toShortString(),
214 												e);
215 									}
216 								}
217 							} else {
218 								if (result == null || result.runner == null) {
219 									logger.error("Transfer in     FAILURE with no Id", future.getCause());
220 									inError = true;
221 									error = ErrorCode.Unknown.ordinal();
222 								}
223 								if (result.runner.getErrorInfo() == ErrorCode.Warning) {
224 									logger.warn("Transfer is     WARNED     " + result.runner.toShortString() +
225 											"     <REMOTE>" + rhost + "</REMOTE>", future.getCause());
226 									inError = true;
227 									error = result.code.ordinal();
228 								} else {
229 									logger.error("Transfer in     FAILURE     " + result.runner.toShortString() +
230 											"     <REMOTE>" + rhost + "</REMOTE>", future.getCause());
231 									inError = true;
232 									error = result.code.ordinal();
233 								}
234 							}
235 						}
236 					}
237 				}
238 			}
239 			if (inError) {
240 				networkTransaction.closeAll();
241 				System.exit(error);
242 			}
243 		} catch (Exception e) {
244 			logger.warn("exc", e);
245 		} finally {
246 			networkTransaction.closeAll();
247 			System.exit(0);
248 		}
249 	}
