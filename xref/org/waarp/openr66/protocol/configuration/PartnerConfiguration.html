65  	public static enum FIELDS {
66  		HOSTID("nohostid"), VERSION(R66Versions.V2_4_12.getVersion()), 
67  		DIGESTALGO(DigestAlgo.MD5.name), FILESIZE(false), FINALHASH(false),

81  	/**
82  	 * Constructor for an external HostId
83  	 * @param id
84  	 * @param json mainly the version information
85  	 */
86  	public PartnerConfiguration(String id, String json) {
87  		this.id = id;
88  		JsonHandler.setValue(root, FIELDS.HOSTID, id);
89  		int pos = json.lastIndexOf('{');
90  		String version = null;
91  		if (pos > 1) {
92  			version = json.substring(0, pos-1);
93  		} else {
94  			version = json;
95  		}
96  		JsonHandler.setValue(root, FIELDS.VERSION, version);
97  		if (isVersion2GEQVersion1(R66Versions.V2_4_12.getVersion(), version)) {
98  			JsonHandler.setValue(root, FIELDS.FILESIZE, true);
99  			JsonHandler.setValue(root, FIELDS.FINALHASH, true);
100 		} else {
101 			JsonHandler.setValue(root, FIELDS.FILESIZE, (Boolean) FIELDS.FILESIZE.defaultValue);
102 			JsonHandler.setValue(root, FIELDS.FINALHASH, (Boolean) FIELDS.FINALHASH.defaultValue);
103 		}
104 		JsonHandler.setValue(root, FIELDS.DIGESTALGO, Configuration.configuration.digest.name);
105 		JsonHandler.setValue(root, FIELDS.PROXIFIED, (Boolean) FIELDS.PROXIFIED.defaultValue);
106 		String sep = SEPARATOR_FIELD;
107 		if (! isVersion2GEQVersion1(R66Versions.V2_4_13.getVersion(), version)) {
108 			sep = BLANK_SEPARATOR_FIELD;
109 		}
110 		if (isVersion2GEQVersion1(R66Versions.V2_4_17.getVersion(), version)) {
111 			logger.debug("UseJson for "+id+":"+json);
112 			useJson = true;
113 		} else {
114 			logger.debug("NOT UseJson for "+id+":"+json);
115 			useJson = false;
116 		}
117 		JsonHandler.setValue(root, FIELDS.SEPARATOR, sep);
118 		
119 		if (json != null && pos > 1) {
120 			String realjson = json.substring(pos);
121 			ObjectNode info = JsonHandler.getFromString(realjson);
122 			if (info != null) {
123 				root.putAll(info);
124 			}
125 		}
126 		if (this.isProxified()) {
127 			Configuration.configuration.blacklistBadAuthent = false;
128 		}
129 		logger.debug("Info HostId: "+root.toString());
130 	}

216 	public final static DigestAlgo getDigestAlgo(String algo) {
217 		for (DigestAlgo alg : DigestAlgo.values()) {
218 			if (alg.name.equals(algo)) {
219 				return alg;
220 			}
221 		}
222 		try {
223 			return DigestAlgo.valueOf(algo);
224 		} catch (IllegalArgumentException e) {
225 		}
226 		return Configuration.configuration.digest;
227 	}

241 	/**
242 	 * Compare 2 versions
243 	 * @param version1
244 	 * @param version2
245 	 * @return True if version2 >= version1
246 	 */
247 	public final static boolean isVersion2GEQVersion1(String version1, String version2) {
248 		if (version1 == null || version2 == null) {
249 			return false;
250 		}
251 		int major1 = 0;
252 		int rank1 = 0;
253 		int subversion1 = 0;
254 		String [] vals = version1.split("\\.");
255 		major1 = Integer.parseInt(vals[0]);
256 		rank1 = Integer.parseInt(vals[1]);
257 		subversion1 = Integer.parseInt(vals[2]);
258 		int major2 = 0;
259 		int rank2 = 0;
260 		int subversion2 = 0;
261 		vals = version2.split("\\.");
262 		major2 = Integer.parseInt(vals[0]);
263 		rank2 = Integer.parseInt(vals[1]);
264 		subversion2 = Integer.parseInt(vals[2]);
265 		logger.debug("1: "+major1+":"+rank1+":"+subversion1+" <=? "+major2+":"+rank2+":"+subversion2+ " = "+
266 				(major1 < major2 || (major1 == major2 && (rank1 < rank2 || (rank1 == rank2 && subversion1 <= subversion2)))));
267 		return (major1 < major2 || (major1 == major2 && (rank1 < rank2 || (rank1 == rank2 && subversion1 <= subversion2))));
268 	}

276 	public final static boolean isVersion2GTVersion1(String version1, String version2) {
277 		if (version1 == null || version2 == null) {
278 			return false;
279 		}
280 		int major1 = 0;
281 		int rank1 = 0;
282 		int subversion1 = 0;
283 		String [] vals = version1.split("\\.");
284 		major1 = Integer.parseInt(vals[0]);
285 		rank1 = Integer.parseInt(vals[1]);
286 		subversion1 = Integer.parseInt(vals[2]);
287 		int major2 = 0;
288 		int rank2 = 0;
289 		int subversion2 = 0;
290 		vals = version2.split("\\.");
291 		major2 = Integer.parseInt(vals[0]);
292 		rank2 = Integer.parseInt(vals[1]);
293 		subversion2 = Integer.parseInt(vals[2]);
294 		logger.debug("1: "+major1+":"+rank1+":"+subversion1+" <? "+major2+":"+rank2+":"+subversion2+ " = "+
295 				(major1 < major2 || (major1 == major2 && (rank1 < rank2 || (rank1 == rank2 && subversion1 < subversion2)))));
296 		return (major1 < major2 || (major1 == major2 && (rank1 < rank2 || (rank1 == rank2 && subversion1 < subversion2))));
297 	}
