62  	/**
63  	 * Try to restart one Transfer Runner Task
64  	 * 
65  	 * @param taskRunner
66  	 * @return the associated Result
67  	 * @throws WaarpDatabaseException
68  	 */
69  	public static R66Result restartTransfer(DbTaskRunner taskRunner, LocalChannelReference lcr)
70  			throws WaarpDatabaseException {
71  		R66Result finalResult = new R66Result(null, true, ErrorCode.InitOk, taskRunner);
72  		if (lcr != null) {
73  			finalResult.code = ErrorCode.QueryStillRunning;
74  			finalResult.other = Messages.getString("TransferUtils.0"); //$NON-NLS-1$
75  		} else {
76  			if (taskRunner.isSendThrough()) {
77  				// XXX FIXME TODO cannot be restarted... Really?
78  				/*if (false) {
79  					finalResult.code = ErrorCode.PassThroughMode;
80  					finalResult.other = "Transfer cannot be restarted since it is in PassThrough mode";
81  					return finalResult;
82  				}*/
83  			}
84  			// Transfer is not running
85  			// but maybe need action on database
86  			try {
87  				if (taskRunner.restart(true)) {
88  					taskRunner.forceSaveStatus();
89  					finalResult.code = ErrorCode.PreProcessingOk;
90  					finalResult.other = Messages.getString("TransferUtils.1"); //$NON-NLS-1$
91  				} else {
92  					if (taskRunner.isSelfRequested() &&
93  							(taskRunner.getGloballaststep() < TASKSTEP.POSTTASK.ordinal())) {
94  						// send a VALID packet with VALID code to the requester except if client
95  						DbHostAuth host = R66Auth.getServerAuth(DbConstant.admin.session,
96  								taskRunner.getRequester());
97  						if (host == null || host.isClient()) {
98  							// cannot be relaunch from there
99  							finalResult.code = ErrorCode.ConnectionImpossible;
100 							finalResult.other = Messages.getString("TransferUtils.2"); //$NON-NLS-1$
101 							logger.warn(Messages.getString("TransferUtils.3")); //$NON-NLS-1$
102 						} else {
103 							R66Future result = new R66Future(true);
104 							logger.info(Messages.getString("TransferUtils.4")+taskRunner.toShortString()); //$NON-NLS-1$
105 							RequestTransfer requestTransfer =
106 									new RequestTransfer(result, taskRunner.getSpecialId(),
107 											taskRunner.getRequested(), taskRunner.getRequester(),
108 											false, false, true,
109 											Configuration.configuration.getInternalRunner().
110 													getNetworkTransaction());
111 							requestTransfer.run();
112 							result.awaitUninterruptibly();
113 							R66Result finalValue = result.getResult();
114 							switch (finalValue.code) {
115 								case QueryStillRunning:
116 									finalResult.code = ErrorCode.QueryStillRunning;
117 									finalResult.other = Messages.getString("TransferUtils.5"); //$NON-NLS-1$
118 									break;
119 								case Running:
120 									finalResult.code = ErrorCode.Running;
121 									finalResult.other = Messages.getString("TransferUtils.6"); //$NON-NLS-1$
122 									break;
123 								case PreProcessingOk:
124 									finalResult.code = ErrorCode.PreProcessingOk;
125 									finalResult.other = Messages.getString("TransferUtils.7"); //$NON-NLS-1$
126 									break;
127 								case CompleteOk:
128 									finalResult.code = ErrorCode.CompleteOk;
129 									finalResult.other = Messages.getString("TransferUtils.8"); //$NON-NLS-1$
130 									taskRunner.setPostTask();
131 									TransferUtils.finalizeTaskWithNoSession(taskRunner, lcr);
132 									taskRunner.setErrorExecutionStatus(ErrorCode.QueryAlreadyFinished);
133 									taskRunner.forceSaveStatus();
134 									break;
135 								case RemoteError:
136 									finalResult.code = ErrorCode.RemoteError;
137 									finalResult.other = Messages.getString("TransferUtils.9"); //$NON-NLS-1$
138 									break;
139 								default:
140 									finalResult.code = ErrorCode.Internal;
141 									finalResult.other = Messages.getString("TransferUtils.10"); //$NON-NLS-1$
142 									break;
143 							}
144 						}
145 					} else {
146 						finalResult.code = ErrorCode.CompleteOk;
147 						finalResult.other = Messages.getString("TransferUtils.11"); //$NON-NLS-1$
148 						taskRunner.setPostTask();
149 						TransferUtils.finalizeTaskWithNoSession(taskRunner, lcr);
150 						taskRunner.setErrorExecutionStatus(ErrorCode.QueryAlreadyFinished);
151 						taskRunner.forceSaveStatus();
152 					}
153 				}
154 			} catch (OpenR66RunnerErrorException e) {
155 				finalResult.code = ErrorCode.PreProcessingOk;
156 				finalResult.other = Messages.getString("TransferUtils.1"); //$NON-NLS-1$
157 			}
158 		}
159 		return finalResult;
160 	}

162 	private static void stopOneTransfer(DbTaskRunner taskRunner,
163 			StringBuilder builder, R66Session session, String body) {
164 		LocalChannelReference lcr =
165 				Configuration.configuration.getLocalTransaction().
166 						getFromRequest(taskRunner.getKey());
167 		ErrorCode result;
168 		ErrorCode code = ErrorCode.StoppedTransfer;
169 		if (lcr != null) {
170 			int rank = taskRunner.getRank();
171 			lcr.sessionNewState(R66FiniteDualStates.ERROR);
172 			ErrorPacket perror = new ErrorPacket(Messages.getString("TransferUtils.13") + rank, //$NON-NLS-1$
173 					code.getCode(), ErrorPacket.FORWARDCLOSECODE);
174 			try {
175 				// XXX ChannelUtils.writeAbstractLocalPacket(lcr, perror);
176 				// inform local instead of remote
177 				ChannelUtils.writeAbstractLocalPacketToLocal(lcr, perror);
178 			} catch (Exception e) {
179 				logger.warn("Write local packet error", e);
180 			}
181 			result = ErrorCode.StoppedTransfer;
182 		} else {
183 			// Transfer is not running
184 			// if in ERROR already just ignore it
185 			if (taskRunner.getUpdatedInfo() == UpdatedInfo.INERROR) {
186 				result = ErrorCode.TransferError;
187 			} else {
188 				// the database saying it is not stopped
189 				result = ErrorCode.TransferError;
190 				if (taskRunner != null) {
191 					if (taskRunner.stopOrCancelRunner(code)) {
192 						result = ErrorCode.StoppedTransfer;
193 					}
194 				}
195 			}
196 		}
197 		ErrorCode last = taskRunner.getErrorInfo();
198 		taskRunner.setErrorExecutionStatus(result);
199 		if (builder != null) {
200 			builder.append(taskRunner.toSpecializedHtml(session, body,
201 					lcr != null ? Messages.getString("HttpSslHandler.Active") : Messages.getString("HttpSslHandler.NotActive")));
202 		}
203 		taskRunner.setErrorExecutionStatus(last);
204 	}

206 	/**
207 	 * Stop all selected transfers
208 	 * 
209 	 * @param dbSession
210 	 * @param limit
211 	 * @param builder
212 	 * @param session
213 	 * @param body
214 	 * @param startid
215 	 * @param stopid
216 	 * @param tstart
217 	 * @param tstop
218 	 * @param rule
219 	 * @param req
220 	 * @param pending
221 	 * @param transfer
222 	 * @param error
223 	 * @return the associated StringBuilder if the one given as parameter is not null
224 	 */
225 	public static StringBuilder stopSelectedTransfers(DbSession dbSession, int limit,
226 			StringBuilder builder, R66Session session, String body,
227 			String startid, String stopid, Timestamp tstart, Timestamp tstop, String rule,
228 			String req, boolean pending, boolean transfer, boolean error) {
229 		if (dbSession == null || dbSession.isDisconnected) {
230 			// do it without DB
231 			if (ClientRunner.activeRunners != null) {
232 				for (ClientRunner runner : ClientRunner.activeRunners) {
233 					DbTaskRunner taskRunner = runner.getTaskRunner();
234 					stopOneTransfer(taskRunner, builder, session, body);
235 				}
236 			}
237 			if (CommanderNoDb.todoList != null) {
238 				CommanderNoDb.todoList.clear();
239 			}
240 			return builder;
241 		}
242 		DbPreparedStatement preparedStatement = null;
243 		try {
244 			preparedStatement =
245 					DbTaskRunner.getFilterPrepareStatement(dbSession, limit, true,
246 							startid, stopid, tstart, tstop, rule, req,
247 							pending, transfer, error, false, false);
248 			preparedStatement.executeQuery();
249 			while (preparedStatement.getNext()) {
250 				DbTaskRunner taskRunner = DbTaskRunner.getFromStatement(preparedStatement);
251 				stopOneTransfer(taskRunner, builder, session, body);
252 			}
253 			preparedStatement.realClose();
254 			return builder;
255 		} catch (WaarpDatabaseException e) {
256 			if (preparedStatement != null) {
257 				preparedStatement.realClose();
258 			}
259 			logger.error(Messages.getString("TransferUtils.14"), e.getMessage()); //$NON-NLS-1$
260 			return null;
261 		}
262 	}

265 	 * Method to delete the temporary file
266 	 * @param taskRunner
267 	 * @param builder
268 	 * @param session
269 	 * @param body
270 	 */
271 	public static void cleanOneTransfer(DbTaskRunner taskRunner, StringBuilder builder, R66Session session, String body) {
272 		if (! taskRunner.isSender() && ! taskRunner.isAllDone()) {
273 			String name = null;
274 			try {
275 				if (session != null) {
276 					session.getDir().changeDirectory("/");
277 					session.setBadRunner(taskRunner, ErrorCode.QueryAlreadyFinished);
278 					R66File file = session.getFile();
279 					if (file != null) {
280 						name = file.getFile();
281 						if (file.exists()) {
282 							logger.info(Messages.getString("TransferUtils.18")+file); //$NON-NLS-1$
283 							if (! file.delete()) {
284 								logger.warn(Messages.getString("TransferUtils.19")+file); //$NON-NLS-1$
285 							} else {
286 								taskRunner.setRankAtStartup(0);
287 								taskRunner.setFilename("###FILE DELETED### "+name);
288 								taskRunner.update();
289 							}
290 						} else if (! name.contains("###FILE DELETED### ")){
291 							taskRunner.setRankAtStartup(0);
292 							taskRunner.setFilename("###FILE DELETED### "+name);
293 							taskRunner.update();
294 						}
295 					}
296 				}
297 			} catch (CommandAbstractException e1) {
298 				logger.warn(Messages.getString("TransferUtils.19")+name, e1); //$NON-NLS-1$
299 			} catch (WaarpDatabaseException e) {
300 			}
301 		}
302 		if (builder != null) {
303 			LocalChannelReference lcr =
304 					Configuration.configuration.getLocalTransaction().
305 							getFromRequest(taskRunner.getKey());
306 			builder.append(taskRunner.toSpecializedHtml(session, body,
307 					lcr != null ? Messages.getString("HttpSslHandler.Active") : Messages.getString("HttpSslHandler.NotActive")));
308 		}
309 	}

355 				DbTaskRunner taskRunner = DbTaskRunner.getFromStatement(preparedStatement);
356 				stopOneTransfer(taskRunner, null, session, null);
357 				cleanOneTransfer(taskRunner, builder, session, body);
358 			}
359 			preparedStatement.realClose();

371 	 * Finalize a local task since only Post action has to be done
372 	 * 
373 	 * @param taskRunner
374 	 * @param localChannelReference
375 	 * @throws OpenR66RunnerErrorException
376 	 */
377 	public static void finalizeTaskWithNoSession(DbTaskRunner taskRunner,
378 			LocalChannelReference localChannelReference)
379 			throws OpenR66RunnerErrorException {
380 		R66Session session = new R66Session();
381 		session.setStatus(50);
382 		String remoteId = taskRunner.isSelfRequested() ?
383 				taskRunner.getRequester() :
384 				taskRunner.getRequested();
385 		session.getAuth().specialNoSessionAuth(false, remoteId);
386 		session.setNoSessionRunner(taskRunner, localChannelReference);
387 		if (taskRunner.isSender()) {
388 			// Change dir
389 			try {
390 				session.getDir().changeDirectory(taskRunner.getRule().getSendPath());
391 			} catch (CommandAbstractException e) {
392 				throw new OpenR66RunnerErrorException(e);
393 			}
394 		} else {
395 			// Change dir
396 			try {
397 				session.getDir().changeDirectory(taskRunner.getRule().getWorkPath());
398 			} catch (CommandAbstractException e) {
399 				throw new OpenR66RunnerErrorException(e);
400 			}
401 		}
402 		try {
403 			try {
404 				session.setFileAfterPreRunner(false);
405 			} catch (CommandAbstractException e) {
406 				throw new OpenR66RunnerErrorException(e);
407 			}
408 		} catch (OpenR66RunnerErrorException e) {
409 			logger.error(Messages.getString("TransferUtils.27"), taskRunner.getFilename()); //$NON-NLS-1$
410 			taskRunner.changeUpdatedInfo(UpdatedInfo.INERROR);
411 			taskRunner.setErrorExecutionStatus(ErrorCode.FileNotFound);
412 			try {
413 				taskRunner.update();
414 			} catch (WaarpDatabaseException e1) {
415 			}
416 			throw new OpenR66RunnerErrorException(Messages.getString("TransferUtils.28"), e); //$NON-NLS-1$
417 		}
418 		R66File file = session.getFile();
419 		R66Result finalValue = new R66Result(null, true, ErrorCode.CompleteOk, taskRunner);
420 		finalValue.file = file;
421 		finalValue.runner = taskRunner;
422 		taskRunner.finishTransferTask(ErrorCode.TransferOk);
423 		try {
424 			taskRunner.finalizeTransfer(localChannelReference, file, finalValue, true);
425 		} catch (OpenR66ProtocolSystemException e) {
426 			logger.error(Messages.getString("TransferUtils.29"), taskRunner.toShortString()); //$NON-NLS-1$
427 			taskRunner.changeUpdatedInfo(UpdatedInfo.INERROR);
428 			taskRunner.setErrorExecutionStatus(ErrorCode.Internal);
429 			try {
430 				taskRunner.update();
431 			} catch (WaarpDatabaseException e1) {
432 			}
433 			throw new OpenR66RunnerErrorException(Messages.getString("TransferUtils.30"), e); //$NON-NLS-1$
434 		}
