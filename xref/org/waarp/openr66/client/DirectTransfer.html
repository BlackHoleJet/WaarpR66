39  /**
40   * Direct Transfer from a client with or without database connection
41   * 
42   * @author Frederic Bregier
43   * 
44   */
45  public class DirectTransfer extends AbstractTransfer {
46  	protected final NetworkTransaction networkTransaction;
47  
48  	public DirectTransfer(R66Future future, String remoteHost,
49  			String filename, String rulename, String fileinfo, boolean isMD5, int blocksize,
50  			long id,
51  			NetworkTransaction networkTransaction) {
52  		// no starttime since it is direct (blocking request, no delay)
53  		super(DirectTransfer.class,
54  				future, filename, rulename, fileinfo, isMD5, remoteHost, blocksize, id, null);
55  		this.networkTransaction = networkTransaction;
56  	}
57  
58  	/**
59  	 * Prior to call this method, the pipeline and NetworkTransaction must have been initialized. It
60  	 * is the responsibility of the caller to finish all network resources.
61  	 */
62  	public void run() {
63  		if (logger == null) {
64  			logger = WaarpInternalLoggerFactory.getLogger(DirectTransfer.class);
65  		}
66  		DbTaskRunner taskRunner = this.initRequest();
67  		ClientRunner runner = new ClientRunner(networkTransaction, taskRunner, future);
68  		OpenR66ProtocolNotYetConnectionException exc = null;
69  		for (int i = 0; i < Configuration.RETRYNB; i++) {
70  			try {
71  				runner.runTransfer();
72  				exc = null;
73  				break;
74  			} catch (OpenR66RunnerErrorException e) {
75  				logger.debug("Cannot Transfer", e);
76  				future.setResult(new R66Result(e, null, true,
77  						ErrorCode.Internal, taskRunner));
78  				future.setFailure(e);
79  				return;
80  			} catch (OpenR66ProtocolNoConnectionException e) {
81  				logger.debug("Cannot Connect", e);
82  				future.setResult(new R66Result(e, null, true,
83  						ErrorCode.ConnectionImpossible, taskRunner));
84  				// since no connection : just forget it
85  				if (nolog) {
86  					try {
87  						taskRunner.delete();
88  					} catch (WaarpDatabaseException e1) {
89  					}
90  				}
91  				future.setFailure(e);
92  				return;
93  			} catch (OpenR66ProtocolPacketException e) {
94  				logger.debug("Bad Protocol", e);
95  				future.setResult(new R66Result(e, null, true,
96  						ErrorCode.TransferError, taskRunner));
97  				future.setFailure(e);
98  				return;
99  			} catch (OpenR66ProtocolNotYetConnectionException e) {
100 				logger.debug("Not Yet Connected", e);
101 				exc = e;
102 				continue;
103 			}
104 		}
105 		if (exc != null) {
106 			taskRunner.setLocalChannelReference(new LocalChannelReference());
107 			logger.debug("Cannot Connect", exc);
108 			future.setResult(new R66Result(exc, null, true,
109 					ErrorCode.ConnectionImpossible, taskRunner));
110 			// since no connection : just forget it
111 			if (nolog) {
112 				try {
113 					taskRunner.delete();
114 				} catch (WaarpDatabaseException e1) {
115 				}
116 			}
117 			future.setFailure(exc);
118 			return;
119 		}
120 	}

122 	public static void main(String[] args) {
123 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
124 		if (logger == null) {
125 			logger = WaarpInternalLoggerFactory.getLogger(DirectTransfer.class);
126 		}
127 		if (!getParams(args, false)) {
128 			logger.error("Wrong initialization");
129 			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
130 				DbConstant.admin.close();
131 			}
132 			ChannelUtils.stopLogger();
133 			System.exit(2);
134 		}
135 		long time1 = System.currentTimeMillis();
136 		R66Future future = new R66Future(true);
137 
138 		Configuration.configuration.pipelineInit();
139 		NetworkTransaction networkTransaction = new NetworkTransaction();
140 		try {
141 			DirectTransfer transaction = new DirectTransfer(future,
142 					rhost, localFilename, rule, fileInfo, ismd5, block, idt,
143 					networkTransaction);
144 			logger.debug("rhost: "+rhost+":"+transaction.remoteHost);
145 			transaction.run();
146 			future.awaitUninterruptibly();
147 			long time2 = System.currentTimeMillis();
148 			logger.debug("finish transfer: " + future.isSuccess());
149 			long delay = time2 - time1;
150 			R66Result result = future.getResult();
151 			if (future.isSuccess()) {
152 				if (result.runner.getErrorInfo() == ErrorCode.Warning) {
153 					logger.warn("Transfer in status: WARNED     "
154 							+ result.runner.toShortString()
155 							+
156 							"     <REMOTE>"
157 							+ rhost
158 							+ "</REMOTE>"
159 							+
160 							"     <FILEFINAL>"
161 							+
162 							(result.file != null ? result.file.toString() + "</FILEFINAL>"
163 									: "no file")
164 							+ "     delay: " + delay);
165 				} else {
166 					logger.info("Transfer in status: SUCCESS     "
167 							+ result.runner.toShortString()
168 							+
169 							"     <REMOTE>"
170 							+ rhost
171 							+ "</REMOTE>"
172 							+
173 							"     <FILEFINAL>"
174 							+
175 							(result.file != null ? result.file.toString() + "</FILEFINAL>"
176 									: "no file")
177 							+ "     delay: " + delay);
178 				}
179 				if (nolog) {
180 					// In case of success, delete the runner
181 					try {
182 						result.runner.delete();
183 					} catch (WaarpDatabaseException e) {
184 						logger.warn("Cannot apply nolog to     " + result.runner.toShortString(),
185 								e);
186 					}
187 				}
188 			} else {
189 				if (result == null || result.runner == null) {
190 					logger.error("Transfer in     FAILURE with no Id", future.getCause());
191 					networkTransaction.closeAll();
192 					System.exit(ErrorCode.Unknown.ordinal());
193 				}
194 				if (result.runner.getErrorInfo() == ErrorCode.Warning) {
195 					logger.warn("Transfer is     WARNED     " + result.runner.toShortString() +
196 							"     <REMOTE>" + rhost + "</REMOTE>", future.getCause());
197 					networkTransaction.closeAll();
198 					System.exit(result.code.ordinal());
199 				} else {
200 					logger.error("Transfer in     FAILURE     " + result.runner.toShortString() +
201 							"     <REMOTE>" + rhost + "</REMOTE>", future.getCause());
202 					networkTransaction.closeAll();
203 					System.exit(result.code.ordinal());
204 				}
205 			}
206 		} catch (Exception e) {
207 			logger.debug("exc", e);
208 		} finally {
209 			logger.debug("finish transfer: " + future.isDone() + ":" + future.isSuccess());
210 			networkTransaction.closeAll();
211 			// In case something wrong append
212 			if (future.isDone() && future.isSuccess()) {
213 				System.exit(0);
214 			} else {
215 				System.exit(66);
216 			}
217 		}
218 	}
