67  	private DbTaskRunner initRequest(R66Request request) {
68  		Timestamp ttimestart = null;
69  		if (request.isSetStart()) {
70  			Date date;
71  			try {
72  				SimpleDateFormat dateFormat = new SimpleDateFormat("yyyyMMddHHmmss");
73  				date = dateFormat.parse(request.getStart());
74  				ttimestart = new Timestamp(date.getTime());
75  			} catch (ParseException e) {
76  			}
77  		} else if (request.isSetDelay()) {
78  			if (request.getDelay().charAt(0) == '+') {
79  				ttimestart = new Timestamp(System.currentTimeMillis() +
80  						Long.parseLong(request.getDelay().substring(1)));
81  			} else {
82  				ttimestart = new Timestamp(Long.parseLong(request.getDelay()));
83  			}
84  		}
85  		DbRule rule;
86  		try {
87  			rule = new DbRule(DbConstant.admin.session, request.getRule());
88  		} catch (WaarpDatabaseException e) {
89  			logger.warn("Cannot get Rule: " + request.getRule(), e);
90  			return null;
91  		}
92  		int mode = rule.mode;
93  		if (request.isMd5()) {
94  			mode = RequestPacket.getModeMD5(mode);
95  		}
96  		DbTaskRunner taskRunner = null;
97  		long tid = DbConstant.ILLEGALVALUE;
98  		if (request.isSetTid()) {
99  			tid = request.getTid();
100 		}
101 		if (tid != DbConstant.ILLEGALVALUE) {
102 			try {
103 				taskRunner = new DbTaskRunner(DbConstant.admin.session, tid,
104 						request.getDestuid());
105 			} catch (WaarpDatabaseException e) {
106 				logger.warn("Cannot get task", e);
107 				return null;
108 			}
109 		} else {
110 			RequestPacket requestPacket = new RequestPacket(request.getRule(),
111 					mode, request.getFile(), request.getBlocksize(), 0,
112 					tid, request.getInfo());
113 			// Not isRecv since it is the requester, so send => isRetrieve is true
114 			boolean isRetrieve = !RequestPacket.isRecvMode(requestPacket.getMode());
115 			try {
116 				taskRunner =
117 						new DbTaskRunner(DbConstant.admin.session, rule, isRetrieve, requestPacket,
118 								request.getDestuid(), ttimestart);
119 			} catch (WaarpDatabaseException e) {
120 				logger.warn("Cannot get task", e);
121 				return null;
122 			}
123 		}
124 		return taskRunner;
125 	}

127 	@Override
128 	public R66Result transferRequestQuery(R66Request request) throws TException {
129 		DbTaskRunner runner = initRequest(request);
130 		if (runner != null) {
131 			runner.changeUpdatedInfo(AbstractDbData.UpdatedInfo.TOSUBMIT);
132 			try {
133 				runner.update();
134 			} catch (WaarpDatabaseException e) {
135 				logger.warn("Cannot prepare task", e);
136 				return new R66Result(request.getMode(), ErrorCode.CommandNotFound, 
137 						"ERROR: Cannot prepare transfer");
138 			}
139 
140 			R66Result result = new R66Result(request.getMode(), ErrorCode.InitOk, 
141 					"Transfer Scheduled");
142 			if (request.getMode() == RequestMode.SYNCTRANSFER) {
143 				// now need to wait but first, reload the runner
144 				try {
145 					runner.select();
146 					while (! runner.isFinished()) {
147 						try {
148 							Thread.sleep(1000);
149 							runner.select();
150 						} catch (InterruptedException e) {
151 							break;
152 						}
153 					}
154 				} catch (WaarpDatabaseException e1) {
155 				}
156 				setResultFromRunner(runner, result);
157 				if (runner.isAllDone()) {
158 					result.setCode(ErrorCode.CompleteOk);
159 					result.setResultinfo("Transfer Done");
160 				} else {
161 					result.setCode(ErrorCode.valueOf(runner.getErrorInfo().name()));
162 					result.setResultinfo(runner.getErrorInfo().mesg);
163 				}
164 			} else {
165 				try {
166 					runner.select();
167 				} catch (WaarpDatabaseException e) {
168 				}
169 				setResultFromRunner(runner, result);
170 			}
171 			return result;
172 		} else {
173 			logger.warn("ERROR: Transfer NOT scheduled");
174 			R66Result result = new R66Result(request.getMode(), ErrorCode.Internal, 
175 					"ERROR: Transfer NOT scheduled");
176 			return result;
177 		}

200 		R66Session session = lcr.getSession();
201 		DbTaskRunner runner = null;
202 		if (session != null) {
203 			runner = session.getRunner();
204 		} else {
205 			ClientRunner run = lcr.getClientRunner();
206 			if (run != null) {
207 				runner = run.getTaskRunner();
208 			}
209 		}
210 		if (runner != null) {
211 			setResultFromRunner(runner, result);

227 	private R66Result stopOrCancel(R66Request request, LocalChannelReference lcr,
228 			org.waarp.openr66.context.ErrorCode r66code) {
229 		// stop the current transfer
230 		R66Result resulttest;
231 		if (lcr != null) {
232 			int rank = 0;
233 			if (r66code == org.waarp.openr66.context.ErrorCode.StoppedTransfer && lcr.getSession() != null) {
234 				DbTaskRunner taskRunner = lcr.getSession().getRunner();
235 				if (taskRunner != null) {
236 					rank = taskRunner.getRank();
237 				}
238 			}
239 			ErrorPacket error = new ErrorPacket(r66code.name() + " " + rank,
240 					r66code.getCode(), ErrorPacket.FORWARDCLOSECODE);
241 			try {
242 				// inform local instead of remote
243 				ChannelUtils.writeAbstractLocalPacketToLocal(lcr, error);
244 			} catch (Exception e) {
245 			}
246 			resulttest = new R66Result(request.getMode(), ErrorCode.CompleteOk, 
247 					r66code.name());
248 			setResultFromLCR(lcr, resulttest);
249 		} else {
250 			// Transfer is not running
251 			// but maybe need action on database
252 			int test = stopOrCancelRunner(request.getTid(), request.getDestuid(), request.getFromuid(), r66code);
253 			if (test > 0) {
254 				resulttest = new R66Result(request.getMode(), ErrorCode.CompleteOk, 
255 						r66code.name());
256 			} else if (test == 0) {
257 				resulttest = new R66Result(request.getMode(), ErrorCode.TransferOk, 
258 						r66code.name());
259 			} else {
260 				resulttest = new R66Result(request.getMode(), ErrorCode.CommandNotFound, 
261 						"Error: cannot accomplished task on transfer");
262 			}
263 		}
264 		return resulttest;
265 	}

267 	private R66Result restart(R66Request request, LocalChannelReference lcr) {
268 		// Try to validate a restarting transfer
269 		// validLimit on requested side
270 		if (Configuration.configuration.constraintLimitHandler.checkConstraints()) {
271 			logger.warn("Limit exceeded while asking to relaunch a task "
272 					+ request.toString());
273 			return new R66Result(request.getMode(), ErrorCode.ServerOverloaded, 
274 					"Limit exceeded while asking to relaunch a task");
275 		}
276 		// Try to validate a restarting transfer
277 		// header = ?; middle = requested+blank+requester+blank+specialId
278 		DbTaskRunner taskRunner = null;
279 		try {
280 			taskRunner = new DbTaskRunner(DbConstant.admin.session, null,
281 					null, request.getTid(), request.getFromuid(), request.getDestuid());
282 			org.waarp.openr66.context.R66Result resulttest = TransferUtils.restartTransfer(taskRunner, lcr);
283 			return new R66Result(request.getMode(), ErrorCode.valueOf(resulttest.code.name()), 
284 					"Restart asked correctly");
285 		} catch (WaarpDatabaseException e1) {
286 			logger.warn("Exception while trying to restart transfer", e1);
287 			return new R66Result(request.getMode(), ErrorCode.Internal, 
288 					"Exception while trying to restart transfer");
289 		}
290 	}

293 	public R66Result infoTransferQuery(R66Request request) throws TException {
294 		RequestMode mode = request.getMode();
295 		if (mode != RequestMode.INFOREQUEST) {
296 			// error
297 			logger.warn("Mode is uncompatible with infoTransferQuery");
298 			return new R66Result(request.getMode(), ErrorCode.Unimplemented, 
299 					"Mode is uncompatible with infoTransferQuery");
300 		}
301 		// now check if enough arguments are provided
302 		if ((!request.isSetTid()) || (!request.isSetDestuid() && !request.isSetFromuid()) ||
303 				(!request.isSetAction())) {
304 			// error
305 			logger.warn("Not enough arguments");
306 			return new R66Result(request.getMode(), ErrorCode.RemoteError, 
307 					"Not enough arguments");
308 		}
309 		// requested+blank+requester+blank+specialId
310 		LocalChannelReference lcr =
311 				Configuration.configuration.getLocalTransaction().
312 						getFromRequest(request.getDestuid()+" "+request.getFromuid()+" "+request.getTid());
313 		org.waarp.openr66.context.ErrorCode r66code;
314 		switch (request.getAction()) {
315 			case Detail: {
316 				R66Result result = new R66Result(request.getMode(), ErrorCode.CompleteOk, 
317 						"Existence test OK");
318 				result.setAction(Action.Exist);
319 				result.setDestuid(request.getDestuid());
320 				result.setFromuid(request.getFromuid());
321 				result.setTid(request.getTid());
322 				if (lcr != null) {
323 					setResultFromLCR(lcr, result);
324 				} else {
325 					try {
326 						DbTaskRunner runner = new DbTaskRunner(DbConstant.admin.session, null, 
327 								null, request.getTid(), request.getFromuid(), 
328 								request.getDestuid());
329 						if (runner != null) {
330 							setResultFromRunner(runner, result);
331 						}
332 					} catch (WaarpDatabaseException e) {
333 						result.setCode(ErrorCode.FileNotFound);
334 					}
335 				}
336 				return result;
337 			}
338 			case Restart:
339 				return restart(request, lcr);
340 			case Cancel:
341 				r66code = org.waarp.openr66.context.ErrorCode.CanceledTransfer;
342 				return stopOrCancel(request, lcr, r66code);
343 			case Stop:
344 				r66code = org.waarp.openr66.context.ErrorCode.StoppedTransfer;
345 				return stopOrCancel(request, lcr, r66code);
346 			default:
347 				logger.warn("Uncompatible with "+request.getAction().name());
348 				return new R66Result(request.getMode(), ErrorCode.Unimplemented, 
349 						"Uncompatible with "+request.getAction().name());
350 		}

369 	public List<String> infoListQuery(R66Request request) throws TException {
370 		List<String> list = new ArrayList<String>();
371 		RequestMode mode = request.getMode();
372 		if (mode != RequestMode.INFOFILE) {
373 			// error
374 			logger.warn("Not correct mode for infoListQuery");
375 			list.add("Not correct mode for infoListQuery");
376 			return list;
377 		}
378 		// now check if enough arguments are provided
379 		if ((!request.isSetRule()) || (!request.isSetAction())) {
380 			// error
381 			logger.warn("Not enough arguments");
382 			list.add("Not enough arguments");
383 			return list;
384 		}
385 		R66Session session = new R66Session();
386 		session.getAuth().specialNoSessionAuth(false, Configuration.configuration.HOST_ID);
387 		DbRule rule;
388 		try {
389 			rule = new DbRule(DbConstant.admin.session, request.getRule());
390 		} catch (WaarpDatabaseException e) {
391 			logger.warn("Rule is unknown: " + request.getRule());
392 			list.add("Rule is unknown: " + request.getRule());
393 			return list;
394 		}
395 		try {
396 			if (RequestPacket.isRecvMode(rule.mode)) {
397 				session.getDir().changeDirectory(rule.workPath);
398 			} else {
399 				session.getDir().changeDirectory(rule.sendPath);
400 			}
401 
402 			if (request.getAction() == Action.List ||
403 					request.getAction() == Action.Mlsx) {
404 				// ls or mls from current directory
405 				if (request.getAction() == Action.List) {
406 					list = session.getDir().list(request.getFile());
407 				} else {
408 					list = session.getDir().listFull(request.getFile(), false);
409 				}
410 				return list;
411 			} else {
412 				// ls pr mls from current directory and filename
413 				if (! request.isSetFile()) {
414 					logger.warn("File missing");
415 					list.add("File missing");
416 					return list;
417 				}
418 				R66File file = (R66File) session.getDir().setFile(request.getFile(), false);
419 				String sresult = null;
420 				if (request.getAction() == Action.Exist) {
421 					sresult = "" + file.exists();
422 					list.add(sresult);
423 				} else if (request.getAction() == Action.Detail) {
424 					sresult = session.getDir().fileFull(request.getFile(), false);
425 					String[] slist = sresult.split("\n");
426 					sresult = slist[1];
427 					list.add(sresult);
428 				}
429 				return list;
430 			}
431 		} catch (CommandAbstractException e) {
432 			logger.warn("Error occurs during: " + request.toString(), e);
433 			list.add("Error occurs during: " + request.toString());
434 			return list;
435 		}
