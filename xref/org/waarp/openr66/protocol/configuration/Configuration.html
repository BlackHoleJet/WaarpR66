580     public Configuration() {
581         // Init signal handler
582         shutdownConfiguration.timeout = TIMEOUTCON;
583         if (WaarpShutdownHook.shutdownHook == null) {
584             new R66ShutdownHook(shutdownConfiguration);
585         }
586         computeNbThreads();
587         scheduledExecutorService = Executors.newScheduledThreadPool(this.SERVER_THREAD, new WaarpThreadFactory(
588                 "ScheduledTask"));
589         // Init FiniteStates
590         R66FiniteDualStates.initR66FiniteStates();
591         if (!SystemPropertyUtil.isFileEncodingCorrect()) {
592             logger.error("Issue while trying to set UTF-8 as default file encoding: use -Dfile.encoding=UTF-8 as java command argument");
593             logger.warn("Currently file.encoding is: " + SystemPropertyUtil.get(SystemPropertyUtil.FILE_ENCODING));
594         }
595         isExecuteErrorBeforeTransferAllowed = SystemPropertyUtil.getBoolean(
596                 R66SystemProperties.OPENR66_EXECUTEBEFORETRANSFERRED, true);
597         boolean useSpaceSeparator = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_USESPACESEPARATOR, false);
598         if (useSpaceSeparator) {
599             PartnerConfiguration.SEPARATOR_FIELD = PartnerConfiguration.BLANK_SEPARATOR_FIELD;
600         }
601         isHostProxyfied = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_ISHOSTPROXYFIED, false);
602         warnOnStartup = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_STARTUP_WARNING, true);
603         FileBasedConfiguration.checkDatabase = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_STARTUP_DATABASE_CHECK, true);
604         chrootChecked = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_CHROOT_CHECKED, true);
605         blacklistBadAuthent = SystemPropertyUtil.getBoolean(R66SystemProperties.OPENR66_BLACKLIST_BADAUTHENT, true);
606         maxfilenamelength = SystemPropertyUtil.getInt(R66SystemProperties.OPENR66_FILENAME_MAXLENGTH, 255);
607         timeStat = SystemPropertyUtil.getInt(R66SystemProperties.OPENR66_TRACE_STATS, 0);
608         limitCache = SystemPropertyUtil.getInt(R66SystemProperties.OPENR66_CACHE_LIMIT, 20000);
609         if (limitCache <= 100) {
610             limitCache = 100;
611         }
612         timeLimitCache = SystemPropertyUtil.getLong(R66SystemProperties.OPENR66_CACHE_TIMELIMIT, 180000);
613         if (timeLimitCache < 1000) {
614             timeLimitCache = 1000;
615         }
616         DbTaskRunner.createLruCache(limitCache, timeLimitCache);
617         if (limitCache > 0 && timeLimitCache > 1000) {
618             launchInFixedDelay(new CleanLruCache(), timeLimitCache, TimeUnit.MILLISECONDS);
619         }
620         if (isHostProxyfied) {
621             blacklistBadAuthent = false;
622         }
623     }

625     public String toString() {
626         String rest = null;
627         for (RestConfiguration config : restConfigurations) {
628             if (rest == null) {
629                 rest = (config.REST_ADDRESS != null ? "'" + config.REST_ADDRESS + ":" : "'All:") + config.REST_PORT
630                         + "'";
631             } else {
632                 rest += ", " + (config.REST_ADDRESS != null ? "'" + config.REST_ADDRESS + ":" : "'All:")
633                         + config.REST_PORT + "'";
634             }
635         }
636         return "Config: { ServerPort: " + SERVER_PORT + ", ServerSslPort: " + SERVER_SSLPORT + ", ServerView: "
637                 + SERVER_HTTPPORT + ", ServerAdmin: " + SERVER_HTTPSPORT +
638                 ", ThriftPort: " + (thriftport > 0 ? thriftport : "'NoThriftSupport'") + ", RestAddress: ["
639                 + (rest != null ? rest : "'NoRestSupport'") + "]" +
640                 ", TimeOut: " + TIMEOUTCON + ", BaseDir: '" + baseDirectory + "', DigestAlgo: '" + digest.name
641                 + "', checkRemote: " + checkRemoteAddress +
642                 ", checkClient: " + checkClientAddress + ", snmpActive: " + (agentSnmp != null) + ", chrootChecked: "
643                 + chrootChecked +
644                 ", blacklist: " + blacklistBadAuthent + ", isHostProxified: " + isHostProxyfied + "}";
645     }

647     /**
648      * Configure the pipeline for client (to be called only once)
649      */
650     public void pipelineInit() {
651         if (configured) {
652             return;
653         }
654         workerGroup = new NioEventLoopGroup(CLIENT_THREAD, new WaarpThreadFactory("Worker"));
655         handlerGroup = new NioEventLoopGroup(CLIENT_THREAD, new WaarpThreadFactory("Handler"));
656         subTaskGroup = new NioEventLoopGroup(CLIENT_THREAD, new WaarpThreadFactory("SubTask"));
657         localBossGroup = new NioEventLoopGroup(CLIENT_THREAD, new WaarpThreadFactory("LocalBoss"));
658         localWorkerGroup = new NioEventLoopGroup(CLIENT_THREAD, new WaarpThreadFactory("LocalWorker"));
659         localTransaction = new LocalTransaction();
660         WaarpLoggerFactory.setDefaultFactory(WaarpLoggerFactory.getDefaultFactory());
661         if (warnOnStartup) {
662             logger.warn("Server Thread: " + SERVER_THREAD + " Client Thread: " + CLIENT_THREAD
663                     + " Runner Thread: " + RUNNER_THREAD);
664         } else {
665             logger.info("Server Thread: " + SERVER_THREAD + " Client Thread: " + CLIENT_THREAD
666                     + " Runner Thread: " + RUNNER_THREAD);
667         }
668         logger.info("Current launched threads: " + ManagementFactory.getThreadMXBean().getThreadCount());
669         if (useLocalExec) {
670             LocalExecClient.initialize();
671         }
672         configured = true;
673     }

681     /**
682      * Startup the server
683      * 
684      * @throws WaarpDatabaseSqlException
685      * @throws WaarpDatabaseNoConnectionException
686      */
687     public void serverStartup() throws WaarpDatabaseNoConnectionException,
688             WaarpDatabaseSqlException {
689         isServer = true;
690         if (blacklistBadAuthent) {
691             blacklistBadAuthent = !DbHostAuth.hasProxifiedHosts(DbConstant.admin.session);
692         }
693         shutdownConfiguration.timeout = TIMEOUTCON;
694         if (timeLimitCache < TIMEOUTCON * 10) {
695             timeLimitCache = TIMEOUTCON * 10;
696             DbTaskRunner.updateLruCacheTimeout(timeLimitCache);
697         }
698         R66ShutdownHook.addShutdownHook();

701             logger.error(Messages.getString("Configuration.NoSSL")); //$NON-NLS-1$

703         }

705         serverPipelineInit();
706         r66Startup();
707         startHttpSupport();
708         startMonitoring();
709         launchStatistics();
710         startRestSupport();

725         logger.info(Messages.getString("Configuration.Start") + SERVER_PORT + ":" + useNOSSL + ":" + HOST_ID + //$NON-NLS-1$

730         serverChannelGroup = new DefaultChannelGroup("OpenR66", subTaskGroup.next());
731         if (useNOSSL) {
732             serverBootstrap = new ServerBootstrap();
733             WaarpNettyUtil.setServerBootstrap(serverBootstrap, bossGroup, workerGroup, (int) TIMEOUTCON);
734             networkServerInitializer = new NetworkServerInitializer(true);
735             serverBootstrap.childHandler(networkServerInitializer);
736             ChannelFuture future = serverBootstrap.bind(new InetSocketAddress(SERVER_PORT)).awaitUninterruptibly();
737             if (future.isSuccess()) {

739                 serverChannelGroup.add(bindNoSSL);

742             }
743         } else {
744             networkServerInitializer = null;
745             logger.warn(Messages.getString("Configuration.NOSSLDeactivated")); //$NON-NLS-1$

749             serverSslBootstrap = new ServerBootstrap();

751             networkSslServerInitializer = new NetworkSslServerInitializer(false);
752             serverSslBootstrap.childHandler(networkSslServerInitializer);
753             ChannelFuture future = serverSslBootstrap.bind(new InetSocketAddress(SERVER_SSLPORT))
754                     .awaitUninterruptibly();
755             if (future.isSuccess()) {

757                 serverChannelGroup.add(bindSSL);
758             } else {
759                 logger.warn(Messages.getString("Configuration.SSLMODEDeactivated")); //$NON-NLS-1$
760             }
761         } else {
762             networkSslServerInitializer = null;
763             logger.warn(Messages.getString("Configuration.SSLMODEDeactivated")); //$NON-NLS-1$
764         }
765 
766         // Factory for TrafficShapingHandler
767         globalTrafficShapingHandler = new GlobalTrafficHandler(subTaskGroup, serverGlobalWriteLimit,
768                 serverGlobalReadLimit, serverChannelWriteLimit, serverChannelReadLimit, delayLimit);
769         this.constraintLimitHandler.setHandler(globalTrafficShapingHandler);
770 
771         // Now start the InternalRunner
772         internalRunner = new InternalRunner();
773 
774         if (thriftport > 0) {
775             thriftService = new R66ThriftServerService(new WaarpFuture(true), thriftport);
776             execOtherWorker.execute(thriftService);
777             thriftService.awaitInitialization();

783     public void startHttpSupport() {
784         // Now start the HTTP support
785         logger.info(Messages.getString("Configuration.HTTPStart") + SERVER_HTTPPORT + //$NON-NLS-1$
786                 " HTTPS: " + SERVER_HTTPSPORT);
787         httpChannelGroup = new DefaultChannelGroup("HttpOpenR66", subTaskGroup.next());
788         // Configure the server.
789         httpBootstrap = new ServerBootstrap();
790         WaarpNettyUtil.setServerBootstrap(httpBootstrap, httpBossGroup, httpWorkerGroup, (int) TIMEOUTCON);
791         // Set up the event pipeline factory.
792         httpBootstrap.childHandler(new HttpInitializer(useHttpCompression));
793         // Bind and start to accept incoming connections.
794         if (SERVER_HTTPPORT > 0) {
795             ChannelFuture future = httpBootstrap.bind(new InetSocketAddress(SERVER_HTTPPORT)).awaitUninterruptibly();
796             if (future.isSuccess()) {
797                 httpChannelGroup.add(future.channel());
798             }
799         }
800         // Now start the HTTPS support
801         // Configure the server.
802         httpsBootstrap = new ServerBootstrap();
803         // Set up the event pipeline factory.
804         WaarpNettyUtil.setServerBootstrap(httpsBootstrap, httpBossGroup, httpWorkerGroup, (int) TIMEOUTCON);
805         if (httpModel == 0) {
806             httpsBootstrap.childHandler(new HttpSslInitializer(useHttpCompression, false));
807         } else {
808             // Default
809             httpsBootstrap.childHandler(new HttpReponsiveSslInitializer(useHttpCompression, false));
810         }
811         // Bind and start to accept incoming connections.
812         if (SERVER_HTTPSPORT > 0) {
813             ChannelFuture future = httpsBootstrap.bind(new InetSocketAddress(SERVER_HTTPSPORT)).awaitUninterruptibly();
814             if (future.isSuccess()) {
815                 httpChannelGroup.add(future.channel());
816             }
817         }

1037     /**
1038      * Reset the global monitor for bandwidth limitation and change future channel monitors
1039      * 
1040      * @param writeGlobalLimit
1041      * @param readGlobalLimit
1042      * @param writeSessionLimit
1043      * @param readSessionLimit
1044      * @param delayLimit
1045      */
1046     public void changeNetworkLimit(long writeGlobalLimit, long readGlobalLimit,
1047             long writeSessionLimit, long readSessionLimit, long delayLimit) {
1048         long newWriteLimit = writeGlobalLimit > 1024 ? writeGlobalLimit
1049                 : serverGlobalWriteLimit;
1050         if (writeGlobalLimit <= 0) {
1051             newWriteLimit = 0;
1052         }
1053         long newReadLimit = readGlobalLimit > 1024 ? readGlobalLimit : serverGlobalReadLimit;
1054         if (readGlobalLimit <= 0) {
1055             newReadLimit = 0;
1056         }
1057         serverGlobalReadLimit = newReadLimit;
1058         serverGlobalWriteLimit = newWriteLimit;
1059         this.delayLimit = delayLimit;
1060         if (globalTrafficShapingHandler != null) {
1061             globalTrafficShapingHandler.configure(serverGlobalWriteLimit, serverGlobalReadLimit, delayLimit);
1062             logger.warn(Messages.getString("Configuration.BandwidthChange"), globalTrafficShapingHandler); //$NON-NLS-1$
1063         }
1064         newWriteLimit = writeSessionLimit > 1024 ? writeSessionLimit
1065                 : serverChannelWriteLimit;
1066         if (writeSessionLimit <= 0) {
1067             newWriteLimit = 0;
1068         }
1069         newReadLimit = readSessionLimit > 1024 ? readSessionLimit
1070                 : serverChannelReadLimit;
1071         if (readSessionLimit <= 0) {
1072             newReadLimit = 0;
1073         }
1074         serverChannelReadLimit = newReadLimit;
1075         serverChannelWriteLimit = newWriteLimit;
1076         if (globalTrafficShapingHandler != null && globalTrafficShapingHandler instanceof GlobalChannelTrafficShapingHandler) {
1077             ((GlobalChannelTrafficShapingHandler) globalTrafficShapingHandler).configureChannel(serverChannelWriteLimit, serverChannelReadLimit);
1078         }
1079         anyBandwidthLimitation = (serverGlobalReadLimit > 0 || serverGlobalWriteLimit > 0 ||
1080                 serverChannelReadLimit > 0 || serverChannelWriteLimit > 0);
1081     }

1083     /**
1084      * Compute number of threads for both client and server from the real number of available
1085      * processors (double + 1) if the value is less than 32 threads else (available +1).
1086      */
1087     public void computeNbThreads() {
1088         int nb = Runtime.getRuntime().availableProcessors() * 2 + 1;
1089         if (nb > 32) {
1090             nb = Runtime.getRuntime().availableProcessors() + 1;
1091         }
1092         if (SERVER_THREAD <= 0 || SERVER_THREAD > nb) {
1093             logger.info(Messages.getString("Configuration.ThreadNumberChange") + nb); //$NON-NLS-1$
1094             SERVER_THREAD = nb;
1095             CLIENT_THREAD = SERVER_THREAD * 10;
1096         } else if (CLIENT_THREAD < nb) {
1097             CLIENT_THREAD = nb;
1098         }
1099     }
