132 	@Override
133     protected void checkConnection(Channel channel) throws HttpInvalidAuthenticationException {
134 		logger.debug("Request: {} ### {}",arguments,response);
135 		String user = null;
136 		String key = null;
137 		if (checkAuthent) {
138 			user = arguments.getXAuthUser();
139 			if (user == null || user.isEmpty()) {
140 				status = HttpResponseStatus.UNAUTHORIZED;
141 				throw new HttpInvalidAuthenticationException("Empty Authentication");
142 			}
143 			DbHostAuth host;
144 			try {
145 				host = new DbHostAuth(DbConstant.admin.session, user);
146 				key = new String(host.getHostkey(), WaarpStringUtils.UTF8);
147 			} catch (WaarpDatabaseException e) {
148 				// might be global Admin
149 				if (user.equals(Configuration.configuration.ADMINNAME)) {
150 					key = new String(Configuration.configuration.getSERVERADMINKEY(), WaarpStringUtils.UTF8);
151 				}
152 			}
153 			if (key == null || key.isEmpty()) {
154 				status = HttpResponseStatus.UNAUTHORIZED;
155 				throw new HttpInvalidAuthenticationException("Wrong Authentication");
156 			}
157 			arguments.checkBaseAuthent(key, checkTime);
158 		} else {
159 			// User set only for right access, not for signature check
160 			user = Configuration.configuration.ADMINNAME;
161 			arguments.checkBaseAuthent(null, checkTime);
162 		}
163 		serverHandler.newSession();
164 		R66Session session = serverHandler.getSession();
165 		if (! checkAuthent) {
166 			session.getAuth().specialNoSessionAuth(true, Configuration.configuration.HOST_SSLID);
167 		} else {
168 			// we have one DbSession per connection, only after authentication
169 			DbSession temp = dbSessionFromUser.get(user);
170 			if (temp == null) {
171 				try {
172 					temp = new DbSession(DbConstant.admin, false);
173 					dbSessionFromUser.put(user, temp);
174 				} catch (WaarpDatabaseNoConnectionException e) {
175 				}
176 			}
177 			if (temp != null) {
178 				temp.useConnection();
179 				this.dbSession = temp;
180 			}
181 			try {
182 				session.getAuth().connectionHttps(getDbSession(), user, 
183 						FilesystemBasedDigest.passwdCrypt(key.getBytes(WaarpStringUtils.UTF8)));
184 			} catch (Reply530Exception e) {
185 				status = HttpResponseStatus.UNAUTHORIZED;
186 				throw new HttpInvalidAuthenticationException("Wrong Authentication", e);
187 			} catch (Reply421Exception e) {
188 				status = HttpResponseStatus.SERVICE_UNAVAILABLE;
189 				throw new HttpInvalidAuthenticationException("Service unavailable", e);
190 			}
191 		}
192 		arguments.setXAuthRole(session.getAuth().getRole());
193 		arguments.methodFromUri();
194 		arguments.methodFromHeader();
195     }

214 	/**
215 	 * Initialize the REST service (server side) specifying pathTemp
216 	 * @param pathTemp
217 	 */
218 	public static void initializeService(String pathTemp) {
219 		instantiateHandlers();
220         group = Configuration.configuration.getHttpChannelGroup();
221 		// Configure the server.
222         NioServerSocketChannelFactory httpChannelFactory = new NioServerSocketChannelFactory(
223 				Configuration.configuration.getExecutorService(),
224 				Configuration.configuration.getExecutorService(),
225 				Configuration.configuration.SERVER_THREAD);
226         ServerBootstrap httpBootstrap = new ServerBootstrap(httpChannelFactory);
227 		// Set up the event pipeline factory.
228         HttpRestR66Handler.initialize(pathTemp);
229 
230 		if (Configuration.configuration.REST_SSL) {
231 			httpBootstrap.setPipelineFactory(new HttpRestR66PipelineFactory(false, Configuration.waarpSslContextFactory));
232 		} else {
233 			httpBootstrap.setPipelineFactory(new HttpRestR66PipelineFactory(false, null));
234 		}
235 		httpBootstrap.setOption("child.tcpNoDelay", true);
236 		httpBootstrap.setOption("child.keepAlive", true);
237 		httpBootstrap.setOption("child.reuseAddress", true);
238 		httpBootstrap.setOption("child.connectTimeoutMillis", Configuration.configuration.TIMEOUTCON);
239 		httpBootstrap.setOption("tcpNoDelay", true);
240 		httpBootstrap.setOption("reuseAddress", true);
241 		httpBootstrap.setOption("connectTimeoutMillis", Configuration.configuration.TIMEOUTCON);
242 		// Bind and start to accept incoming connections.
243 		group.add(httpBootstrap.bind(new InetSocketAddress(Configuration.configuration.REST_PORT)));
244 	}
