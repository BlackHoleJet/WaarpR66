34  /**
35   * Ftp Transfer task: synchronous<br>
36   * 
37   * Result of arguments will be as FTP command.<br>
38   * Format is the following:<br>
39   * "-file filepath <br>
40   * -to requestedHost <br>
41   * -port port <br>
42   * -user user <br>
43   * -pwd pwd <br>
44   * [-account account] <br>
45   * [-mode active/passive] <br>
46   * [-ssl no/implicit/explicit]<br>
47   * [-cwd remotepath]  <br>
48   * [-digest (crc,md5,sha1)] <br>
49   * [-pre extraCommand1 with ',' as separator of arguments] <br>
50   * -command command from (get,put,append) <br>
51   * [-post extraCommand2 with ',' as separator of arguments]" <br>
52   * <br>
53   * <br>
54   * The order of commands will be:<br>
55   * 1) connection to requestHost on port (if ssl native => using native ssl link)<br>
56   * 2) User user<br>
57   * 3) PASS pwd<br>
58   * 4) if account => ACCT account<br>
59   * 5) if -ssl & auth => AUTH, PBSZ 0, PROT P <br>
60   * 6) if passive => PASV<br> 
61   * 7) CWD remotepath;  if error => MKD remotepath then CWD remotepath (and ignoring any error)<br>
62   * 8) if pre => extraCommand1 with ',' replaced by ' ' (note: do not use standard commands from FTP like 
63   * ACCT,PASS,REIN,USER,APPE,STOR,STOU,RETR,RMD,RNFR,RNTO,ABOR,CWD,CDUP,MODE,PASV,PORT,STRU,TYPE,MDTM,MLSD,MLST,SIZE,AUTH) <br>
64   * 9) BINARY (binary format)<br>
65   * 10) if get => RETR filepath.basename ; if put => STOR filepath ; if append => APPE filepath.basename<br>
66   * 11) if digest & get/put/append & remote site compatible with XCRC,XMD5,XSHA1 
67   * => FEAT (parsing if found corresponding XCRC,XMD5,XSHA1) ; then XCRC/XMD5/XSHA1 filepath.basename ; 
68   * then locally comparing this XCRC/XMD5/XSHA1 with the local file<br>
69   * 12) if post => extraCommand2 with ',' replaced by ' ' (note: do not use standard commands from FTP like 
70   * ACCT,PASS,REIN,USER,APPE,STOR,STOU,RETR,RMD,RNFR,RNTO,ABOR,CWD,CDUP,MODE,PASV,PORT,STRU,TYPE,MDTM,MLSD,MLST,SIZE,AUTH)<br>
71   * 13) QUIT<br>
72   * 
73   * @author Frederic Bregier
74   * 
75   */
76  public class FtpTransferTask extends AbstractTask {
77  	/**
78  	 * Internal Logger
79  	 */
80  	private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
81  			.getLogger(FtpTransferTask.class);
82  
83  	/**
84  	 * @param argRule
85  	 * @param delay
86  	 * @param argTransfer
87  	 * @param session
88  	 */
89  	public FtpTransferTask(String argRule, int delay, String argTransfer,
90  			R66Session session) {
91  		super(TaskType.FTP, delay, argRule, argTransfer, session);
92  	}
93  
94  	@Override
95  	public void run() {
96  		logger.info("FtpTransfer with " + argRule + ":" + argTransfer + " and {}",
97  				session);
98  		String finalname = argRule;
99  		finalname = getReplacedValue(finalname, argTransfer.split(" "));
100 		String[] args = finalname.split(" ");
101 		if (args.length < 10) {
102 			OpenR66RunnerErrorException exception = new OpenR66RunnerErrorException("Not enough argument in Transfer");
103 			R66Result result = new R66Result(exception, session, false, ErrorCode.CommandNotFound, session.getRunner());
104 			logger.error("Not enough arguments");
105 			futureCompletion.setResult(result);
106 			futureCompletion.setFailure(exception);
107 			return;
108 		}
109 		String filepath = null;
110 		String filename = null;
111 		String requested = null;
112 		int port = 21;
113 		String user = null;
114 		String pwd = null;
115 		String acct = null;
116 		boolean isPassive = true;
117 		int ssl = 0; // -1 native, 1 auth
118 		String cwd = null;
119 		int digest = 0; // 1 CRC, 2 MD5, 3 SHA1
120 		String command = null;
121 		int codeCommand = 0; // -1 get, 1 put, 2 append
122 		String preArgs = null;
123 		String postArgs = null;
124 		/*
125 		 * "-file filepath <br>
126 		 * -to requestedHost <br>
127 		 * -port port <br>
128 		 * -user user <br>
129 		 * -pwd pwd <br>
130 		 * [-account account] <br>
131 		 * [-mode active/passive] <br>
132 		 * [-ssl no/implicit/explicit]<br>
133 		 * [-cwd remotepath]  <br>
134 		 * [-digest (crc,md5,sha1)] <br>
135 		 * [-pre extraCommand1 with ',' as separator of arguments] <br>
136 		 * -command command from (get,put,append) <br>
137 		 * [-post extraCommand2 with ',' as separator of arguments]" <br>
138 		 */
139 		for (int i = 0; i < args.length; i++) {
140 			if (args[i].equalsIgnoreCase("-file")) {
141 				i++;
142 				filepath = args[i];
143 				filename = new File(filepath).getName();
144 			} else if (args[i].equalsIgnoreCase("-to")) {
145 				i++;
146 				requested = args[i];
147 			} else if (args[i].equalsIgnoreCase("-port")) {
148 				i++;
149 				port = Integer.parseInt(args[i]);
150 			} else if (args[i].equalsIgnoreCase("-user")) {
151 				i++;
152 				user = args[i];
153 			} else if (args[i].equalsIgnoreCase("-pwd")) {
154 				i++;
155 				pwd = args[i];
156 			} else if (args[i].equalsIgnoreCase("-account")) {
157 				i++;
158 				acct = args[i];
159 			} else if (args[i].equalsIgnoreCase("-mode")) {
160 				i++;
161 				isPassive = (args[i].equalsIgnoreCase("passive"));
162 			} else if (args[i].equalsIgnoreCase("-ssl")) {
163 				i++;
164 				if (args[i].equalsIgnoreCase("implicit")) {
165 					ssl = -1;
166 				} else if (args[i].equalsIgnoreCase("explicit")) {
167 					ssl = 1;
168 				} else {
169 					ssl = 0;
170 				}
171 			} else if (args[i].equalsIgnoreCase("-cwd")) {
172 				i++;
173 				cwd = args[i];
174 			} else if (args[i].equalsIgnoreCase("-digest")) {
175 				i++;
176 				if (args[i].equalsIgnoreCase("crc")) {
177 					digest = 1;
178 				} else if (args[i].equalsIgnoreCase("md5")) {
179 					digest = 2;
180 				} else if (args[i].equalsIgnoreCase("sha1")) {
181 					digest = 3;
182 				} else {
183 					digest = 0;
184 				}
185 			} else if (args[i].equalsIgnoreCase("-pre")) {
186 				i++;
187 				preArgs = args[i].replace(',', ' ');
188 			} else if (args[i].equalsIgnoreCase("-post")) {
189 				i++;
190 				postArgs = args[i].replace(',', ' ');
191 			} else if (args[i].equalsIgnoreCase("-command")) {
192 				i++;
193 				command = args[i];
194 				// get,put,append,list
195 				// -1 get, 1 put, 2 append
196 				if (command.equalsIgnoreCase("get")) {
197 					codeCommand = -1;
198 				} else if (command.equalsIgnoreCase("put")) {
199 					codeCommand = 1;
200 				} else if (command.equalsIgnoreCase("append")) {
201 					codeCommand = 2;
202 				} else {
203 					// error
204 					codeCommand = 0;
205 				}
206 			}
207 		}
208 		if (filepath == null || requested == null || port <= 0 || user == null || pwd == null ||
209 				codeCommand == 0) {
210 			OpenR66RunnerErrorException exception = new OpenR66RunnerErrorException("Not enough argument in Transfer");
211 			R66Result result = new R66Result(exception, session, false, ErrorCode.CommandNotFound, session.getRunner());
212 			int code = 0 +
213 					(filepath == null ? 1 : 0) + (requested == null ? 10 : 0) + 
214 					(port <= 0 ? 100 : 0) + (user == null ? 1000 : 0) + (pwd == null ? 10000 : 0) +
215 					(codeCommand == 0 ? 100000 : 0);
216 			logger.error("Not enough arguments: "+code);
217 			futureCompletion.setResult(result);
218 			futureCompletion.setFailure(exception);
219 			return;
220 		}
221 		WaarpFtp4jClient ftpClient = 
222 				new WaarpFtp4jClient(requested, port, user, pwd, acct, isPassive, ssl, 
223 						5000, (int) Configuration.configuration.TIMEOUTCON);
224 		boolean status = false;
225 		for (int i = 0; i < Configuration.RETRYNB; i++) {
226 			if (ftpClient.connect()) {
227 				status = true;
228 				break;
229 			}
230 		}
231 		if (! status) {
232 			OpenR66RunnerErrorException exception = new OpenR66RunnerErrorException("Cannot connect to remote FTP host");
233 			R66Result result = new R66Result(exception, session, false, ErrorCode.ConnectionImpossible, session.getRunner());
234 			futureCompletion.setResult(result);
235 			futureCompletion.setFailure(exception);
236 			logger.error(ftpClient.getResult());
237 			return;
238 		}
239 		try {
240 			if (cwd != null && !ftpClient.changeDir(cwd)) {
241 				ftpClient.makeDir(cwd);
242 				if (!ftpClient.changeDir(cwd)) {
243 					logger.warn("Cannot change od directory: "+cwd);
244 				}
245 			}
246 			if (preArgs != null) {
247 				String [] result = ftpClient.executeCommand(preArgs);
248 				for (String string : result) {
249 					logger.debug("PRE: "+string);
250 				}
251 			}
252 			if (! ftpClient.transferFile(filepath, filename, codeCommand)) {
253 				OpenR66RunnerErrorException exception = new OpenR66RunnerErrorException("Cannot transfert file from/to remote FTP host");
254 				R66Result result = new R66Result(exception, session, false, ErrorCode.TransferError, session.getRunner());
255 				futureCompletion.setResult(result);
256 				futureCompletion.setFailure(exception);
257 				logger.error(ftpClient.getResult());
258 				return;
259 			}
260 			if (digest > 0) {
261 				// digest check
262 				String params = null;
263 				DigestAlgo algo = null;
264 				switch (digest) {
265 					case 1: // CRC
266 						params = "XCRC ";
267 						algo = DigestAlgo.CRC32;
268 						break;
269 					case 2: // MD5
270 						params = "XMD5 ";
271 						algo = DigestAlgo.MD5;
272 						break;
273 					case 3: // SHA1
274 						params = "XSHA1 ";
275 						algo = DigestAlgo.SHA1;
276 						break;
277 				}
278 				params += filename;
279 				String [] values = ftpClient.executeCommand(params);
280 				String hashresult = null;
281 				if (values != null) {
282 					values = values[0].split(" ");
283 					hashresult = (values.length > 3 ? values[1] : values[0]);
284 				}
285 				if (hashresult == null) {
286 					OpenR66RunnerErrorException exception = new OpenR66RunnerErrorException("Hash cannot be computed while FTP transfer is done");
287 					R66Result result = new R66Result(exception, session, false, ErrorCode.TransferError, session.getRunner());
288 					futureCompletion.setResult(result);
289 					futureCompletion.setFailure(exception);
290 					logger.error("Hash cannot be computed: "+ftpClient.getResult());
291 					return;
292 				}
293 				// now check locally
294 				String hash;
295 				try {
296 					hash = FilesystemBasedDigest.getHex(
297 							FilesystemBasedDigest.getHash(new File(filepath), false, algo));
298 				} catch (IOException e) {
299 					hash = null;
300 				}
301 				if (hash == null || (!hash.equalsIgnoreCase(hashresult))) {
302 					OpenR66RunnerErrorException exception = new OpenR66RunnerErrorException("Hash not equal while FTP transfer is done");
303 					R66Result result = new R66Result(exception, session, false, ErrorCode.TransferError, session.getRunner());
304 					futureCompletion.setResult(result);
305 					futureCompletion.setFailure(exception);
306 					logger.error("Hash not equal: "+ftpClient.getResult());
307 					return;
308 				}
309 			}
310 			if (postArgs != null) {
311 				String [] result = ftpClient.executeCommand(postArgs);
312 				for (String string : result) {
313 					logger.debug("POST: "+string);
314 				}
315 			}
316 		} finally {
317 			ftpClient.logout();
318 		}
319 		R66Result result = new R66Result(session, false, ErrorCode.TransferOk, session.getRunner());
320 		futureCompletion.setResult(result);
321 		logger.info("FTP transfer in\n    SUCCESS\n    " + session.getRunner().toShortString() +
322 					"\n    <REMOTE>" + requested + "</REMOTE>");
323 		futureCompletion.setSuccess();
324 	}
