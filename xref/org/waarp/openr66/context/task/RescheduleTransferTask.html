32  /**
33   * Reschedule Transfer task to a time delayed by the specified number of milliseconds, if the error
34   * code is one of the specified codes and the optional intervals of date are compatible with the new
35   * time schedule<br>
36   * <br>
37   * 
38   * Result of arguments will be as following options (the two first are mandatory):<br>
39   * <br>
40   * 
41   * "-delay ms" where ms is the added number of ms on current time before retry on schedule<br>
42   * <br>
43   * "-case errorCode,errorCode,..." where errorCode is one of the following error of the current
44   * transfer (either literal or code in 1 character:<br>
45   * ConnectionImpossible(C), ServerOverloaded(l), BadAuthent(A), ExternalOp(E), TransferError(T),
46   * MD5Error(M), Disconnection(D), RemoteShutdown(r), FinalOp(F), Unimplemented(U), Shutdown(S),
47   * RemoteError(R), Internal(I), StoppedTransfer(H), CanceledTransfer(K), Warning(W), Unknown(-),
48   * QueryAlreadyFinished(Q), QueryStillRunning(s), NotKnownHost(N), QueryRemotelyUnknown(u),
49   * FileNotFound(f), CommandNotFound(c), PassThroughMode(p)<br>
50   * <br>
51   * "-between start;end" and/or "-notbetween start;end" (multiple times are allowed, start or end can
52   * be not set) and where start and stop are in the following format:<br>
53   * Yn:Mn:Dn:Hn:mn:Sn where n is a number for each time specification, each specification is
54   * optional, as Y=Year, M=Month, D=Day, H=Hour, m=minute, s=second.<br>
55   * Format can be X+n, X-n, X=n or Xn where X+-n means adding/subtracting n to current date value,
56   * while X=n or Xn means setting exact value<br>
57   * If one time specification is not set, it is based on the current date.<br>
58   * <br>
59   * "-count limit" will be the limit of retry. The value limit is taken from the "info on transfer" and not from the rule
60   * as "#CPTLIMIT#newlimit#CPTLIMIT#". Each time this function is called, the 
61   * limit value will be replaced as newlimit = limit - 1 in the "info of transfer" as "#CPTLIMIT#newlimit#CPTLIMIT#".<br>
62   * To ensure correctness, the value must be in the "info of transfer" since this value will be
63   * changed statically in the "info of transfer". If taken from the rule, it will be wrong since 
64   * the value will never decrease. However, a value must be setup in the rule in order to reset the value
65   * when the count reach 0.
66   * <br>So in the rule, "-count resetlimit" must be present, where resetlimit will be
67   * the new value set when the limit reach 0, and in the "info on transfer",
68   * "#CPTLIMIT#limit#CPTLIMIT#" must be present. If one is missing, the condition is not applied.
69   * <br>
70   * <br>
71   * If "-notbetween" is specified, the planned date must not be in the area.<br>
72   * If "-between" is specified, the planned date must be found in any such specified areas (could be
73   * in any of the occurrence). If not specified, it only depends on "-notbetween".<br>
74   * If none is specified, the planned date is always valid.<br>
75   * <br>
76   * 
77   * Note that if a previous called to a reschedule was done for this attempt and was successful, the
78   * following calls will be ignored.<br>
79   * <br>
80   * <B>Important note: any subsequent task will be ignored and not executed once the reschedule is accepted.</B><br>
81   * <br>
82   * In case start > end, end will be +1 day<br>
83   * In case start and end < current planned date, both will have +1 day.<br>
84   * <br>
85   * 
86   * Example: -delay 3600000 -case ConnectionImpossible,ServerOverloaded,Shutdown -notbetween
87   * H7:m0:S0;H19:m0:S0 -notbetween H1:m0:S0;H=3:m0:S0<br>
88   * means retry in case of error during initialization of connection in 1 hour if not between 7AM to
89   * 7PM and not between 1AM to 3AM.<br>
90   * 
91   * @author Frederic Bregier
92   * 
93   */

126 	@Override
127 	public void run() {
128 		logger.info("Reschedule with " + argRule + ":" + argTransfer +
129 				" and {}", session);
130 		DbTaskRunner runner = session.getRunner();
131 		if (runner == null) {
132 			futureCompletion.setFailure(new OpenR66RunnerErrorException(
133 					"No valid runner in Reschedule"));
134 			return;
135 		}
136 		if (runner.isRescheduledTransfer()) {
137 			// Already rescheduled so ignore
138 			R66Result result = new R66Result(session, false, ErrorCode.Warning,
139 					runner);
140 			futureCompletion.setResult(result);
141 			logger.warn("Transfer already Rescheduled: " + runner.toShortString());
142 			futureCompletion.setSuccess();
143 			return;
144 		}
145 		if (runner.isSelfRequested()) {
146 			// Self Requested Request so reschedule is ignored
147 			R66Result result = new R66Result(session, false, ErrorCode.LoopSelfRequestedHost,
148 					runner);
149 			futureCompletion.setResult(result);
150 			futureCompletion.setFailure(new OpenR66RunnerErrorException(
151 					"No valid runner in Reschedule since Self Requested"));
152 			return;
153 		}
154 		String finalname = argRule;
155 		finalname = getReplacedValue(finalname, argTransfer.split(" "));
156 		String[] args = finalname.split(" ");
157 		if (args.length < 4) {
158 			R66Result result = new R66Result(session, false, ErrorCode.Warning,
159 					runner);
160 			futureCompletion.setResult(result);
161 			logger.warn("Not enough argument in Reschedule: " + runner.toShortString());
162 			futureCompletion.setSuccess();
163 			return;
164 		}
165 		if (!validateArgs(args)) {
166 			R66Result result = new R66Result(session, false, ErrorCode.Warning,
167 					runner);
168 			futureCompletion.setResult(result);
169 			logger.warn("Reschedule unallowed due to argument: " + runner.toShortString());
170 			futureCompletion.setSuccess();
171 			return;
172 		}
173 		String newFileInformation = null;
174 		if (countUsed) {
175 			limitCount --;
176 			if (limitCount >= 0) {
177 				// restart is allowed so resetting to new limitCount 
178 				resetCount = limitCount;
179 			}
180 			newFileInformation = resetInformation(runner, resetCount);
181 			if (limitCount < 0) {
182 				// Must not reschedule since limit is reached
183 				runner.setFileInformation(newFileInformation);
184 				try {
185 					runner.saveStatus();
186 				} catch (OpenR66RunnerErrorException e) {
187 				}
188 				R66Result result = new R66Result(session, false, ErrorCode.Warning,
189 						runner);
190 				futureCompletion.setResult(result);
191 				logger.warn("Reschedule unallowed due to limit reached: " + runner.toShortString());
192 				futureCompletion.setSuccess();
193 				return;
194 			}
195 		}
196 		Timestamp start = new Timestamp(newdate);
197 		try {
198 			if (runner.setStart(start)) {
199 				runner.setFileInformation(newFileInformation);
200 				runner.saveStatus();
201 			}
202 		} catch (OpenR66RunnerErrorException e) {
203 			logger.error(
204 					"Prepare transfer in\n    FAILURE\n     " +
205 							runner.toShortString() + "\n    <AT>" +
206 							(new Date(newdate)).toString() + "</AT>", e);
207 			futureCompletion.setFailure(new OpenR66RunnerErrorException(
208 					"Reschedule failed: " + e.getMessage(), e));
209 			return;
210 		}
211 		runner.setRescheduledTransfer();
212 		R66Result result = new R66Result(session, false, ErrorCode.Warning,
213 				runner);
214 		futureCompletion.setResult(result);
215 		logger.warn("Reschedule transfer in\n    SUCCESS\n    " +
216 				runner.toShortString() + "\n    <AT>" +
217 				(new Date(newdate)).toString() + "</AT>");
218 		futureCompletion.setSuccess();
219 	}

221 	protected String resetInformation(DbTaskRunner runner, int value) {
222 		String info = runner.getFileInformation();
223 		int pos = info.indexOf(CPTLIMIT);
224 		if (pos >= 0) {
225 			pos += CPTLIMIT.length();
226 			int endpos = info.indexOf(CPTLIMIT, pos);
227 			if (endpos >= 0) {
228 				// now string replace
229 				String limit = info.substring(pos, endpos);
230 				return info.replaceFirst(CPTLIMIT+limit+CPTLIMIT, CPTLIMIT+value+CPTLIMIT);
231 			}
232 		}
233 		return info;
234 	}

236 	protected boolean validateArgs(String[] args) {
237 		boolean validCode = false;
238 		for (int i = 0; i < args.length; i++) {
239 			if (args[i].equalsIgnoreCase("-delay")) {
240 				i++;
241 				try {
242 					newdate = Long.parseLong(args[i]);
243 				} catch (NumberFormatException e) {
244 					logger.warn("Bad Long format: args[i]");
245 					return false;
246 				}
247 			} else if (args[i].equalsIgnoreCase("-case")) {
248 				i++;
249 				if (!validCode) {
250 					String[] codes = args[i].split(",");
251 					for (int j = 0; j < codes.length; j++) {
252 						ErrorCode code = ErrorCode.getFromCode(codes[j]);
253 						if (session.getLocalChannelReference().getCurrentCode() == code) {
254 							logger.debug("Code valid: " + code);
255 							validCode = true;
256 						}
257 					}
258 				}
259 			} else if (args[i].equalsIgnoreCase("-count")) {
260 				i++;
261 				try {
262 					resetCount = Integer.parseInt(args[i]);
263 				} catch (NumberFormatException e) {
264 					logger.warn("ResetLimit is not an integer: "+args[i]);
265 					countUsed = false;
266 					return false;
267 				}
268 				String []find = argTransfer.split(CPTLIMIT);
269 				if (find.length > 1) {
270 					// should be 2nd value
271 					try {
272 						limitCount = Integer.parseInt(find[1]);
273 					} catch (NumberFormatException e) {
274 						logger.warn("Limit is not an integer: "+find[1]);
275 						countUsed = false;
276 						return false;
277 					}
278 				}
279 				countUsed = true;
280 			}
281 		}
282 		// now we have new delay plus code
283 		if (!validCode) {
284 			logger.warn("No valid Code found");
285 			return false;
286 		}
287 		if (newdate <= 0) {
288 			logger.warn("Delay is negative: " + newdate);
289 			return false;
290 		}
291 		newdate += System.currentTimeMillis();
292 		newDate = Calendar.getInstance();
293 		newDate.setTimeInMillis(newdate);
294 		boolean betweenTest = false;
295 		boolean betweenResult = false;
296 		for (int i = 0; i < args.length; i++) {
297 			if (args[i].equalsIgnoreCase("-notbetween")) {
298 				i++;
299 				String[] elmts = args[i].split(";");
300 				boolean startModified = false;
301 				String[] values = elmts[0].split(":");
302 				Calendar start = getCalendar(values);
303 				if (start != null) {
304 					startModified = true;
305 				} else {
306 					start = Calendar.getInstance();
307 				}
308 				boolean stopModified = false;
309 				values = elmts[1].split(":");
310 				Calendar stop = getCalendar(values);
311 				if (stop != null) {
312 					stopModified = true;
313 				} else {
314 					stop = Calendar.getInstance();
315 				}
316 				logger.debug("Dates before check: Not between " + start.getTime() + " and "
317 						+ stop.getTime());
318 				// Check that start < stop
319 				if (start.compareTo(stop) > 0) {
320 					// no so add 24H to stop
321 					stop.add(Calendar.DAY_OF_MONTH, 1);
322 				}
323 				// Check that start and stop > newDate (only start is checked since start <= stop)
324 				if (start.compareTo(newDate) < 0) {
325 					start.add(Calendar.DAY_OF_MONTH, 1);
326 					stop.add(Calendar.DAY_OF_MONTH, 1);
327 				}
328 				logger.debug("Dates after check: Not between " + start.getTime() + " and "
329 						+ stop.getTime());
330 				if (!startModified) {
331 					if (newDate.compareTo(stop) < 0) {
332 						logger.debug("newDate: " + newDate.getTime() + " Should not be between "
333 								+ start.getTime() + " and " + stop.getTime());
334 						return false;
335 					}
336 				} else if (start.compareTo(newDate) < 0) {
337 					if ((!stopModified) || (newDate.compareTo(stop) < 0)) {
338 						logger.debug("newDate: " + newDate.getTime() + " Should not be between "
339 								+ start.getTime() + " and " + stop.getTime());
340 						return false;
341 					}
342 				}
343 			} else if (args[i].equalsIgnoreCase("-between")) {
344 				i++;
345 				betweenTest = true;
346 				String[] elmts = args[i].split(";");
347 				boolean startModified = false;
348 				String[] values = elmts[0].split(":");
349 				Calendar start = getCalendar(values);
350 				if (start != null) {
351 					startModified = true;
352 				} else {
353 					start = Calendar.getInstance();
354 				}
355 				boolean stopModified = false;
356 				values = elmts[1].split(":");
357 				Calendar stop = getCalendar(values);
358 				if (stop != null) {
359 					stopModified = true;
360 				} else {
361 					stop = Calendar.getInstance();
362 				}
363 				logger.debug("Dates before check: Between " + start.getTime() + " and "
364 						+ stop.getTime());
365 				// Check that start < stop
366 				if (start.compareTo(stop) > 0) {
367 					// no so add 24H to stop
368 					stop.add(Calendar.DAY_OF_MONTH, 1);
369 				}
370 				// Check that start and stop > newDate (only start is checked since start <= stop)
371 				if (start.compareTo(newDate) < 0) {
372 					start.add(Calendar.DAY_OF_MONTH, 1);
373 					stop.add(Calendar.DAY_OF_MONTH, 1);
374 				}
375 				logger.debug("Dates before check: Between " + start.getTime() + " and "
376 						+ stop.getTime());
377 				if (!startModified) {
378 					if (newDate.compareTo(stop) < 0) {
379 						logger.debug("newDate: " + newDate.getTime() + " is between "
380 								+ start.getTime() + " and " + stop.getTime());
381 						betweenResult = true;
382 					}
383 				} else if (start.compareTo(newDate) < 0) {
384 					if ((!stopModified) || (newDate.compareTo(stop) < 0)) {
385 						logger.debug("newDate: " + newDate.getTime() + " is between "
386 								+ start.getTime() + " and " + stop.getTime());
387 						betweenResult = true;
388 					}
389 				}
390 			}
391 		}
392 		if (betweenTest) {
393 			logger.debug("Since between is specified, do we found newDate: " + newDate.getTime()
394 					+ " Result: " + betweenResult);
395 			return betweenResult;
396 		}
397 		logger.debug("newDate: " + newDate.getTime() + " rescheduled");
398 		return true;
399 	}

401 	/**
402 	 * 
403 	 * @param values
404 	 *            as X+n or X-n or X=n or Xn where X=Y/M/D/H/m/s, n=number and +/- meaning
405 	 *            adding/subtracting from current date and = meaning specific set value
406 	 * @return the Calendar if any specification, or null if no calendar specified
407 	 */
408 	private Calendar getCalendar(String[] values) {
409 		Calendar newcal = Calendar.getInstance();
410 		boolean isModified = false;
411 		for (int j = 0; j < values.length; j++) {
412 			if (values[j].length() > 1) {
413 				int addvalue = 0; // will be different of 0
414 				int value = -1; // will be >= 0
415 				switch (values[j].charAt(0)) {
416 					case '+':
417 						try {
418 							addvalue = Integer.parseInt(values[j].substring(2));
419 						} catch (NumberFormatException e) {
420 							continue;
421 						}
422 						break;
423 					case '-':
424 						try {
425 							addvalue = Integer.parseInt(values[j].substring(1));
426 						} catch (NumberFormatException e) {
427 							continue;
428 						}
429 						break;
430 					case '=':
431 						try {
432 							value = Integer.parseInt(values[j].substring(2));
433 						} catch (NumberFormatException e) {
434 							continue;
435 						}
436 						break;
437 					default: // no sign
438 						try {
439 							value = Integer.parseInt(values[j].substring(1));
440 						} catch (NumberFormatException e) {
441 							continue;
442 						}
443 				}
444 				switch (values[j].charAt(0)) {
445 					case 'Y':
446 						if (value >= 0) {
447 							newcal.set(Calendar.YEAR, value);
448 						} else {
449 							newcal.add(Calendar.YEAR, addvalue);
450 						}
451 						isModified = true;
452 						break;
453 					case 'M':
454 						if (value >= 0) {
455 							newcal.set(Calendar.MONTH, value);
456 						} else {
457 							newcal.add(Calendar.MONTH, addvalue);
458 						}
459 						isModified = true;
460 						break;
461 					case 'D':
462 						if (value >= 0) {
463 							newcal.set(Calendar.DAY_OF_MONTH, value);
464 						} else {
465 							newcal.add(Calendar.DAY_OF_MONTH, addvalue);
466 						}
467 						isModified = true;
468 						break;
469 					case 'H':
470 						if (value >= 0) {
471 							newcal.set(Calendar.HOUR_OF_DAY, value);
472 						} else {
473 							newcal.add(Calendar.HOUR_OF_DAY, addvalue);
474 						}
475 						isModified = true;
476 						break;
477 					case 'm':
478 						if (value >= 0) {
479 							newcal.set(Calendar.MINUTE, value);
480 						} else {
481 							newcal.add(Calendar.MINUTE, addvalue);
482 						}
483 						isModified = true;
484 						break;
485 					case 'S':
486 						if (value >= 0) {
487 							newcal.set(Calendar.SECOND, value);
488 						} else {
489 							newcal.add(Calendar.SECOND, addvalue);
490 						}
491 						isModified = true;
492 						break;
493 				}
494 			}
495 		}
496 		if (isModified)
497 			return newcal;
498 		return null;
499 	}
