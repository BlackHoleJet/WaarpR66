43  /**
44   * Direct Transfer from a client with or without database connection 
45   * or Submit Transfer from a client with database connection 
46   * to transfer files from a spooled directory to possibly multiple hosts at once.<br>
47   * Hosts will have to be separated by ','.<br>
48   * <br>
49   * Mandatory additional elements:<br>
50   * -directory source (directory to spooled on)<br>
51   * -statusfile file (file to use as permanent status (if process is killed or aborts))<br>
52   * -stopfile file (file when created will stop the dameon)<br>
53   * Other options:<br>
54   * -regex regex (regular expression to filter file names from directory source)<br>
55   * -elapse elapse (elapse time between 2 checks of the directory)<br>
56   * -submit (to submit only: default)<br>
57   * -direct (to directly transfer only)<br>
58   * 
59   * @author Frederic Bregier
60   * 
61   */
62  public class SpooledDirectoryTransfer implements Runnable {

147 	@Override
148 	public void run() {
149 		if (submit && ! DbConstant.admin.isConnected) {
150 			logger.error(Messages.getString("SpooledDirectoryTransfer.2")); //$NON-NLS-1$
151 			this.future.cancel();
152 			return;
153 		}
154 		sent = 0;
155 		error  = 0;
156 		final String [] allrhosts = remoteHosts.split(",");
157 		// first check if rule is for SEND
158 		DbRule dbrule = null;
159 		try {
160 			dbrule = new DbRule(DbConstant.admin.session, rulename);
161 		} catch (WaarpDatabaseException e1) {
162 			logger.error(Messages.getString("Transfer.18"), e1); //$NON-NLS-1$
163 			this.future.setFailure(e1);
164 			return;
165 		}
166 		if (dbrule.isRecvMode()) {
167 			logger.error(Messages.getString("SpooledDirectoryTransfer.5")); //$NON-NLS-1$
168 			this.future.cancel();
169 			return;
170 		}
171 		File status = new File(statusFile);
172 		if (status.isDirectory()) {
173 			logger.error(Messages.getString("SpooledDirectoryTransfer.6")); //$NON-NLS-1$
174 			this.future.cancel();
175 			return;
176 		}
177 		File stop = new File(stopFile);
178 		if (stop.isDirectory()) {
179 			logger.error(Messages.getString("SpooledDirectoryTransfer.7")); //$NON-NLS-1$
180 			this.future.cancel();
181 			return;
182 		} else if (stop.exists()) {
183 			logger.warn(Messages.getString("SpooledDirectoryTransfer.8")); //$NON-NLS-1$
184 			this.future.setSuccess();
185 			return;
186 		}
187 		File dir = new File(directory);
188 		if (!dir.isDirectory()) {
189 			logger.error(Messages.getString("SpooledDirectoryTransfer.9")); //$NON-NLS-1$
190 			this.future.cancel();
191 			return;
192 		}
193 		FileFilter filter = null;
194 		if (regexFilter != null) {
195 			filter = new RegexFileFilter(regexFilter);
196 		}
197 		FileMonitorCommand commandValidFile = new FileMonitorCommand() {
198 			public void run(File file) {
199 				for (String host : allrhosts) {
200 					host = host.trim();
201 					if (host != null && ! host.isEmpty()) {
202 						String filename = file.getAbsolutePath();
203 						logger.info("Launch transfer to "+host+" with file "+filename);
204 						R66Future future = new R66Future(true);
205 						String text = null;
206 						if (submit) {
207 							text = "Submit Transfer";
208 							SubmitTransfer transaction = new SubmitTransfer(future,
209 									host, filename, rulename, fileinfo, isMD5, blocksize, 
210 									DbConstant.ILLEGALVALUE, null);
211 							transaction.run();
212 						} else {
213 							text = "Direct Transfer";
214 							DirectTransfer transaction = new DirectTransfer(future,
215 									host, filename, rule, fileInfo, ismd5, block, 
216 									DbConstant.ILLEGALVALUE, networkTransaction);
217 							logger.debug("rhost: "+host+":"+transaction.remoteHost);
218 							transaction.run();
219 						}
220 						future.awaitUninterruptibly();
221 						R66Result result = future.getResult();
222 						if (future.isSuccess()) {
223 							sent++;
224 							DbTaskRunner runner = null;
225 							if (result != null) {
226 								runner = result.runner;
227 								if (runner != null) {
228 									String status = Messages.getString("RequestInformation.Success"); //$NON-NLS-1$
229 									if (runner.getErrorInfo() == ErrorCode.Warning) {
230 										status = Messages.getString("RequestInformation.Warned"); //$NON-NLS-1$
231 									}
232 									logger.warn(text+" in status: "+status+"     "
233 											+ runner.toShortString()
234 											+"     <REMOTE>"+ host+ "</REMOTE>"
235 											+"     <FILEFINAL>"+
236 											(result.file != null ? 
237 													result.file.toString() + "</FILEFINAL>"
238 													: "no file"));
239 									if (nolog && !submit) {
240 										// In case of success, delete the runner
241 										try {
242 											runner.delete();
243 										} catch (WaarpDatabaseException e) {
244 											logger.warn("Cannot apply nolog to     " +
245 													runner.toShortString(),
246 													e);
247 										}
248 									}
249 								} else {
250 									logger.warn(text+Messages.getString("RequestInformation.Success")  //$NON-NLS-1$
251 											+"<REMOTE>" + host + "</REMOTE>");
252 								}
253 							} else {
254 								logger.warn(text+Messages.getString("RequestInformation.Success")  //$NON-NLS-1$
255 										+"<REMOTE>" + host + "</REMOTE>");
256 							}
257 						} else {
258 							error++;
259 							DbTaskRunner runner = null;
260 							if (result != null) {
261 								runner = result.runner;
262 								if (runner != null) {
263 									logger.error(text+Messages.getString("RequestInformation.Failure") +  //$NON-NLS-1$
264 											runner.toShortString() +
265 										"<REMOTE>" + host + "</REMOTE>", future.getCause());
266 								} else {
267 									logger.error(text+Messages.getString("RequestInformation.Failure"),  //$NON-NLS-1$
268 											future.getCause());
269 								}
270 							} else {
271 								logger.error(text+Messages.getString("RequestInformation.Failure") //$NON-NLS-1$
272 										+"<REMOTE>" + host + "</REMOTE>", 
273 										future.getCause());
274 							}
275 						}
276 					}
277 				}
278 			}
279 		};
280 		FileMonitor monitor = new FileMonitor(status, stop, dir, null, elapsed, filter, 
281 				commandValidFile, null);
282 		monitor.start();
283 		monitor.waitForStopFile();
284 		this.future.setSuccess();
285 	}

309 	protected static boolean getParams(String[] args) {
310 		_INFO_ARGS = Messages.getString("SpooledDirectoryTransfer.0"); //$NON-NLS-1$
311 		if (args.length < 11) {
312 			logger
313 					.error(_INFO_ARGS);
314 			return false;
315 		}
316 		if (!FileBasedConfiguration
317 				.setClientConfigurationFromXml(Configuration.configuration, args[0])) {
318 			logger
319 					.error(Messages.getString("Configuration.NeedCorrectConfig")); //$NON-NLS-1$
320 			return false;
321 		}
322 		// Now set default values from configuration
323 		block = Configuration.configuration.BLOCKSIZE;
324 		int i = 1;
325 		try {
326 			for (i = 1; i < args.length; i++) {
327 				if (args[i].equalsIgnoreCase("-to")) {
328 					i++;
329 					rhosts = args[i];
330 				} else if (args[i].equalsIgnoreCase("-directory")) {
331 					i++;
332 					localDirectory = args[i];
333 				} else if (args[i].equalsIgnoreCase("-rule")) {
334 					i++;
335 					rule = args[i];
336 				} else if (args[i].equalsIgnoreCase("-statusfile")) {
337 					i++;
338 					statusfile = args[i];
339 				} else if (args[i].equalsIgnoreCase("-stopfile")) {
340 					i++;
341 					stopfile = args[i];
342 				} else if (args[i].equalsIgnoreCase("-info")) {
343 					i++;
344 					fileInfo = args[i];
345 				} else if (args[i].equalsIgnoreCase("-md5")) {
346 					ismd5 = true;
347 				} else if (args[i].equalsIgnoreCase("-block")) {
348 					i++;
349 					block = Integer.parseInt(args[i]);
350 					if (block < 100) {
351 						logger.error(Messages.getString("AbstractTransfer.1") + block); //$NON-NLS-1$
352 						return false;
353 					}
354 				} else if (args[i].equalsIgnoreCase("-nolog")) {
355 					noLog = true;
356 				} else if (args[i].equalsIgnoreCase("-submit")) {
357 					tosubmit = true;
358 				} else if (args[i].equalsIgnoreCase("-direct")) {
359 					tosubmit = false;
360 				} else if (args[i].equalsIgnoreCase("-regex")) {
361 					i++;
362 					regex = args[i];
363 				} else if (args[i].equalsIgnoreCase("-elapse")) {
364 					i++;
365 					elapsed = Long.parseLong(args[i]);
366 				}
367 			}
368 		} catch (NumberFormatException e) {
369 			logger.error(Messages.getString("AbstractTransfer.20")+i); //$NON-NLS-1$
370 			return false;
371 		}
372 		if (fileInfo == null) {
373 			fileInfo = NO_INFO_ARGS;
374 		}
375 		if (tosubmit && ! DbConstant.admin.isConnected) {
376 			logger.error(Messages.getString("SpooledDirectoryTransfer.2")); //$NON-NLS-1$
377 			return false;
378 		}
379 		if (rhosts != null && rule != null && localDirectory != null && statusfile != null && stopfile != null) {
380 			return true;
381 		}
382 		logger.error(Messages.getString("SpooledDirectoryTransfer.56")+ //$NON-NLS-1$
383 				_INFO_ARGS);
384 		return false;
385 	}

388 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
389 		if (logger == null) {
390 			logger = WaarpInternalLoggerFactory.getLogger(SpooledDirectoryTransfer.class);
391 		}
392 		if (!getParams(args)) {
393 			logger.error(Messages.getString("Configuration.WrongInit")); //$NON-NLS-1$
394 			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
395 				DbConstant.admin.close();
396 			}
397 			ChannelUtils.stopLogger();
398 			System.exit(2);
399 		}
400 
401 		Configuration.configuration.pipelineInit();
402 		NetworkTransaction networkTransaction = new NetworkTransaction();
403 		try {
404 			R66Future future = new R66Future(true);
405 			SpooledDirectoryTransfer spooled =
406 					new SpooledDirectoryTransfer(future, localDirectory, statusfile, stopfile,
407 							rule, fileInfo, ismd5, rhosts, block, regex, elapsed, tosubmit, noLog,
408 							networkTransaction);
409 			spooled.run();
410 			future.awaitUninterruptibly();
411 			logger.warn(Messages.getString("SpooledDirectoryTransfer.58")+spooled.sent+" sent and "+spooled.error+Messages.getString("SpooledDirectoryTransfer.60")); //$NON-NLS-1$ //$NON-NLS-2$ //$NON-NLS-3$
412 		} catch (Exception e) {
413 			logger.warn("exc", e);
414 		} finally {
415 			networkTransaction.closeAll();
416 			System.exit(0);
417 		}
