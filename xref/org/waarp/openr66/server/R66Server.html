18  package org.waarp.openr66.server;
19  
20  import org.jboss.netty.logging.InternalLoggerFactory;
21  import org.waarp.common.logging.WaarpInternalLogger;
22  import org.waarp.common.logging.WaarpInternalLoggerFactory;
23  import org.waarp.common.logging.WaarpSlf4JLoggerFactory;
24  import org.waarp.openr66.configuration.FileBasedConfiguration;
25  import org.waarp.openr66.protocol.configuration.Configuration;
26  import org.waarp.openr66.protocol.configuration.Messages;
27  import org.waarp.openr66.protocol.exception.OpenR66ProtocolPacketException;
28  import org.waarp.openr66.protocol.utils.R66ShutdownHook;
29  
30  /**
31   * R66Server startup main class
32   * 
33   * @author Frederic Bregier
34   */
35  public class R66Server {
36  	private static WaarpInternalLogger logger;
37  
38  	/**
39  	 * @param args
40  	 *            as first argument the configuration file
41  	 * @throws OpenR66ProtocolPacketException
42  	 */
43  	public static void main(String[] args)
44  			throws OpenR66ProtocolPacketException {
45  		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
46  		logger = WaarpInternalLoggerFactory
47  				.getLogger(R66Server.class);
48  		if (args.length < 1) {
49  			logger
50  					.error(Messages.getString("Configuration.NeedConfig")); //$NON-NLS-1$
51  			return;
52  		}
53  		try {
54  			R66ShutdownHook.registerMain(R66Server.class, args);
55  			if (initialize(args[0])) {
56  				logger.warn(Messages.getString("R66Server.ServerStart") + Configuration.configuration.HOST_ID + " : " + Configuration.configuration.toString()); //$NON-NLS-1$
57  				System.err.println(Messages.getString("R66Server.ServerStart") + Configuration.configuration.HOST_ID); //$NON-NLS-1$
58  			} else {
59  				logger.error(Messages.getString("R66Server.CannotStart") + Configuration.configuration.HOST_ID); //$NON-NLS-1$
60  				System.err.println(Messages.getString("R66Server.CannotStart") + Configuration.configuration.HOST_ID); //$NON-NLS-1$
61  				System.exit(1);
62  			}
63  		} catch (Exception e) {
64  			logger.error(Messages.getString("R66Server.StartError"), e); //$NON-NLS-1$
65  		}
66  	}
67  
68  	public static boolean initialize(String config) {
69  		if (logger == null) {
70  			logger = WaarpInternalLoggerFactory
71  					.getLogger(R66Server.class);
72  		}
73  		if (!FileBasedConfiguration
74  				.setConfigurationServerFromXml(Configuration.configuration, config)) {
75  			logger
76  					.error(Messages.getString("Configuration.NeedCorrectConfig")); //$NON-NLS-1$
77  			return false;
78  		}
79  		try {
80  			Configuration.configuration.serverStartup();
81  		} catch (Throwable e) {
82  			logger
83  					.error(Messages.getString("R66Server.StartError"), e); //$NON-NLS-1$
84  			R66ShutdownHook.terminate(false);
85  			return false;
86  		}
87  		return true;
88  	}
89  }
