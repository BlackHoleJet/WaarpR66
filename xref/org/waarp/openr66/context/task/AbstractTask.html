256 	/**
257 	 * 
258 	 * @param arg
259 	 *            as the Format string where FIXED items will be replaced by context values and next
260 	 *            using argFormat as format second argument; this arg comes from the rule itself
261 	 * @param argFormat
262 	 *            as format second argument; this argFormat comes from the transfer Information
263 	 *            itself
264 	 * @return The string with replaced values from context and second argument
265 	 */
266 	protected String getReplacedValue(String arg, Object[] argFormat) {
267 		StringBuilder builder = new StringBuilder(arg);
268 		// check NOWAIT and LOCALEXEC
269 		if (arg.contains(NOWAIT)) {
270 			waitForValidation = false;
271 			WaarpStringUtils.replaceAll(builder, NOWAIT, "");
272 		}
273 		if (arg.contains(LOCALEXEC)) {
274 			useLocalExec = true;
275 			WaarpStringUtils.replaceAll(builder, LOCALEXEC, "");
276 		}
277 		File trueFile = null;
278 		if (session.getFile() != null) {
279 			trueFile = session.getFile().getTrueFile();
280 		}
281 		if (trueFile != null) {
282 			WaarpStringUtils.replaceAll(builder, TRUEFULLPATH, trueFile.getAbsolutePath());
283 			WaarpStringUtils.replaceAll(builder, TRUEFILENAME, R66Dir
284 					.getFinalUniqueFilename(session.getFile()));
285 			WaarpStringUtils.replaceAll(builder, FILESIZE, Long.toString(trueFile.length()));
286 		} else {
287 			WaarpStringUtils.replaceAll(builder, TRUEFULLPATH, "nofile");
288 			WaarpStringUtils.replaceAll(builder, TRUEFILENAME, "nofile");
289 			WaarpStringUtils.replaceAll(builder, FILESIZE, "0");
290 		}
291 		DbTaskRunner runner = session.getRunner();
292 		if (runner != null) {
293 			WaarpStringUtils.replaceAll(builder, ORIGINALFULLPATH, runner
294 				.getOriginalFilename());
295 			WaarpStringUtils.replaceAll(builder, ORIGINALFILENAME,
296 				R66File.getBasename(runner
297 						.getOriginalFilename()));
298 			WaarpStringUtils.replaceAll(builder, RULE, runner
299 				.getRuleId());
300 		}
301 		DateFormat dateFormat = new SimpleDateFormat("yyyyMMdd");
302 		Date date = new Date();
303 		WaarpStringUtils.replaceAll(builder, DATE, dateFormat.format(date));
304 		dateFormat = new SimpleDateFormat("HHmmss");
305 		WaarpStringUtils.replaceAll(builder, HOUR, dateFormat.format(date));
306 		if (session.getAuth() != null) {
307 			WaarpStringUtils.replaceAll(builder, REMOTEHOST, session.getAuth().getUser());
308 			try {
309 				WaarpStringUtils.replaceAll(builder, LOCALHOST,
310 						Configuration.configuration.getHostId(session.getAuth().isSsl()));
311 			} catch (OpenR66ProtocolNoSslException e) {
312 				// replace by standard name
313 				WaarpStringUtils.replaceAll(builder, LOCALHOST,
314 						Configuration.configuration.HOST_ID);
315 			}
316 		}
317 		if (session.getRemoteAddress() != null) {
318 			WaarpStringUtils.replaceAll(builder, REMOTEHOSTADDR, session.getRemoteAddress()
319 					.toString());
320 			WaarpStringUtils.replaceAll(builder, LOCALHOSTADDR, session.getLocalAddress()
321 					.toString());
322 		} else {
323 			WaarpStringUtils.replaceAll(builder, REMOTEHOSTADDR, "unknown");
324 			WaarpStringUtils.replaceAll(builder, LOCALHOSTADDR, "unknown");
325 		}
326 		if (runner != null) {
327 			WaarpStringUtils.replaceAll(builder, TRANSFERID, Long.toString(runner.getSpecialId()));
328 			String requester = runner.getRequester();
329 			WaarpStringUtils.replaceAll(builder, REQUESTERHOST, requester);
330 			String requested = runner.getRequested();
331 			WaarpStringUtils.replaceAll(builder, REQUESTEDHOST, requested);
332 			WaarpStringUtils.replaceAll(builder, FULLTRANSFERID, 
333 					runner.getSpecialId() + "_" + requester + "_" + requested);
334 			WaarpStringUtils.replaceAll(builder, RANKTRANSFER, Integer.toString(runner.getRank()));
335 		}
336 		WaarpStringUtils.replaceAll(builder, BLOCKSIZE, Integer.toString(session
337 				.getBlockSize()));
338 		R66Dir dir = new R66Dir(session);
339 		if (runner != null) {
340 			if (runner.isRecvThrough() || runner.isSendThrough()) {
341 				try {
342 					dir.changeDirectoryNotChecked(runner.getRule().getRecvPath());
343 					WaarpStringUtils.replaceAll(builder, INPATH, dir.getFullPath());
344 				} catch (CommandAbstractException e) {
345 				}
346 				dir = new R66Dir(session);
347 				try {
348 					dir.changeDirectoryNotChecked(runner.getRule().getSendPath());
349 					WaarpStringUtils.replaceAll(builder, OUTPATH, dir.getFullPath());
350 				} catch (CommandAbstractException e) {
351 				}
352 				dir = new R66Dir(session);
353 				try {
354 					dir.changeDirectoryNotChecked(runner.getRule().getWorkPath());
355 					WaarpStringUtils.replaceAll(builder, WORKPATH, dir.getFullPath());
356 				} catch (CommandAbstractException e) {
357 				}
358 				dir = new R66Dir(session);
359 				try {
360 					dir.changeDirectoryNotChecked(runner.getRule().getArchivePath());
361 					WaarpStringUtils.replaceAll(builder, ARCHPATH, dir.getFullPath());
362 				} catch (CommandAbstractException e) {
363 				}
364 			} else {
365 				try {
366 					dir.changeDirectory(runner.getRule().getRecvPath());
367 					WaarpStringUtils.replaceAll(builder, INPATH, dir.getFullPath());
368 				} catch (CommandAbstractException e) {
369 				}
370 				dir = new R66Dir(session);
371 				try {
372 					dir.changeDirectory(runner.getRule().getSendPath());
373 					WaarpStringUtils.replaceAll(builder, OUTPATH, dir.getFullPath());
374 				} catch (CommandAbstractException e) {
375 				}
376 				dir = new R66Dir(session);
377 				try {
378 					dir.changeDirectory(runner.getRule().getWorkPath());
379 					WaarpStringUtils.replaceAll(builder, WORKPATH, dir.getFullPath());
380 				} catch (CommandAbstractException e) {
381 				}
382 				dir = new R66Dir(session);
383 				try {
384 					dir.changeDirectory(runner.getRule().getArchivePath());
385 					WaarpStringUtils.replaceAll(builder, ARCHPATH, dir.getFullPath());
386 				} catch (CommandAbstractException e) {
387 				}
388 			}
389 		} else {
390 			try {
391 				dir.changeDirectory(Configuration.configuration.inPath);
392 				WaarpStringUtils.replaceAll(builder, INPATH, dir.getFullPath());
393 			} catch (CommandAbstractException e) {
394 			}
395 			dir = new R66Dir(session);
396 			try {
397 				dir.changeDirectory(Configuration.configuration.outPath);
398 				WaarpStringUtils.replaceAll(builder, OUTPATH, dir.getFullPath());
399 			} catch (CommandAbstractException e) {
400 			}
401 			dir = new R66Dir(session);
402 			try {
403 				dir.changeDirectory(Configuration.configuration.workingPath);
404 				WaarpStringUtils.replaceAll(builder, WORKPATH, dir.getFullPath());
405 			} catch (CommandAbstractException e) {
406 			}
407 			dir = new R66Dir(session);
408 			try {
409 				dir.changeDirectory(Configuration.configuration.archivePath);
410 				WaarpStringUtils.replaceAll(builder, ARCHPATH, dir.getFullPath());
411 			} catch (CommandAbstractException e) {
412 			}
413 		}
414 		WaarpStringUtils.replaceAll(builder, HOMEPATH, Configuration.configuration.baseDirectory);
415 		if (session.getLocalChannelReference() == null) {
416 			WaarpStringUtils.replaceAll(builder, ERRORMSG, "NoError");
417 			WaarpStringUtils.replaceAll(builder, ERRORCODE, "-");
418 			WaarpStringUtils.replaceAll(builder, ERRORSTRCODE, ErrorCode.Unknown.name());
419 		} else {
420 			try {
421 				WaarpStringUtils.replaceAll(builder, ERRORMSG, session.getLocalChannelReference()
422 						.getErrorMessage());
423 			} catch (NullPointerException e) {
424 				WaarpStringUtils.replaceAll(builder, ERRORMSG, "NoError");
425 			}
426 			try {
427 				WaarpStringUtils.replaceAll(builder, ERRORCODE, session.getLocalChannelReference()
428 						.getCurrentCode().getCode());
429 			} catch (NullPointerException e) {
430 				WaarpStringUtils.replaceAll(builder, ERRORCODE, "-");
431 			}
432 			try {
433 				WaarpStringUtils.replaceAll(builder, ERRORSTRCODE, session
434 						.getLocalChannelReference().getCurrentCode().name());
435 			} catch (NullPointerException e) {
436 				WaarpStringUtils.replaceAll(builder, ERRORSTRCODE, ErrorCode.Unknown.name());
437 			}
438 		}
439 		// finalname
440 		if (argFormat != null && argFormat.length > 0)
441 			return String.format(builder.toString(), argFormat);
442 		return builder.toString();
443 	}
