37  /**
38   * 
39   * Utility class to write external file with cache enable properties
40   * 
41   * @author Frederic Bregier
42   * 
43   */
44  public class HttpWriteCacheEnable {
45  
46  	/**
47  	 * Write a file, taking into account cache enabled and removing session cookie
48  	 * 
49  	 * @param request
50  	 * @param channel
51  	 * @param filename
52  	 * @param cookieNameToRemove
53  	 */
54  	public static void writeFile(HttpRequest request, Channel channel, String filename,
55  			String cookieNameToRemove) {
56  		// Convert the response content to a ChannelBuffer.
57          HttpResponse response;
58          if (request.containsHeader(HttpHeaders.Names.IF_MODIFIED_SINCE)) {
59              response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
60                      HttpResponseStatus.NOT_MODIFIED);
61              handleCookies(request, response, cookieNameToRemove);
62              channel.write(response);
63              return;
64          }
65          File file = new File(filename);
66          if (!file.isFile() || !file.canRead()) {
67              response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
68                      HttpResponseStatus.NOT_FOUND);
69              handleCookies(request, response, cookieNameToRemove);
70              channel.write(response);
71              return;
72          }
73          long size = file.length();
74          ChunkedNioFile nioFile;
75  		try {
76  			nioFile = new ChunkedNioFile(file);
77  		} catch (IOException e) {
78              response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
79                      HttpResponseStatus.NOT_FOUND);
80              handleCookies(request, response, cookieNameToRemove);
81              channel.write(response);
82              return;
83  		}
84          response = new DefaultHttpResponse(HttpVersion.HTTP_1_1,
85                  HttpResponseStatus.OK);
86          response.setHeader(HttpHeaders.Names.CONTENT_LENGTH,
87                  String.valueOf(size));
88          ArrayList<String> cache_control = new ArrayList<String>(2);
89          cache_control.add(HttpHeaders.Values.PUBLIC);
90          cache_control.add(HttpHeaders.Values.MAX_AGE + "=" + 31536000);// 1 year
91          response.setHeader(HttpHeaders.Names.CACHE_CONTROL, cache_control);
92          response.setHeader(HttpHeaders.Names.LAST_MODIFIED,
93                  "Tue, 15 Nov 1994 12:45:26 GMT");
94          handleCookies(request, response, cookieNameToRemove);
95          // Write the response.
96          channel.write(response);
97          channel.write(nioFile);
98  	}

100 	/**
101 	 * Remove the given named cookie
102 	 * 
103 	 * @param request
104 	 * @param response
105 	 * @param cookieNameToRemove
106 	 */
107 	public static void handleCookies(HttpRequest request, HttpResponse response,
108 			String cookieNameToRemove) {
109 		String cookieString = request.getHeader(HttpHeaders.Names.COOKIE);
110 		if (cookieString != null) {
111 			CookieDecoder cookieDecoder = new CookieDecoder();
112 			Set<Cookie> cookies = cookieDecoder.decode(cookieString);
113 			if (!cookies.isEmpty()) {
114 				// Reset the sessions if necessary.
115 				CookieEncoder cookieEncoder = new CookieEncoder(true);
116 				// Remove all Session for images
117 				for (Cookie cookie : cookies) {
118 					if (cookie.getName().equalsIgnoreCase(cookieNameToRemove)) {
119 					} else {
120 						cookieEncoder.addCookie(cookie);
121 						response.addHeader(HttpHeaders.Names.SET_COOKIE, cookieEncoder.encode());
122 						cookieEncoder = new CookieEncoder(true);
123 					}
124 				}
125 			}
126 		}
127 	}
