49   * Log Export from a local client without database connection

68  	public LogExport(R66Future future, boolean purgeLog, boolean clean,
69  			Timestamp start, Timestamp stop,
70  			NetworkTransaction networkTransaction) {
71  		this.future = future;
72  		this.purgeLog = purgeLog;
73  		this.clean = clean;
74  		this.start = start;
75  		this.stop = stop;
76  		this.networkTransaction = networkTransaction;
77  		this.host = Configuration.configuration.HOST_SSLAUTH;
78  	}

81  		this.host = host;

84  	/**
85  	 * Prior to call this method, the pipeline and NetworkTransaction must have been initialized. It
86  	 * is the responsibility of the caller to finish all network resources.
87  	 */
88  	public void run() {
89  		if (logger == null) {
90  			logger = WaarpInternalLoggerFactory.getLogger(LogExport.class);
91  		}
92  		String lstart = (start != null) ? start.toString() : null;
93  		String lstop = (stop != null) ? stop.toString() : null;
94  		byte type = (purgeLog) ? LocalPacketFactory.LOGPURGEPACKET : LocalPacketFactory.LOGPACKET;
95  		ValidPacket valid = new ValidPacket(lstart, lstop, type);
96  		SocketAddress socketAddress = host.getSocketAddress();
97  		boolean isSSL = host.isSsl();
98  
99  		// first clean if ask
100 		if (clean && (host.getHostid().equals(Configuration.configuration.HOST_ID)
101 				|| host.getHostid().equals(Configuration.configuration.HOST_SSLAUTH))) {
102 			// Update all UpdatedInfo to DONE
103 			// where GlobalLastStep = ALLDONETASK and status = CompleteOk
104 			try {
105 				DbTaskRunner.changeFinishedToDone(DbConstant.admin.session);
106 			} catch (WaarpDatabaseNoConnectionException e) {
107 				logger.warn("Clean cannot be done {}", e.getMessage());
108 			}
109 		}
110 		LocalChannelReference localChannelReference = networkTransaction
111 				.createConnectionWithRetry(socketAddress, isSSL, future);
112 		socketAddress = null;
113 		if (localChannelReference == null) {
114 			host = null;
115 			logger.error("Cannot Connect");
116 			future.setResult(new R66Result(
117 					new OpenR66ProtocolNoConnectionException("Cannot connect to server"),
118 					null, true, ErrorCode.Internal, null));
119 			future.setFailure(future.getResult().exception);
120 			return;
121 		}
122 		localChannelReference.sessionNewState(R66FiniteDualStates.VALIDOTHER);
123 		try {
124 			ChannelUtils.writeAbstractLocalPacket(localChannelReference, valid, false);
125 		} catch (OpenR66ProtocolPacketException e) {
126 			logger.error("Bad Protocol", e);
127 			Channels.close(localChannelReference.getLocalChannel());
128 			localChannelReference = null;
129 			host = null;
130 			valid = null;
131 			future.setResult(new R66Result(e, null, true,
132 					ErrorCode.TransferError, null));
133 			future.setFailure(e);
134 			return;
135 		}
136 		host = null;
137 		future.awaitUninterruptibly();
138 		logger.info("Request done with " + (future.isSuccess() ? "success" : "error"));
139 		Channels.close(localChannelReference.getLocalChannel());
140 		localChannelReference = null;
141 	}

149 	protected static boolean getParams(String[] args) {
150 		if (args.length < 1) {
151 			logger.error("Need at least the configuration file as first argument then optionally\n"
152 					+
153 					"    -purge\n"
154 					+
155 					"    -clean\n"
156 					+
157 					"    -start timestamp in format yyyyMMddHHmmssSSS possibly truncated and where one of ':-. ' can be separators\n"
158 					+
159 					"    -stop timestamp in same format than start\n" +
160 					"If not start and no stop are given, stop is Today Midnight (00:00:00)\n" +
161 					"If start is equals or greater than stop, stop is start+24H\n"+
162 					"    -host host (optional)");
163 			return false;
164 		}
165 		if (!FileBasedConfiguration
166 				.setClientConfigurationFromXml(Configuration.configuration, args[0])) {
167 			logger.error("Need at least the configuration file as first argument then optionally\n"
168 					+
169 					"    -purge\n"
170 					+
171 					"    -clean\n"
172 					+
173 					"    -start timestamp in format yyyyMMddHHmmssSSS possibly truncated and where one of ':-. ' can be separators\n"
174 					+
175 					"    -stop timestamp in same format than start\n" +
176 					"If not start and no stop are given, stop is Today Midnight (00:00:00)\n" +
177 					"If start is equals or greater than stop, stop is start+24H\n"+
178 					"    -host host (optional)");
179 			return false;
180 		}
181 		String ssstart = null;
182 		String ssstop = null;
183 		for (int i = 1; i < args.length; i++) {
184 			if (args[i].equalsIgnoreCase("-purge")) {
185 				spurgeLog = true;
186 			} else if (args[i].equalsIgnoreCase("-clean")) {
187 				sclean = true;
188 			} else if (args[i].equalsIgnoreCase("-start")) {
189 				i++;
190 				ssstart = args[i];
191 			} else if (args[i].equalsIgnoreCase("-stop")) {
192 				i++;
193 				ssstop = args[i];
194 			} else if (args[i].equalsIgnoreCase("-host")) {
195 				i++;
196 				stohost = args[i];
197 			}
198 		}
199 		if (ssstart != null) {
200 			Timestamp tstart = WaarpStringUtils.fixDate(ssstart);
201 			if (tstart != null) {
202 				sstart = tstart;
203 			}
204 		}
205 		if (ssstop != null) {
206 			Timestamp tstop = WaarpStringUtils.fixDate(ssstop, sstart);
207 			if (tstop != null) {
208 				sstop = tstop;
209 			}
210 		}
211 		if (ssstart == null && ssstop == null) {
212 			sstop = WaarpStringUtils.getTodayMidnight();
213 		}
214 		return true;
215 	}

217 	public static void main(String[] args) {
218 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
219 		if (logger == null) {
220 			logger = WaarpInternalLoggerFactory.getLogger(LogExport.class);
221 		}
222 		if (!getParams(args)) {
223 			logger.error("Wrong initialization");
224 			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
225 				DbConstant.admin.close();
226 			}
227 			System.exit(1);
228 		}
229 		long time1 = System.currentTimeMillis();
230 		R66Future future = new R66Future(true);
231 
232 		Configuration.configuration.pipelineInit();
233 		NetworkTransaction networkTransaction = new NetworkTransaction();
234 		try {
235 			LogExport transaction = new LogExport(future,
236 					spurgeLog, sclean, sstart, sstop,
237 					networkTransaction);
238 			if (stohost != null) {
239 				try {
240 					transaction.setHost(new DbHostAuth(DbConstant.admin.session, stohost));
241 				} catch (WaarpDatabaseException e) {
242 					logger.error("LogExport in     FAILURE since Host is not found: "+stohost, e);
243 					networkTransaction.closeAll();
244 					System.exit(10);
245 				}
246 			} else {
247 				stohost = Configuration.configuration.HOST_SSLID;
248 			}
249 			transaction.run();
250 			future.awaitUninterruptibly();
251 			long time2 = System.currentTimeMillis();
252 			long delay = time2 - time1;
253 			R66Result result = future.getResult();
254 			if (future.isSuccess()) {
255 				if (result.code == ErrorCode.Warning) {
256 					logger.warn("WARNED on file:     " +
257 							(result.other != null ? ((ValidPacket) result.other).getSheader() :
258 									"no file")
259 							+ "     delay: " + delay);
260 				} else {
261 					logger.warn("SUCCESS on Final file:     " +
262 							(result.other != null ? ((ValidPacket) result.other).getSheader() :
263 									"no file")
264 							+ "     delay: " + delay);
265 				}
266 			} else {
267 				if (result.code == ErrorCode.Warning) {
268 					logger.warn("LogExport is     WARNED", future.getCause());
269 					networkTransaction.closeAll();
270 					System.exit(result.code.ordinal());
271 				} else {
272 					logger.error("LogExport in     FAILURE", future.getCause());
273 					networkTransaction.closeAll();
274 					System.exit(result.code.ordinal());
275 				}
276 			}
277 		} finally {
278 			networkTransaction.closeAll();
279 		}
280 	}
