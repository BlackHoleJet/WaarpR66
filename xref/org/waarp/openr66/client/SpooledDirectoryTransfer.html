42  /**
43   * Direct Transfer from a client with or without database connection 
44   * or Submit Transfer from a client with database connection 
45   * to transfer files from a spooled directory to possibly multiple hosts at once.<br>
46   * Hosts will have to be separated by ','.<br>
47   * <br>
48   * Mandatory additional elements:<br>
49   * -directory source (directory to spooled on)<br>
50   * -statusfile file (file to use as permanent status (if process is killed or aborts))<br>
51   * -stopfile file (file when created will stop the dameon)<br>
52   * Other options:<br>
53   * -regex regex (regular expression to filter file names from directory source)<br>
54   * -elapse elapse (elapse time between 2 checks of the directory)<br>
55   * -submit (to submit only: default)<br>
56   * -direct (to directly transfer only)<br>
57   * 
58   * @author Frederic Bregier
59   * 
60   */
61  public class SpooledDirectoryTransfer implements Runnable {

170 	@Override
171 	public void run() {
172 		sent = 0;
173 		error  = 0;
174 		final String [] allrhosts = remoteHosts.split(",");
175 		// first check if rule is for SEND
176 		DbRule dbrule = null;
177 		try {
178 			dbrule = new DbRule(DbConstant.admin.session, rulename);
179 		} catch (WaarpDatabaseException e1) {
180 			logger.error("Rule error: ", e1);
181 			this.future.setFailure(e1);
182 			return;
183 		}
184 		if (dbrule.isRecvMode()) {
185 			logger.error("Rule error: not SEND MODE");
186 			this.future.cancel();
187 			return;
188 		}
189 		File status = new File(statusFile);
190 		if (status.isDirectory()) {
191 			logger.error("Status File is a directory but must be a file");
192 			this.future.cancel();
193 			return;
194 		}
195 		File stop = new File(stopFile);
196 		if (stop.isDirectory()) {
197 			logger.error("Stop File is a directory but must be a file");
198 			this.future.cancel();
199 			return;
200 		} else if (stop.exists()) {
201 			logger.warn("Stop File is found");
202 			this.future.setSuccess();
203 			return;
204 		}
205 		File dir = new File(directory);
206 		if (!dir.isDirectory()) {
207 			logger.error("Directory is not a directory but must be a directory");
208 			this.future.cancel();
209 			return;
210 		}
211 		FileFilter filter = null;
212 		if (regexFilter != null) {
213 			filter = new RegexFileFilter(regexFilter);
214 		}
215 		FileMonitorCommand commandValidFile = new FileMonitorCommand() {
216 			public void run(File file) {
217 				for (String host : allrhosts) {
218 					host = host.trim();
219 					if (host != null && ! host.isEmpty()) {
220 						String filename = file.getAbsolutePath();
221 						logger.info("Launch transfer to "+host+" with file "+filename);
222 						R66Future future = new R66Future(true);
223 						String text = null;
224 						if (submit) {
225 							text = "Submit Transfer";
226 							SubmitTransfer transaction = new SubmitTransfer(future,
227 									host, filename, rulename, fileinfo, isMD5, blocksize, 
228 									DbConstant.ILLEGALVALUE, null);
229 							transaction.run();
230 						} else {
231 							text = "Direct Transfer";
232 							DirectTransfer transaction = new DirectTransfer(future,
233 									host, filename, rule, fileInfo, ismd5, block, 
234 									DbConstant.ILLEGALVALUE, networkTransaction);
235 							logger.debug("rhost: "+host+":"+transaction.remoteHost);
236 							transaction.run();
237 						}
238 						future.awaitUninterruptibly();
239 						R66Result result = future.getResult();
240 						if (future.isSuccess()) {
241 							sent++;
242 							DbTaskRunner runner = null;
243 							if (result != null) {
244 								runner = result.runner;
245 								if (runner != null) {
246 									String status = "SUCCESS";
247 									if (runner.getErrorInfo() == ErrorCode.Warning) {
248 										status = "WARNED";
249 									}
250 									logger.warn(text+" in status: "+status+"     "
251 											+ runner.toShortString()
252 											+"     <REMOTE>"+ host+ "</REMOTE>"
253 											+"     <FILEFINAL>"+
254 											(result.file != null ? 
255 													result.file.toString() + "</FILEFINAL>"
256 													: "no file"));
257 									if (nolog && !submit) {
258 										// In case of success, delete the runner
259 										try {
260 											runner.delete();
261 										} catch (WaarpDatabaseException e) {
262 											logger.warn("Cannot apply nolog to     " +
263 													runner.toShortString(),
264 													e);
265 										}
266 									}
267 								} else {
268 									logger.warn(text+" in     SUCCESS     " 
269 											+"<REMOTE>" + host + "</REMOTE>");
270 								}
271 							} else {
272 								logger.warn(text+" in     SUCCESS     " 
273 										+"<REMOTE>" + host + "</REMOTE>");
274 							}
275 						} else {
276 							error++;
277 							DbTaskRunner runner = null;
278 							if (result != null) {
279 								runner = result.runner;
280 								if (runner != null) {
281 									logger.error(text+" in     FAILURE      " + 
282 											runner.toShortString() +
283 										"<REMOTE>" + host + "</REMOTE>", future.getCause());
284 								} else {
285 									logger.error(text+" in     FAILURE      ", 
286 											future.getCause());
287 								}
288 							} else {
289 								logger.error(text+" in     FAILURE      "
290 										+"<REMOTE>" + host + "</REMOTE>", 
291 										future.getCause());
292 							}
293 						}
294 					}
295 				}
296 			}
297 		};
298 		FileMonitor monitor = new FileMonitor(status, stop, dir, null, elapsed, filter, 
299 				commandValidFile, null);
300 		monitor.start();
301 		monitor.waitForStopFile();
302 		this.future.setSuccess();
303 	}

321 	/**
322 	 * Parse the parameter and set current values
323 	 * 
324 	 * @param args
325 	 * @return True if all parameters were found and correct
326 	 */
327 	protected static boolean getParams(String[] args) {
328 		if (args.length < 11) {
329 			logger
330 					.error(_INFO_ARGS);
331 			return false;
332 		}
333 		if (!FileBasedConfiguration
334 				.setClientConfigurationFromXml(Configuration.configuration, args[0])) {
335 			logger
336 					.error("Needs a correct configuration file as first argument");
337 			return false;
338 		}
339 		// Now set default values from configuration
340 		block = Configuration.configuration.BLOCKSIZE;
341 		int i = 1;
342 		try {
343 			for (i = 1; i < args.length; i++) {
344 				if (args[i].equalsIgnoreCase("-to")) {
345 					i++;
346 					rhosts = args[i];
347 				} else if (args[i].equalsIgnoreCase("-directory")) {
348 					i++;
349 					localDirectory = args[i];
350 				} else if (args[i].equalsIgnoreCase("-rule")) {
351 					i++;
352 					rule = args[i];
353 				} else if (args[i].equalsIgnoreCase("-statusfile")) {
354 					i++;
355 					statusfile = args[i];
356 				} else if (args[i].equalsIgnoreCase("-stopfile")) {
357 					i++;
358 					stopfile = args[i];
359 				} else if (args[i].equalsIgnoreCase("-info")) {
360 					i++;
361 					fileInfo = args[i];
362 				} else if (args[i].equalsIgnoreCase("-md5")) {
363 					ismd5 = true;
364 				} else if (args[i].equalsIgnoreCase("-block")) {
365 					i++;
366 					block = Integer.parseInt(args[i]);
367 					if (block < 100) {
368 						logger.error("Block size is too small: " + block);
369 						return false;
370 					}
371 				} else if (args[i].equalsIgnoreCase("-nolog")) {
372 					noLog = true;
373 				} else if (args[i].equalsIgnoreCase("-submit")) {
374 					tosubmit = true;
375 				} else if (args[i].equalsIgnoreCase("-direct")) {
376 					tosubmit = false;
377 				} else if (args[i].equalsIgnoreCase("-regex")) {
378 					i++;
379 					regex = args[i];
380 				} else if (args[i].equalsIgnoreCase("-elapse")) {
381 					i++;
382 					elapsed = Long.parseLong(args[i]);
383 				}
384 			}
385 		} catch (NumberFormatException e) {
386 			logger.error("Number Format exception at Rank "+i);
387 			return false;
388 		}
389 		if (fileInfo == null) {
390 			fileInfo = NO_INFO_ARGS;
391 		}
392 		if (rhosts != null && rule != null && localDirectory != null && statusfile != null && stopfile != null) {
393 			return true;
394 		}
395 		logger.error("All params are not set! Need at least (-to -rule -statusfile and -directory)\n"+
396 				_INFO_ARGS);
397 		return false;
398 	}

400 	public static void main(String[] args) {
401 		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
402 		if (logger == null) {
403 			logger = WaarpInternalLoggerFactory.getLogger(SpooledDirectoryTransfer.class);
404 		}
405 		if (!getParams(args)) {
406 			logger.error("Wrong initialization");
407 			if (DbConstant.admin != null && DbConstant.admin.isConnected) {
408 				DbConstant.admin.close();
409 			}
410 			ChannelUtils.stopLogger();
411 			System.exit(2);
412 		}
413 
414 		Configuration.configuration.pipelineInit();
415 		NetworkTransaction networkTransaction = new NetworkTransaction();
416 		try {
417 			R66Future future = new R66Future(true);
418 			SpooledDirectoryTransfer spooled =
419 					new SpooledDirectoryTransfer(future, localDirectory, statusfile, stopfile,
420 							rule, fileInfo, ismd5, rhosts, block, regex, elapsed, tosubmit, noLog,
421 							networkTransaction);
422 			spooled.run();
423 			future.awaitUninterruptibly();
424 			logger.warn("End of SpooledDirectory with "+spooled.sent+" sent and "+spooled.error+" in error");
425 		} catch (Exception e) {
426 			logger.warn("exc", e);
427 		} finally {
428 			networkTransaction.closeAll();
429 			System.exit(0);
430 		}
431 	}
