29  /**
30   * FTP client compatible for Waarp Gateway Kernel as JavaExecutor<br>
31   * <br>
32   * Ftp Transfer task: synchronous<br>
33   * 
34   * Result of arguments will be as FTP command.<br>
35   * Format is the following:<br>
36   * "-file filepath <br>
37   * -to requestedHost <br>
38   * -port port <br>
39   * -user user <br>
40   * -pwd pwd <br>
41   * [-account account] <br>
42   * [-mode active/passive] <br>
43   * [-ssl no/implicit/explicit]<br>
44   * [-cwd remotepath] <br>
45   * [-digest (crc,md5,sha1)] <br>
46   * [-pre extraCommand1 with ',' as separator of arguments] <br>
47   * -command command from (get,put,append) <br>
48   * [-post extraCommand2 with ',' as separator of arguments]" <br>
49   * <br>
50   * <br>
51   * The order of commands will be:<br>
52   * 1) connection to requestHost on port (if ssl native => using native ssl link)<br>
53   * 2) User user<br>
54   * 3) PASS pwd<br>
55   * 4) if account => ACCT account<br>
56   * 5) if -ssl & auth => AUTH, PBSZ 0, PROT P <br>
57   * 6) if passive => PASV<br>
58   * 7) CWD remotepath; if error => MKD remotepath then CWD remotepath (and ignoring any error)<br>
59   * 8) if pre => extraCommand1 with ',' replaced by ' ' (note: do not use standard commands from FTP
60   * like
61   * ACCT,PASS,REIN,USER,APPE,STOR,STOU,RETR,RMD,RNFR,RNTO,ABOR,CWD,CDUP,MODE,PASV,PORT,STRU,TYPE,
62   * MDTM,MLSD,MLST,SIZE,AUTH) <br>
63   * 9) BINARY (binary format)<br>
64   * 10) if get => RETR filepath.basename ; if put => STOR filepath ; if append => APPE
65   * filepath.basename<br>
66   * 11) if digest & get/put/append & remote site compatible with XCRC,XMD5,XSHA1 => FEAT (parsing if
67   * found corresponding XCRC,XMD5,XSHA1) ; then XCRC/XMD5/XSHA1 filepath.basename ; then locally
68   * comparing this XCRC/XMD5/XSHA1 with the local file<br>
69   * 12) if post => extraCommand2 with ',' replaced by ' ' (note: do not use standard commands from
70   * FTP like
71   * ACCT,PASS,REIN,USER,APPE,STOR,STOU,RETR,RMD,RNFR,RNTO,ABOR,CWD,CDUP,MODE,PASV,PORT,STRU,TYPE
72   * ,MDTM,MLSD,MLST,SIZE,AUTH)<br>
73   * 13) QUIT<br>
74   * 
75   * 
76   * @author "Frederic Bregier"
77   * 
78   */
79  public class JavaExecutorWaarpFtp4jClient implements GatewayRunnable {
80  	/**
81  	 * Internal Logger
82  	 */
83  	private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
84  			.getLogger(JavaExecutorWaarpFtp4jClient.class);
85  
86  	boolean waitForValidation;
87  	boolean useLocalExec;
88  	int delay;
89  	String[] args;
90  	int status = 0;
91  
92  	public JavaExecutorWaarpFtp4jClient() {
93  	}
94  
95  	@Override
96  	public void run() {
97  		logger.info("FtpTransfer with " + args.length + " arguments");
98  		if (args.length < 10) {
99  			status = -1;
100 			logger.error("Not enough arguments");
101 			return;
102 		}
103 		String filepath = null;
104 		String filename = null;
105 		String requested = null;
106 		int port = 21;
107 		String user = null;
108 		String pwd = null;
109 		String acct = null;
110 		boolean isPassive = true;
111 		int ssl = 0; // -1 native, 1 auth
112 		String cwd = null;
113 		int digest = 0; // 1 CRC, 2 MD5, 3 SHA1
114 		String command = null;
115 		int codeCommand = 0; // -1 get, 1 put, 2 append
116 		String preArgs = null;
117 		String postArgs = null;
118 		/*
119 		 * "-file filepath <br> -to requestedHost <br> -port port <br> -user user <br> -pwd pwd <br>
120 		 * [-account account] <br> [-mode active/passive] <br> [-ssl no/implicit/explicit]<br> [-cwd
121 		 * remotepath] <br> [-digest (crc,md5,sha1)] <br> [-pre extraCommand1 with ',' as separator
122 		 * of arguments] <br> -command command from (get,put,append) <br> [-post extraCommand2 with
123 		 * ',' as separator of arguments]" <br>
124 		 */
125 		for (int i = 0; i < args.length; i++) {
126 			if (args[i].equalsIgnoreCase("-file")) {
127 				i++;
128 				filepath = args[i];
129 				filename = new File(filepath).getName();
130 			} else if (args[i].equalsIgnoreCase("-to")) {
131 				i++;
132 				requested = args[i];
133 			} else if (args[i].equalsIgnoreCase("-port")) {
134 				i++;
135 				port = Integer.parseInt(args[i]);
136 			} else if (args[i].equalsIgnoreCase("-user")) {
137 				i++;
138 				user = args[i];
139 			} else if (args[i].equalsIgnoreCase("-pwd")) {
140 				i++;
141 				pwd = args[i];
142 			} else if (args[i].equalsIgnoreCase("-account")) {
143 				i++;
144 				acct = args[i];
145 			} else if (args[i].equalsIgnoreCase("-mode")) {
146 				i++;
147 				isPassive = (args[i].equalsIgnoreCase("passive"));
148 			} else if (args[i].equalsIgnoreCase("-ssl")) {
149 				i++;
150 				if (args[i].equalsIgnoreCase("implicit")) {
151 					ssl = -1;
152 				} else if (args[i].equalsIgnoreCase("explicit")) {
153 					ssl = 1;
154 				} else {
155 					ssl = 0;
156 				}
157 			} else if (args[i].equalsIgnoreCase("-cwd")) {
158 				i++;
159 				cwd = args[i];
160 			} else if (args[i].equalsIgnoreCase("-digest")) {
161 				i++;
162 				if (args[i].equalsIgnoreCase("crc")) {
163 					digest = 1;
164 				} else if (args[i].equalsIgnoreCase("md5")) {
165 					digest = 2;
166 				} else if (args[i].equalsIgnoreCase("sha1")) {
167 					digest = 3;
168 				} else {
169 					digest = 0;
170 				}
171 			} else if (args[i].equalsIgnoreCase("-pre")) {
172 				i++;
173 				preArgs = args[i].replace(',', ' ');
174 			} else if (args[i].equalsIgnoreCase("-post")) {
175 				i++;
176 				postArgs = args[i].replace(',', ' ');
177 			} else if (args[i].equalsIgnoreCase("-command")) {
178 				i++;
179 				command = args[i];
180 				// get,put,append,list
181 				// -1 get, 1 put, 2 append
182 				if (command.equalsIgnoreCase("get")) {
183 					codeCommand = -1;
184 				} else if (command.equalsIgnoreCase("put")) {
185 					codeCommand = 1;
186 				} else if (command.equalsIgnoreCase("append")) {
187 					codeCommand = 2;
188 				} else {
189 					// error
190 					codeCommand = 0;
191 				}
192 			}
193 		}
194 		if (filepath == null || requested == null || port <= 0 || user == null || pwd == null ||
195 				codeCommand == 0) {
196 			status = -2;
197 			int code = 0 +
198 					(filepath == null ? 1 : 0) + (requested == null ? 10 : 0) +
199 					(port <= 0 ? 100 : 0) + (user == null ? 1000 : 0) + (pwd == null ? 10000 : 0) +
200 					(codeCommand == 0 ? 100000 : 0);
201 			logger.error("Not enough arguments: " + code);
202 			return;
203 		}
204 		int timeout = 30000;
205 		if (delay > 1000) {
206 			timeout = delay;
207 		}
208 		WaarpFtp4jClient ftpClient =
209 				new WaarpFtp4jClient(requested, port, user, pwd, acct, isPassive, ssl,
210 						5000, timeout);
211 		boolean connected = false;
212 		for (int i = 0; i < 3; i++) {
213 			if (ftpClient.connect()) {
214 				connected = true;
215 				break;
216 			}
217 		}
218 		if (!connected) {
219 			status = -3;
220 			logger.error(ftpClient.getResult());
221 			return;
222 		}
223 		try {
224 			if (cwd != null && !ftpClient.changeDir(cwd)) {
225 				ftpClient.makeDir(cwd);
226 				if (!ftpClient.changeDir(cwd)) {
227 					logger.warn("Cannot change od directory: " + cwd);
228 				}
229 			}
230 			if (preArgs != null) {
231 				String[] result = ftpClient.executeCommand(preArgs);
232 				for (String string : result) {
233 					logger.debug("PRE: " + string);
234 				}
235 			}
236 			if (!ftpClient.transferFile(filepath, filename, codeCommand)) {
237 				status = -4;
238 				logger.error(ftpClient.getResult());
239 				return;
240 			}
241 			if (digest > 0) {
242 				// digest check
243 				String params = null;
244 				DigestAlgo algo = null;
245 				switch (digest) {
246 					case 1: // CRC
247 						params = "XCRC ";
248 						algo = DigestAlgo.CRC32;
249 						break;
250 					case 2: // MD5
251 						params = "XMD5 ";
252 						algo = DigestAlgo.MD5;
253 						break;
254 					case 3: // SHA1
255 					default:
256 						params = "XSHA1 ";
257 						algo = DigestAlgo.SHA1;
258 						break;
259 				}
260 				params += filename;
261 				String[] values = ftpClient.executeCommand(params);
262 				String hashresult = null;
263 				if (values != null) {
264 					values = values[0].split(" ");
265 					hashresult = (values.length > 3 ? values[1] : values[0]);
266 				}
267 				if (hashresult == null) {
268 					status = -5;
269 					logger.error("Hash cannot be computed: " + ftpClient.getResult());
270 					return;
271 				}
272 				// now check locally
273 				String hash;
274 				try {
275 					hash = FilesystemBasedDigest.getHex(
276 							FilesystemBasedDigest.getHash(new File(filepath), false, algo));
277 				} catch (IOException e) {
278 					hash = null;
279 				}
280 				if (hash == null || (!hash.equalsIgnoreCase(hashresult))) {
281 					status = -6;
282 					logger.error("Hash not equal: " + ftpClient.getResult());
283 					return;
284 				}
285 			}
286 			if (postArgs != null) {
287 				String[] result = ftpClient.executeCommand(postArgs);
288 				for (String string : result) {
289 					logger.debug("POST: " + string);
290 				}
291 			}
292 		} finally {
293 			ftpClient.logout();
294 		}
295 		logger.info("FTP transfer in\n    SUCCESS\n    " + filepath +
296 				"\n    <REMOTE>" + requested + "</REMOTE>");
297 		status = 0;
298 	}
