47  /**
48   * Bandwidth Http REST interface: http://host/bandwidth?... + BandwidthJsonPacket as GET or PUT
49   * @author "Frederic Bregier"
50   *
51   */
52  public class HttpRestBandwidthR66Handler extends HttpRestAbstractR66Handler {
53  	
54  	public static final String BASEURI = "bandwidth";
55  	/**
56       * Internal Logger
57       */
58      private static final WaarpInternalLogger logger = WaarpInternalLoggerFactory
59              .getLogger(HttpRestBandwidthR66Handler.class);
60     
61  	public HttpRestBandwidthR66Handler(RestConfiguration config, METHOD ... methods) {
62  		super(BASEURI, config, METHOD.OPTIONS);
63  		setIntersectionMethods(methods, METHOD.GET, METHOD.PUT);
64  	}
65  	
66  	@Override
67  	public void endParsingRequest(HttpRestHandler handler, RestArgument arguments, RestArgument result, Object body)
68  			throws HttpIncorrectRequestException, HttpInvalidAuthenticationException {
69  		logger.debug("debug: {} ### {}",arguments,result);
70  		if (body != null) {
71  			logger.debug("Obj: {}", body);
72  		}
73  		handler.setWillClose(false);
74  		ServerActions serverHandler = ((HttpRestR66Handler) handler).serverHandler;
75  		// now action according to body
76  		JsonPacket json = (JsonPacket) body;
77  		if (json == null) {
78  			result.setDetail("not enough information");
79  			setError(handler, result, HttpResponseStatus.BAD_REQUEST);
80  			return;
81  		}
82  		result.getAnswer().put(AbstractDbData.JSON_MODEL, RESTHANDLERS.Bandwidth.name());
83  		try {
84  			if (json instanceof BandwidthJsonPacket) {//
85  				// setter, writeglobal, readglobal, writesession, readsession
86  				BandwidthJsonPacket node = (BandwidthJsonPacket) json;
87  				boolean setter = node.isSetter();
88  				if (setter && arguments.getMethod() != METHOD.PUT) {
89  					// wrong
90  					result.setDetail("Setter should be requested with a PUT method");
91  					setError(handler, result, HttpResponseStatus.CONFLICT);
92  					return;
93  				} else if (! setter && arguments.getMethod() != METHOD.GET) {
94  					// wrong
95  					result.setDetail("Getter should not be requested with a GET method");
96  					setError(handler, result, HttpResponseStatus.CONFLICT);
97  					return;
98  				}
99  				if (setter) {
100 					result.setCommand(ACTIONS_TYPE.SetBandwidth.name());
101 				} else {
102 					result.setCommand(ACTIONS_TYPE.GetBandwidth.name());
103 				}
104 				// request of current values or set new values
105 				long [] lresult = serverHandler.bandwidth(setter, 
106 						node.getWriteglobal(), node.getReadglobal(),
107 						node.getWritesession(),node.getReadsession());
108 				// Now answer
109 				node.setWriteglobal(lresult[0]);
110 				node.setReadglobal(lresult[1]);
111 				node.setWritesession(lresult[2]);
112 				node.setReadsession(lresult[3]);
113 				setOk(handler, result, json, HttpResponseStatus.OK);
114 			} else {
115 				logger.info("Validation is ignored: " + json);
116 				result.setDetail("Unknown command");
117 				setError(handler, result, json, HttpResponseStatus.PRECONDITION_FAILED);
118 			}
119 		} catch (OpenR66ProtocolNotAuthenticatedException e) {
120 			throw new HttpInvalidAuthenticationException(e);
121 		}
122 	}

124 	protected ArrayNode getDetailedAllow() {
125 		ArrayNode node = JsonHandler.createArrayNode();
126 		
127 		BandwidthJsonPacket node3 = new BandwidthJsonPacket();
128 		node3.setComment("Bandwidth getter (GET)");
129 		node3.setRequestUserPacket();
130 		ObjectNode node2;
131 		ArrayNode node1 = JsonHandler.createArrayNode();
132 		try {
133 			node1.add(node3.createObjectNode());
134 			if (this.methods.contains(METHOD.GET)) {
135 				node2 = RestArgument.fillDetailedAllow(METHOD.GET, this.path, ACTIONS_TYPE.GetBandwidth.name(), node3.createObjectNode(), node1);
136 				node.add(node2);
137 			}
138 		} catch (OpenR66ProtocolPacketException e1) {
139 		}
140 		
141 		if (this.methods.contains(METHOD.PUT)) {
142 			node3.setComment("Bandwidth setter (PUT)");
143 			try {
144 				node2 = RestArgument.fillDetailedAllow(METHOD.PUT, this.path, ACTIONS_TYPE.SetBandwidth.name(), node3.createObjectNode(), node1);
145 				node.add(node2);
146 			} catch (OpenR66ProtocolPacketException e1) {
147 			}
148 		}
149 		node2 = RestArgument.fillDetailedAllow(METHOD.OPTIONS, this.path, COMMAND_TYPE.OPTIONS.name(), null, null);
150 		node.add(node2);
151 		
152 		return node;
153 	}
