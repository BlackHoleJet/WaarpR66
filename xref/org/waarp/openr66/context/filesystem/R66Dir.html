57  	/**
58  	 * Same as setUnique() except that File will be prefixed by id and postfixed by filename
59  	 * 
60  	 * @param prefix
61  	 * @param filename
62  	 * @return the R66File with a unique filename and a temporary extension
63  	 * @throws CommandAbstractException
64  	 */
65  	public synchronized R66File setUniqueFile(long prefix, String filename)
66  			throws CommandAbstractException {
67  		checkIdentify();
68  		File file = null;
69  		String prename = prefix + "_";
70  		if (prename.length() < 3) {
71  			prename = "xx_" + prename;
72  		}
73  		String basename = R66File.getBasename(filename);
74  		if (basename.length() > Configuration.configuration.maxfilenamelength - 55) {
75  			basename = basename.substring(basename.length() - Configuration.configuration.maxfilenamelength + 55);
76  		}
77  		try {
78  			file = File.createTempFile(prename, "_" + basename +
79  					Configuration.EXT_R66, getFileFromPath(currentDir));
80  		} catch (IOException e) {
81  			throw new Reply550Exception("Cannot create unique file from " +
82  					basename);
83  		}
84  		String currentFile = getRelativePath(file);
85  		return newFile(normalizePath(currentFile), false);
86  	}

93  	public static String getFinalUniqueFilename(R66File file) {
94  		String finalpath = file.getBasename();
95  		int pos = finalpath.lastIndexOf(Configuration.EXT_R66);

97  			finalpath = finalpath.substring(0, pos);

102 	/**
103 	 * Finds all files matching a wildcard expression (based on '?', '~' or '*') but without
104 	 * checking BusinessPath, thus returning absolute path.
105 	 * 
106 	 * @param pathWithWildcard
107 	 *            The wildcard expression with a business path.
108 	 * @return List of String as relative paths matching the wildcard expression. Those files are
109 	 *         tested as valid from business point of view. If Wildcard support is not active, if
110 	 *         the path contains any wildcards, it will throw an error.
111 	 * @throws CommandAbstractException
112 	 */
113 	protected List<String> wildcardFilesNoCheck(String pathWithWildcard)
114 			throws CommandAbstractException {
115 		List<String> resultPaths = new ArrayList<String>();
116 		// First check if pathWithWildcard contains wildcards
117 		if (!(pathWithWildcard.contains("*") || pathWithWildcard.contains("?") || pathWithWildcard
118 				.contains("~"))) {
119 			// No so simply return the list containing this path
120 			resultPaths.add(pathWithWildcard);
121 			return resultPaths;
122 		}
123 		// Do we support Wildcard path
124 		if (!FilesystemBasedDirJdkAbstract.ueApacheCommonsIo) {
125 			throw new Reply553Exception("Wildcards in pathname is not allowed");
126 		}
127 		File wildcardFile = new File(pathWithWildcard);
128 		File rootFile;
129 		if (ISUNIX) {
130 			rootFile = new File("/");
131 		} else {
132 			rootFile = getCorrespondingRoot(wildcardFile);
133 		}
134 		// Split wildcard path into subdirectories.
135 		List<String> subdirs = new ArrayList<String>();
136 		while (wildcardFile != null) {
137 			File parent = wildcardFile.getParentFile();
138 			if (parent == null) {
139 				subdirs.add(0, wildcardFile.getPath());
140 				break;
141 			}
142 			subdirs.add(0, wildcardFile.getName());
143 			if (parent.equals(rootFile)) {
144 				// End of wildcard path
145 				subdirs.add(0, parent.getPath());
146 				break;
147 			}
148 			wildcardFile = parent;
149 		}
150 		List<File> basedPaths = new ArrayList<File>();
151 		// First set root
152 		basedPaths.add(new File(subdirs.get(0)));
153 		int i = 1;
154 		// For each wilcard subdirectory
155 		while (i < subdirs.size()) {
156 			// Set current filter
157 			FileFilter fileFilter = FilesystemBasedCommonsIo
158 					.getWildcardFileFilter(subdirs.get(i));
159 			List<File> newBasedPaths = new ArrayList<File>();
160 			// Look for matches in all the current search paths
161 			for (File dir : basedPaths) {
162 				if (dir.isDirectory()) {
163 					for (File match : dir.listFiles(fileFilter)) {
164 						newBasedPaths.add(match);
165 					}
166 				}
167 			}
168 			// base Search Path changes now
169 			basedPaths = newBasedPaths;
170 			i++;
171 		}
172 		// Valid each file first
173 		for (File file : basedPaths) {
174 			resultPaths.add(file.getAbsolutePath());
175 		}
176 		return resultPaths;
177 	}

179 	/**
180 	 * Create a new file according to the path without checking BusinessPath, so as external File.
181 	 * 
182 	 * @param path
183 	 * @return the File created
184 	 * @throws CommandAbstractException
185 	 */
186 	public R66File setFileNoCheck(String path)
187 			throws CommandAbstractException {
188 		checkIdentify();
189 		String newpath = consolidatePath(path);
190 		List<String> paths = wildcardFilesNoCheck(newpath);
191 		if (paths.size() != 1) {
192 			throw new Reply550Exception("File not found from: " + newpath +" and " +
193 					paths.size() + " founds");
194 		}
195 		String extDir = paths.get(0);
196 		return new R66File((R66Session) getSession(), this, extDir);
197 	}
